\begin{definition}

Sejam $G$ e $E$ conjuntos nĂŁo-vazios e $\oplus$ uma funĂ§ĂŁo tal que:

\[
\oplus: 
\begin{array}{r c l}
G\bigtimes G &\to &E \\
(a, b) &\mapsto &\oplus (a,b)
\end{array}
\]

Definimos a funĂ§ĂŁo acima como a \textbf{operaĂ§ĂŁo binĂˇria de dois elementos de $G$ em $E$} e a escrevemos comumente como: $a\oplus b$.
\end{definition}

\begin{example}
    A adiĂ§ĂŁo usual $+$ Ă© uma operaĂ§ĂŁo binĂˇria de dois elementos de $\mathbb{I}$ em $\mathbb{R}$. Onde $\mathbb{I}$ denota o conjunto dos nĂşmeros irracionais.

\begin{example}
    Sejam $(a,b)\in \mathbb{R}^2$, a funĂ§ĂŁo que define a distĂ˘ncia cartesiana entre dois pontos $a$ e $b$:
    \[
    \text{dist}(a,b): 
    \begin{array}{c}
    \mathbb{R}\bigtimes \mathbb{R} \to \mathbb{R}^+ \\
    (a, b) \mapsto \sqrt{a^2 + b^2}
    \end{array}
    \]
    representa uma operaĂ§ĂŁo binĂˇria de dois elementos de $\mathbb{R}$ em $\mathbb{R}^+$.
    
\end{example}

\begin{definition}

\end{definition}

\textit{ObservaĂ§ĂŁo: caso nĂŁo haja ambiguidade, denotaremos simplesmente \textbf{lei de composiĂ§ĂŁo interna em $G$} para representar a lei de composiĂ§ĂŁo interna de $G\times G \to G$.}

\section{Grupos}

\begin{definition}
    Seja $G$ um conjunto nĂŁo-vazio. \textbf{Dizemos que $(G, \cdot)$ Ă© um grupo} se, e somente se, $\cdot$ Ă© uma lei de composiĂ§ĂŁo interna em $G$ tal que:
    \begin{enumerate}
    \item $\exists \,e \in G \; , \forall \,x \in G \; : x\cdot e = e\cdot x = x $;
    \item $\forall x \in G, \exists \hat{x}\in G: \; x\cdot \hat{x} = \hat{x} \cdot x = e$;
    \item $\forall \, x,y,z \in G : (x\cdot y)\cdot z = x\cdot (y\cdot z)$.
    \end{enumerate}
\end{definition}

\textit{ObservaĂ§Ăµes:}

\textit{Levando em consideraĂ§ĂŁo as notaĂ§Ăµes acima, temos:}
\begin{enumerate}
	\item \textit{Primeiramente, notamos que $e$ e $\hat{x}$ sĂŁo Ăşnicos, uma vez que:}

    \textit{Supondo que existam $e$ e $e'$ pertencentes Ă  $G$ que satisfazem o item 1, temos:}
    \[
        x \cdot e = x = x \cdot e' 
        \;\;\;\implies\;\;\;
        \hat{x}\cdot x \cdot e 
        = \hat{x} \cdot x \cdot e' 
        \;\;\;\implies\;\;\;
        e = e' 
        \quad \qedsymbol
    \]

    \textit{Supondo agora que existam $\hat{x}$ e $\hat{x}'$ que satisfaĂ§am o item 2, temos:}
    \[
        \hat{x}\cdot x 
        = e 
        = \hat{x}'\cdot x 
        \implies
        \hat{x}\cdot x \cdot \hat{x} 
        = \hat{x}'\cdot x \cdot \hat{x} 
        \implies
        \hat{x}\cdot e 
        = \hat{x}'\cdot e 
        \implies
        \hat{x} = \hat{x}' 
        \quad \qedsymbol
    \]
    \item \textit{Notamos por convenĂ§ĂŁo $x^{-1}$ no lugar de $\hat{x}$ no \textbf{item 2} (dada sua unicidade).}
    
    \item \textit{Caso $\forall \,(x,y) \in G \times G:\; x \cdot y = y \cdot x$, dizemos que $G$ Ă© um grupo \emph{abeliano} (ou \emph{comutativo}).}
    
    \item \textit{Caso $G$ seja um grupo abeliano, entĂŁo} 
    \[
      (x\cdot y)^n = x^n \cdot y^n, 
      \quad \forall n \in \mathbb{Z}.
    \]
\end{enumerate}

\[
\odot: 
\begin{array}{c}
\mathbb{Z}_n \bigtimes \mathbb{Z}_n \to \mathbb{Z}_n \\

\textbf{DemonstraĂ§ĂŁo:}

Temos que:

\begin{align*}
	&\overline{a}\odot (\overline{b}\odot \overline{c}) = \overline{a}\odot (\overline{b\cdot c}) =\\[10pt]
	&= \overline{a\cdot (b\cdot c)} = \overline{(a\cdot b)\cdot c} = (\overline{a}\odot \overline{b})\odot \overline{c}.
\end{align*}

\[
\overline{a}\odot \overline{1} = \overline{a\cdot 1} = \overline{a}, \; \forall \overline{a} \in \mathbb{Z}_{p}^{*}.
\]

TambĂ©m temos que para todo elemento de $\mathbb{Z}_{p}^{*}$, existe elemento inverso, pois, sabemos que:

\[
\forall \overline{a}\in \mathbb{Z}_{p}^{*} \implies \text{mdc}(a, p) = 1.
\]

\[

\]

Ora mas isso Ă© a mesma coisa que afirmar que existe uma soluĂ§ĂŁo para a equaĂ§ĂŁo: 

\[
a \cdot x \equiv 1 \pmod{p} \iff \overline{a} \odot \overline{x} = \overline{1}. 
\]

Logo, deduzimos que $\forall \overline{a} \in \mathbb{Z}_{p}^{*}, \; \exists \overline{a}^{-1} \in \mathbb{Z}_{p}^{*}$.

AlĂ©m disso, Ă© evidente que a operaĂ§ĂŁo $\odot$ Ă© comutativa.

\qed


\begin{example}
     Seja $G = \;]-1, 1[, \; (G, \star)$ tal que 
     \[
     \forall x,y \in G \; : x\star y = \frac{x + y}{1 + xy}
     \]
     define um grupo abeliano.
\end{example}

\textbf{DemonstraĂ§ĂŁo:}

Provemos primeiramente que $\forall x, y \in G, x\star y \in G$.

Fixando $y \in G$ temos a seguinte funĂ§ĂŁo de $x \in G$:

\[
f(x) = \frac{x + y}{1 + xy}
\]

A funĂ§ĂŁo Ă© derivĂˇvel em $G$. Tomando sua derivada temos:

\[
f'(x) = \frac{1 - y^2}{(1+xy)^2} 
\]

Temos evidentemente $\forall (x, y) \in G \times G, f'(x) > 0$.

(De forma simĂ©trica podemos mostrar o mesmo escrevendo $f$ como uma funĂ§ĂŁo de $y$).

Logo, deduzimos que a funĂ§ĂŁo $f$ Ă© estritamente crescente.

Portanto:

\[
f(-1)<x \star y < f(1) \iff \frac{y-1}{1-y} < x \star y < \frac{1+y}{1+y} \iff -1 <x\star y < 1
\]

Logo, provamos que $x\star y \in G$.

Provemos os outros axiomas:

\textit{ExistĂŞncia do neutro}:

Tomando $y = 0$ temos:

\[
x\star 0 = \frac{x + 0}{1 + 0\cdot x} = x
\]

Portanto, deduzimos que o elemento neutro do grupo $G$ Ă© dado por $e = 0$.

\textit{ExistĂŞncia do inverso}:

Tomando $y = -x$ temos:

\[
x\star -x = \frac{x -x}{1 - (-x)x} = 0
\]

Portanto, deduzimos que o elemento inverso do grupo $G$ existe e Ă© dado por $x^{-1} = -x$.

\textit{Associatividade}:

Sejam $x, y, z \in G$, mostremos que $(x\star y)\star z = x\star (y\star z)$

Temos:

\begin{align*}
(x\star y)\star z &= \frac{(x\star y) + z}{1+(x\star y)z} = \frac{\frac{x+y}{1+xy} + z}{1 + z\frac{x+y}{1 +xy}} = \frac{x + y+z+xyz}{1+xy+xz+yz} = \\[10pt]
&= \frac{x(1+yz) + (y +z)}{(1+yz)+x(y+z)} = \frac{x+\frac{y+z}{1+yz}}{1+x\frac{y+z}{1+yz}} = x\star (y\star z)
\end{align*}

Mostrando, assim, a associatividade.

Ainda, temos que o grupo Ă© evidentemente abeliano.\qed

\section{Subgrupos}

\begin{definition}
    Seja $(G, \cdot)$ um grupo. \textbf{Um subconjunto $H \subseteq G$ Ă© chamado de subgrupo de $G$} (denotamos $H\leq G$) se, e somente se, $(H, \cdot)$ Ă© um grupo.
\end{definition}
\textit{ObservaĂ§ĂŁo: temos ainda que se $H \subset G$, temos entĂŁo $H$ Ă© chamado de subgrupo prĂłprio de $G$ e denotamos como $H< G$.}

\begin{proposition} \hypertarget{prop131}
    Seja $H \subseteq G$ tal que $H \ne \emptyset$ e $(G, \cdot)$ Ă© um grupo. $H\leq G$ Ă© equivalente Ă  satisfazer as seguintes condiĂ§Ăµes:
    \begin{enumerate}
    \item $h_1 \cdot h_2 \in H,\; \forall (h_1,h_2) \in H \times H$;
    \item $h^{-1} \in H, \; \forall h\in H$.
    \end{enumerate}
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Ă‰ necessĂˇrio mostrarmos as duas implicaĂ§Ăµes da equivalĂŞncia:

\begin{align}
    H\leq G \Longrightarrow (1.)\; \text{e} \;(2.)
\end{align}

\begin{align}
    (1.)\; \text{e} \;(2.) \Longrightarrow H\leq G
\end{align}

A implicaĂ§ĂŁo $(1.1)$ Ă© trivial. Ora, se $H\leq G$, entĂŁo pela definiĂ§ĂŁo de subgrupo temos que $h_1\cdot h_2 \in H$ e $h^{-1}\in H$, isto Ă© $\exists h^{-1} \in H : h\cdot h^{-1} = h^{-1} \cdot h = h$.

Para a implicaĂ§ĂŁo $(1.2)$:

Sabemos que $H \subseteq G$, logo, se $h_1 \cdot h_2 \in H \Longrightarrow h_1 \cdot h_2 \in G$. Ora, sabemos que $(G, \cdot)$ Ă© um grupo. Logo, a associatividade Ă© satisfeita. Para demonstrar que $e \in H$, basta tomarmos $h_2 = h^{-1}$ a partir de $(2.)$. Logo, temos $h\cdot h^{-1} = e \in H$. Com isso mostramos todos os axiomas necessĂˇrios e deduzimos que $H\leq G$. \qed

PorĂ©m, sabemos que:

\[
\forall i \in I, \; H_i \leq G
\]

Logo, deduzimos que:

\[
\forall i \in I, \; x_0 \in H_i, \; \exists x^{-1}_{0} \in H_i   
\]

Mas isso Ă© a mesma coisa que:

\[
\forall x \in \bigcap_{i \in I} H_i \implies \exists  x^{-1} \in \bigcap_{i \in I} H_i
\]

Portanto, provamos que:

\[
\bigcap_{i \in I} H_i \leq G
\]  

\qed 

\begin{definition}
    Seja $G$ um grupo. O subconjunto $Z(G)$ tal que: 
    \[
    Z(G) = \{x \in G : xg = gx, \; \forall g \in G\}
    \]
    define um subgrupo de $G$ chamado \textit{centro} de $G$.
\end{definition}


\textbf{DemonstraĂ§ĂŁo:}

Como visto na \hyperlink{prop131}{\textbf{ProposiĂ§ĂŁo 1.3.1}}, para mostrar que $Z(G) \leq G$ Ă© necessĂˇrio mostrar que $x\cdot x^{-1} \in Z(G), \; \forall x \in Z(G)$.

Nota: $Z(G)$ Ă© claramente nĂŁo vazio uma vez que o elemento neutro comuta com todos elementos de $G$ e, portanto, estĂˇ em $Z(G)$.

Temos que:

Se:
\[
x \in Z(G) \Rightarrow x\cdot g = g\cdot x, \; \forall g\in G.
\]

Logo, teremos:

\[
xgx^{-1} = g \Longrightarrow x^{-1}xgx^{-1} = x^{-1}g \Longrightarrow gx^{-1} =x^{-1}g, \; \forall g \in G 
\]

Portanto: 

\[
x^{-1} \in Z(G)
\]

Temos tambĂ©m que:

\[
x_1 \in Z(G) \Longrightarrow x_1g = gx_1, \; \forall g \in G \;\; \text{(I)}
\]

\[
x_2 \in Z(G) \Longrightarrow x_2g = gx_2, \; \forall g \in G \; \; \text{(II)}
\]

Deduzimos de (I):

\[
x_1g = gx_1 \Longrightarrow g = x^{-1}_1gx_1
\]

Substituindo em (II):

\[
x_2x^{-1}_1gx_1 = x^{-1}_1gx_1x_2 \Longrightarrow x_2x^{-1}_1x_1g = x^{-1}_1gx_1x_2 \Longrightarrow 
\]

\[
\Longrightarrow x_2g = x^{-1}_1gx_1x_2 \Longrightarrow (x_1x_2)g = g(x_1x_2)
\]

Logo, deduzimos que:

\[
(x_1, x_2)\in Z(G)\times Z(G) \Longrightarrow x_1\cdot x_2 \in Z(G)
\]

Portanto, $Z(G) \leq G$. \qed

\textit{ObservaĂ§ĂŁo: O subgrupo centro serve o propĂłsito de "medir a comutatividade" de um dado grupo. Por exemplo, observamos que $Z(\mathbb{Z}) = \mathbb{Z}$, $Z(GL_2 (\mathbb{R})) = \{\lambda I: \lambda \in \mathbb{R}^*\}$ e  $Z(S_n) = \{e\},\; n \geq 3$}.

\begin{definition} \hypertarget{def133}
    Seja $(G, \cdot)$ um grupo e $X$ um conjunto nĂŁo-vazio tal que $X\subseteq G$. \textbf{Chamamos de subgrupo gerado por um subconjunto a interseĂ§ĂŁo de todos os subgrupos de $G$ que contĂ©m $X$.}  Denotamos-o como $\langle X\rangle$.

\]

\end{definition}

\begin{proposition}

\end{proposition}

\textbf{DemonstraĂ§ĂŁo}:

Suponha que $J \leq G$ seja o menor subgrupo de $G$ tal que $X \subseteq J$.

Ora, como $J \leq G$ e $X \subseteq J$, entĂŁo: $\langle X \rangle \subseteq J$.

Entretanto, tambĂ©m sabemos que $J$ Ă© o menor subgrupo de $G$ tal que $X \subseteq J$.

Portanto, $J = \langle X \rangle$. 

\qed  


\begin{proposition}

\]

\bar{x}_{p}^{-1} \overset{def}{=} x_{1}^{-1} x_{2}^{-1} ... x_{p}^{-1}, \; p \in \mathbb{Z}_{+}^{*}

Ă‰ evidente que $\bar{x}_p,\; \bar{x}_{p}^{-1} \in \bar{X}$. Assim como $\bar{x}_p \bar{x}_{p}^{-1} \in \bar{X}$, o que nos mostra que $\bar{X} \leq G$.

Onde $I$ Ă© um conjunto nĂŁo-vazio de Ă­ndices.

Evidentemente temos entĂŁo que $\bar{x}_p \in \dot{X}$.

\qed

\begin{proposition} \hypertarget{prop132}
    Se $G$ Ă© um grupo cĂ­clico, entĂŁo $G$ Ă© um grupo abeliano.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Seja $a \in G$ tal que $G = \langle a\rangle$. Podemos representar $G$ como:

\[
G = \{...,\; (a^{-1})^{r},\; ...,\; (a^{-1})^2, \; a^{-1},\; e, \; a,\; a^2, \; ..., a^r, \; ...\} 
\]

Onde $r \in \mathbb{Z}$.

Sejam $(x,y)\in G \times G$, queremos mostrar que $x\cdot y = y\cdot x$.

Sabemos que:

\[
x = a^{r_1}, \; r_1 \in \mathbb{Z}
\]

\[
y = a^{r_2}, \; r_2 \in \mathbb{Z}
\]

Logo:

\[
x\cdot y = a^{r_1} \cdot a^{r_2} = a^{r_1 + r_2} \stackrel{(*)}{=} a^{r_2+ r_1} = a^{r_2}\cdot a^{r_1} = y\cdot x
\]

\textit{$(*)$ : deduz-se que $r_1 + r_2 = r_2 + r_1$ pois estamos trabalhando dentro do grupo abeliano $(\mathbb{Z}, +)$}.

Portanto, $G$ Ă© um grupo abeliano. \qed

\begin{proposition} \hypertarget{prop136}
Seja $G$ um grupo e $\alpha$ um elemento de $G$. EntĂŁo, as seguintes proposiĂ§Ăµes sĂŁo equivalentes:

$(i)$ A ordem $\mathcal{O}(\alpha)$ Ă© finita. Isto Ă©, $\mathcal{O}(\alpha) < \infty$;

$(ii)$ $\exists t \in \mathbb{Z}_{+}^{*} : \alpha ^t = e$, onde $t = \min{\{k \in G : k > 0\}}$.

Note que, caso a proposiĂ§ĂŁo seja satisfeita, entĂŁo temos $\mathcal{O}(\alpha) = t$.

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Queremos provar que: $(i) \iff (ii)$.

Comecemos provando a implicaĂ§ĂŁo $(i) \Longrightarrow (ii):$

Temos, por definiĂ§ĂŁo, que $\langle \alpha \rangle = \{\alpha ^m \; \vert \; m \in \mathbb{Z}\}$. 

Como $\mathcal{O}(\alpha) < \infty$, temos que $\exists p, q \in \mathbb{Z} : p > q$ e $\alpha ^p = \alpha ^q$.

Deduzimos diretamente que: $\alpha ^{p-q} = e$. Como $p-q \in \mathbb{Z}^{*}_{+}$, mostramos $(i) \Longrightarrow (ii)$.

Note que a escolha do valor $p-q$ ocorre sem perda de generalidade, uma vez que o conjunto $\mathbb{Z}_{+}^{*}$ Ă© enumerĂˇvel e sempre podemos garantir a minimalidade de $p-q$. 

Provemos $(ii) \Longrightarrow (i):$

Seja $\alpha \in G$ tal que $\langle \alpha \rangle = \{ \alpha ^n : n\in\mathbb{Z}\}$, sabemos via o Algoritmo de Euclides que: 

\[
n = qt + r, \; q,t\in \mathbb{Z} \; \; \text{e} \; \; r\in \llbracket 0, t-1 \rrbracket
\]

Temos, portanto:

\[
\alpha ^n = \alpha ^{qt + r} = \alpha ^{qt} \alpha ^{r} = (\alpha ^{t})^{q} \alpha ^{r} = e^q \alpha ^r = \alpha ^r
\]

Portanto $\alpha ^n = \alpha ^r, \; \forall n\in \mathbb{Z}$ e $r\in \llbracket 0, t-1\rrbracket$.

Logo, deduzimos que:

\[
\langle \alpha \rangle = \{\alpha^0, \ldots, \alpha ^{t-1} \} \Longrightarrow \mathcal{O}(\alpha) < \infty
\]

Para mostrar que de fato temos $\mathcal{O}(\alpha) = t$, mostremos que $\alpha ^{r_1} \neq \alpha ^{r_2}, \; \forall r_1,r_2 \in \llbracket 0,t-1 \rrbracket$.

Suponha, por absurdo, que $\exists r_1, r_2 \in \llbracket 0,t-1 \rrbracket$ tal que $\alpha ^{r_1} = \alpha ^{r_2}$.

Temos, portanto:

\[
\alpha ^{r_1} = \alpha ^{r_2} \iff \alpha ^{r_1 - r_2} = e
\]

Absurdo, pois $r_1 - r_2 < t$, porĂ©m $t$ Ă© minimal.

Logo, como todos elementos de $\langle \alpha \rangle$ sĂŁo distintos, deduzimos que $\mathcal{O}(\alpha) = t$.

\qed

\section{Teorema de Lagrange}

\begin{definition}
    Seja $G$ um grupo e $H$ um subgrupo de $G$. \textbf{Definimos classe lateral Ă  esquerda de $H$ em $G$ que contĂ©m $x$ o subconjunto $xH$} de $G$ tal que $\forall x \in G$:
    \[
    xH = \{xh \; \vert \; h\in H \}
    \]
    Analogamente definimos \textbf{classe lateral Ă  direita de $H$ em $G$ que contĂ©m $x$ o subconjunto $Hx$} de $G$ tal que $\forall x \in G$:
    \[
    Hx = \{ hx \; \vert \; h \in H\}
    \]
\end{definition}


\textit{ObservaĂ§Ăµes:}
    \begin{itemize}
        \item \textit{As classes laterais de $G$ nĂŁo sĂŁo necessariamente subgrupos de $G$;}
        \item \textit{Quando nĂŁo houver confusĂŁo possĂ­vel, podemos denominar as classes laterais Ă  esquerda/direita de $H$ em $G$ que contĂ©m $x$ como simplesmente: classe lateral Ă  esquerda/direita de $H$}.
    \end{itemize}

\begin{definition} \hypertarget{def142}
	A cardinalidade do conjunto das classes laterais Ă  esquerda \textit{ou} Ă  direita Ă© definida como \textbf{o Ă­ndice de $H$ em $G$}, e serĂˇ denotada por $[G:H]$.
\end{definition} 

\textit{ObservaĂ§ĂŁo: note que o nĂşmero de classes laterais Ă  direita de $H$ Ă© igual ao nĂşmero de classes laterais Ă  esquerda de $H$ (por mais que as classes laterais sejam diferentes).} 

\textit{Isto se dĂˇ pelo fato de que a funĂ§ĂŁo:} 

\[
\phi: 
\begin{array}{c}
\{\text{classes lat. Ă  esquerda}\} \to \{\text{classes lat. Ă  direita} \} \\
xH \mapsto Hx^{-1}
\end{array}
\]

\textit{Ă© claramente uma bijeĂ§ĂŁo.}

\begin{theorem} \hypertarget{lagrange}
{\bfseries Teorema de Lagrange (Grupos)}\normalfont

    Seja $G$ um grupo finito e $H$ um subgrupo de $G$.

    Logo, $\vert H\vert$ divide $\vert G \vert$.
\end{theorem}

\textbf{DemonstraĂ§ĂŁo:}


%A ideia da demonstraĂ§ĂŁo se baseia em, a partir de um subgrupo $H$ de $G$, formar partiĂ§Ăµes de $G$ duas-a-duas disjuntas tais que a cardinalidade dessas partiĂ§Ăµes sejam iguais Ă  ordem de $H$.

%Isto Ă©, se $\{X_i \}_{i\in\mathbb{N}}$ representa uma famĂ­lia de subconjuntos de $G$ e $H$ Ă© um subgrupo de $G$, entĂŁo queremos provar que:

%\[
%\bigcup_{i \in \mathbb{N}}X_i = G; \; \; \bigcap_{i \in \mathbb{N}}X_i = \emptyset\; \; \text{e} \; \; \text{Card}(X_i) = \vert H\vert, \; \forall i \in \mathbb{N}.
%\]

Seja $x\in G \backslash H$, consideremos o conjunto das classes laterais Ă  esquerda de $H$:

\[
xH = \{xh \; \vert \; h \in H\}
\]

Mostremos que $H\cap xH = \emptyset$:

Supondo $\alpha \in H\cap xH$:

\[
\alpha \in H\cap xH \iff \alpha = xh \in H.
\]

Como $\alpha =xh \in H$, logo $\exists h^{-1}\in H$ tal que $hh^{-1} \in H$

Portanto:

\[
\alpha h^{-1} = xhh^{-1} \in H \iff x \in H \Longrightarrow \text{Absurdo, pois   } x\in G\backslash H.
\]

Logo, $H\cap xH = \emptyset$.

Agora mostremos que $\text{Card}(xH) = \vert H\vert$:

Seja $\zeta$ a funĂ§ĂŁo definida abaixo:

\[
\zeta: 
\begin{array}{c}
H \to xH \\
h \mapsto xh
\end{array}
\]

A funĂ§ĂŁo $\zeta$ Ă© claramente sobrejetiva por definiĂ§ĂŁo. 

$\zeta$ tambĂ©m Ă© injetiva pois se $(xh_1, xh_2) \in (xH)^2$:

\[
xh_1 = xh_2 \Longrightarrow x^{-1}xh_1 = x^{-1}xh_2 \Longrightarrow h_1 = h_2.
\]

Portanto, deduzimos que $\text{Card}(xH) = \vert H\vert$.

Consideremos agora o conjunto $yH$ das classes laterais Ă  esquerda de $H$ em $G$ que contĂ©m $y$ tal que $y \notin H\cup xH$.

JĂˇ mostramos anteriormente que $y \notin H$.

Mostremos que $yH \cap xH = \emptyset$

Supondo $\beta \in yH \cap xH$:

EntĂŁo $\beta$ pode ser escrito de duas formas:

\[
\beta = yh_1
\]

\[
\beta = xh_2
\]

Logo, temos:

\[
yh_1 = xh_2 \Longrightarrow y = xh_2h_1^{-1} \in xH \Longrightarrow \text{Absurdo, pois  } y \notin H\cup xH.
\]

Analogamente ao passo anterior podemos provar que $\text{Card}(yH) = \text{Card}(xH) = \vert H\vert$.

Portanto, realizando os passos acima sucessivamente, criamos partiĂ§Ăµes de $G$.

Como $G$ Ă© finito, o processo terĂˇ finalizado apĂłs $n$ etapas.

Portanto, temos: $\vert G\vert = n\vert H\vert$. \qed

\textit{ObservaĂ§Ăµes:} 

\begin{enumerate}
	\item \textit{Segue como consequĂŞncia direta do \textbf{Teorema de Lagrange} que caso $G$ seja um grupo finito e $\alpha \in G$, entĂŁo $\mathcal{O}(\alpha)$ divide $\vert G\vert$.}
	\item \textit{Temos diretamente pela \hyperlink{def142}{\textbf{DefiniĂ§ĂŁo 1.4.2}} que: $\vert G\vert = \vert H\vert [G:H]$.}
\end{enumerate}

\begin{corollary}
	Seja $G$ um grupo nĂŁo finito e $H \leq G$. 

	EntĂŁo vale o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}}. 
\end{corollary}

\textbf{DemonstraĂ§ĂŁo:}

Demonstraremos novamente o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} de forma que o corolĂˇrio acima possa ser justificado de forma clara. 

Seja $G$ um grupo e $H \leq G$.

Ora, sabemos que:

\[
\text{Ou} \; \; xH = yH \; \; \text{ou} \; \; xH \cap yH = \emptyset, \; \forall (x,y) \in G \times G.
\]

Sabemos tambĂ©m que, sendo $I$ um conjunto nĂŁo vazio de Ă­ndices tal que Card($I$) $= [G:H]$:

\[
\bigsqcup_{i \in I} x_i H = G.
\]

Como $\vert G\vert$ Ă© nĂŁo finito, entĂŁo se $\vert H\vert$ ou $[G:H]$ sĂŁo nĂŁo finitos, vale que $\vert G\vert = \vert H\vert[G:H]$.

Suponhamos, agora, que $\vert H\vert < \infty$ e $[G:H] < \infty$.

Como $[G:H] < \infty$, entĂŁo $\vert I\vert < \infty$.

Logo, podemos escrever $I$ como:

\[
I = \{i_1, i_2, \dots, i_n, \; n\in \mathbb{N} \}.
\]

Logo:

\[
G = \bigsqcup_{i_1 \leq i \leq n} x_i H.
\]

Portanto, podemos escrever:

\[
\vert G\vert = \sum_{k = 1}^{n} \vert x_{i_k} H\vert.
\]

Ora, deduzimos na demonstraĂ§ĂŁo do \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} que $\vert xH\vert = \vert H\vert, \; \forall x \in G$.

Portanto temos que:

\[
\vert G\vert = n\vert H\vert.
\]

Ora, mas $\vert G\vert$ Ă© nĂŁo finito e $\vert H\vert < \infty$. Absurdo !

Portanto, deduzimos que $\vert H\vert$ ou $[G:H]$ sĂŁo nĂŁo finitos.

Assim, provamos que o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} vale tambĂ©m para $\vert G\vert$ nĂŁo finito. Isto Ă©:

\[
\vert G\vert = \vert H\vert [G:H].
\]

\qed

\begin{proposition}

    Se $p$ for primo, entĂŁo $G$ Ă© um grupo cĂ­clico.
\end{proposition}

\begin{theorem} 
	\textbf{Teorema de Euler (Grupos)} \\
	Seja $(G, \cdot)$ um grupo finito tal que $\vert G \vert = n, \; n\in \mathbb{Z}$.
	EntĂŁo:
	\[
	\forall g \in G, \; g^n = 1.
	\]
\end{theorem}

\textbf{DemonstraĂ§ĂŁo:}

Seja $g$ um elemento do grupo finito $G$. Sabemos que $\langle g \rangle \leq G$. Sabemos tambĂ©m, pelo \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} que $\mathcal{O}(g)$ divide a ordem de $G$. 

Ora, podemos entĂŁo escrever:

\[
\vert G\vert = k\mathcal{O}(g), \; k\in \mathbb{Z} 
\]

PorĂ©m, pela \hyperlink{prop136}{\textbf{ProposiĂ§ĂŁo 1.3.6}}, deduzimos:

\[
g^{n} = g^{\vert G \vert} = g^{k\mathcal{O}(g)} = \left(g^{\mathcal{O}(g)}\right)^{k} = e^k = e
\]

Ora, demonstramos, com o argumento acima, sem perda de generalidade, tal fato para qualquer elemento de $G$.

\qed

\begin{proposition}
	Seja $G$ um grupo e sejam $K < H < G$.

	Logo $[G : K] = [G : H][H : K].$
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Basta aplicar sucessivamente o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}}:

\begin{align*}
& H < G &&\Rightarrow && \lvert G \rvert = \lvert H \rvert \cdot [G : H] \; \; \; \text{(I)} \\
& K < H &&\Rightarrow && \lvert H \rvert = \lvert K \rvert \cdot [H : K] \; \; \; \text{(II)} \\
& K < G &&\Rightarrow && \lvert G \rvert = \lvert K \rvert \cdot [G : K] \; \; \; \text{(III)}
\end{align*}

Combinando as expressĂµes (I) e (II), obtemos:

\[
\lvert G \rvert = \lvert K \rvert \cdot [H : K] \cdot [G : H] = \lvert K \rvert \cdot [G : K]
\]

Portanto:

\[
[G : K] = [G : H] \cdot [H : K]
\]

\qed

\begin{proposition}
Seja $G$ um grupo e $H\leq G$. 

As afirmaĂ§Ăµes a seguir sĂŁo equivalentes:

$(i)$ A operaĂ§ĂŁo definida na \hyperlink{def152}{\textbf{DefiniĂ§ĂŁo 1.5.2}} estĂˇ bem definida;

$(ii)$ $gHg^{-1} \subseteq H, \; \forall g \in G$;

$(iii)$ $gHg^{-1} = H, \; \forall g \in G$;

$(iv)$ $gH = Hg, \; \forall g \in G$.

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos as equivalĂŞncias:

\[
(i) \iff (ii) \; \; \; \text{(I)}
\]

\[
(ii) \iff (iii) \; \; \; \text{(II)}
\]

\[
(iii) \iff (iv) \; \; \; \text{(III)}
\]

Comecemos mostrando a equivalĂŞncia (I):

Ora, perceba que para $(x,y) \in G\times G$ e $(h, h')\in H\times H$, temos que:

$x$ e $xh$ sĂŁo representantes distintos para a mesma classe lateral $xH$.

$y$ e $yh'$ sĂŁo representantes distintos para a mesma classe lateral $yH$.

Portanto, podemos deduzir que a operaĂ§ĂŁo "$\bullet$" definida na \hyperlink{def152}{\textbf{DefiniĂ§ĂŁo 1.5.2}} sĂł estarĂˇ bem definida se, e somente se:

\[
xyH = xhyh'H, \; \forall (x,y)\in G\times G \; \; \; \text{e} \; \; \; \forall (h,h')\in H\times H.
\]

Logo:

\[
xyH = xhyh'H \iff y^{-1}x^{-1}xyH = y^{-1}x^{-1}xhyh'H \iff H = y^{-1}hyh'H.
\]

Ora, mas isso Ă© equivalente Ă  dizer que a operaĂ§ĂŁo sĂł estarĂˇ bem definida se, e somente se:

\[
ghg^{-1} \in H,\; \forall g\in G, \; \forall h\in H.
\]

Com isso mostramos a equivalĂŞncia (I).

Mostremos a equivalĂŞncia (II):

Para a implicaĂ§ĂŁo $(ii) \implies (iii)$ mostremos que:

\[
gHg^{-1} \subseteq H \implies H \subseteq gHg^{-1}, \; \forall g \in G.
\]

Ora, temos diretamente da hipĂłtese:

\[
gHg^{-1} \subseteq H \implies g^{-1}Hg \subseteq H \implies g(g^{-1}Hg)g^{-1} \subseteq gHg^{-1}
\]

\[
\implies H \subseteq gHg^{-1}
\]

Logo, podemos concluir que:

\[
gHg^{-1} \subseteq H \implies H = gHg^{-1}.
\]

A implicaĂ§ĂŁo $(iii) \implies (ii)$ Ă© evidente.

Com isso mostramos a equivalĂŞncia (II).

Mostremos a equivalĂŞncia (III):

\[
gHg^{-1} = H \iff gHg^{-1}g = Hg \iff gH = Hg.
\]

Assim mostramos a equivalĂŞncia (III).

Tendo mostrado as equivalĂŞncias (I), (II) e (III), mostramos que todas as afirmaĂ§Ăµes sĂŁo duas a duas equivalentes. 

\qed

\[

Logo, deduzimos que:

\qed

\begin{definition}
Sejam $(G, \cdot)$ e $(\mathcal{G}, *)$ dois grupos. \textbf{Uma funĂ§ĂŁo $\varphi : G \rightarrow \mathcal{G}$ Ă© chamada de homomorfismo de grupos se:}

\[
\varphi(a\cdot b) = \varphi(a)*\varphi(b), \; \forall a,b \in G

\end{definition}

\textit{ObservaĂ§Ăµes:}

\textit{Note que, se $\varphi : G \rightarrow \mathcal{G}$} for um homomorfismo de grupos, entĂŁo:

\begin{enumerate}
	\item $\varphi(e_G) = e_\mathcal{G}$;
	\item $\varphi(x^{-1}) = \varphi(x)^{-1}.$
\end{enumerate}

\textit{De fato, note que, para o item 1., temos que}:

\[
\varphi(e_G) = \varphi(e_G \cdot e_G) = \varphi(e_G)*\varphi(e_G) \implies \varphi(e_G) = e_\mathcal{G}
\]

\textit{E para o item 2., temos:}

\[
e_\mathcal{G} = \varphi(e_G) = \varphi(x\cdot x^{-1}) = \varphi(x)*\varphi(x^{-1}) \implies \varphi(x^{-1}) = \varphi(x)^{-1}
\]

\textit{$*$ Note tambĂ©m que muitas vezes as operaĂ§Ăµes de $G$ e de $\mathcal{G}$ serĂŁo confundidas para fins de simplificaĂ§ĂŁo, isto Ă©: $\varphi(ab) = \varphi(a)\varphi(b)$.}

\end{definition}

\end{example}

\end{example}

Portanto, deduzimos que:

\]

\qed

\end{array}
\]

Ă© um isomorfismo de grupos.

Isto Ă©:

\[
\frac{G}{\ker \varphi} \cong \operatorname{Im} \varphi
\]
\end{theorem}

\textbf{DemonstraĂ§ĂŁo}:

Mostremos que $\psi$ Ă© uma funĂ§ĂŁo bem definida e que se trata de uma funĂ§ĂŁo injetiva:

Sejam $\bar{x},\bar{y}\in G/\ker\varphi$, temos:

\[
\bar{x} = \bar{y} \iff xy^{-1} \in \ker\varphi \iff \varphi(xy^{-1}) = e_{\mathcal{G}} \iff
\]

\[
\iff \varphi(x)\varphi(y)^{-1} = e_{\mathcal{G}} \iff \varphi(x) = \varphi(y) \iff \psi(\bar{x}) = \psi(\bar{y})
\]

Logo, a funĂ§ĂŁo $\psi$ estĂˇ de fato bem definida e Ă© injetiva.

Mostremos que $\psi$ Ă© uma funĂ§ĂŁo sobrejetora. 

Seja $y\in \operatorname{Im}\varphi$, entĂŁo:

\[
\exists x\in G ;\; \varphi(x) = y
\]

Temos, portanto:

\[
\psi(\bar{x}) = \varphi(x) = y
\] 

Logo, $\psi$ Ă© sobrejetora.

Mostremos que $\psi$ Ă© um homomorfismo de grupos.

Primeiramente, note que $G/\ker\varphi$ e $\operatorname{Im}\varphi$ sĂŁo de fato grupos (jĂˇ demonstrado).

Sejam $\bar{x},\bar{y} \in G/\ker\varphi$, logo:

\[
\psi(\bar{x}\bar{y}) = \psi(\overline{xy}) = \varphi(xy) = \varphi(x)\varphi(y) = \psi(\bar{x})\psi(\bar{y})
\]

Com isso, mostramos que $\psi$ se trata se um isomorfismo de grupos e deduzimos que:

\[
\frac{G}{\ker\varphi} \cong \operatorname{Im}\varphi
\]

\qed

\begin{theorem}
\textbf{Segundo Teorema dos Isomorfismos}

Seja $G$ um grupo,\; $H\trianglelefteq G$ e $K\leq G$. Temos:

\[
H \trianglelefteq HK
\]

e,

\[
\frac{K}{H\cap K} \cong \frac{HK}{H}
\]
\end{theorem}

\textbf{DemonstraĂ§ĂŁo}:

Note que $HK$ Ă© de fato um grupo, pois $H\trianglelefteq G$ e $K\leq G$.

Tal fato foi mostrado na \hyperlink{prop154}{\textbf{ProposiĂ§ĂŁo 1.5.4}}. 

Mostremos, primeiramente, que $H \trianglelefteq HK$:

Seja $x\in HK$, logo, Ă© suficiente mostrar que $xHx^{-1} \subseteq H$.

Como $H\trianglelefteq G$, temos que:

\[
gH = Hg, \; \; \forall g\in G
\]

Portanto:

\[
xHx^{-1} = Hxx^{-1} = H
\]

Logo, deduzimos que $H \trianglelefteq HK$.

Seja $\varphi$ o homomorfismo de grupos definido a seguir:

\[
\varphi: 
\begin{array}{c}
K \to \frac{HK}{H} \\
x \mapsto xH
\end{array}
\]

Note que $\frac{HK}{H}$ Ă© de fato um grupo, uma vez que $H\trianglelefteq HK$.

Note tambĂ©m que $\varphi$ se trata mesmo de um homomorfismo (restriĂ§ĂŁo da projeĂ§ĂŁo canĂ´nica).

Mostremos que $\varphi$ Ă© um homomorfismo sobrejetivo.

Tome $x\in \frac{HK}{H}$, logo:

\[
x = \underbrace{hk}_{\in HK}H = Hhk = Hk = kH = \varphi(k), \; k\in K
\]

Logo, mostramos que $\varphi$ Ă© sobrejetiva, isto Ă©:

\[
\operatorname{Im}(\varphi) = \frac{HK}{H}
\]

Mostremos agora que $\ker \varphi = H\cap K$.

Sabemos que, por definiĂ§ĂŁo:

\[
\ker \varphi = \{k\in K : \varphi(k) = H\}
\]

Deduzimos portanto que $k\in \ker\varphi \iff k\in H\cap K$.

Logo, deduzimos que $\ker \varphi = H\cap K$.

Aplicando o \hyperlink{primteoiso}{\textbf{Primeiro Teorema do Isomorfismo}} deduzimos diretamente que:

\[
\frac{K}{H\cap K} \cong \frac{HK}{H}
\]

\qed

\begin{theorem}
\textbf{Terceiro Teorema dos Isomorfismos}

Seja $G$ um grupo e $H,K \trianglelefteq G$ tal que $K\subseteq H$.

Logo, 

\[
\frac{H}{K} \trianglelefteq \frac{G}{K}
\]

e,

\[
\frac{G/K}{H/K} \cong \frac{G}{H}
\]

\end{theorem}


\textbf{DemonstraĂ§ĂŁo}:

Mostremos, primeiramente, que $H/K \trianglelefteq G/K$:

Seja $gK \in G/K$ e $hK \in H/K$, mostremos que $(gK)(hK)(gK)^{-1} \subseteq H/K$:

\[
(gK)(hK)(gK)^{-1} = gKhKKg^{-1} = gKhKg^{-1} = ghKKg^{-1} = 
\]

\[
= ghKg^{-1} = \underbrace{ghg^{-1}}_{\in H}K \in H/K 
\]

Logo, concluĂ­mos que:

\[
\frac{H}{K} \trianglelefteq \frac{G}{K}
\]

Seja a funĂ§ĂŁo $\varphi$ definida a seguir:

\[
\varphi: 
\begin{array}{c}
\frac{G}{K} \to \frac{G}{H} \\[0.3cm]
gK \mapsto gH
\end{array}
\]

Note que, como $K\subseteq H$, todo elemento de $G/K$ Ă© evidentemente elemento de $G/H$ e a funĂ§ĂŁo Ă© bem definida, sobrejetora e um homomorfismo de grupos.

Logo, deduzimos que: 

\[
\operatorname{Im} \varphi = \frac{G}{H}
\]

Mostremos que $\ker \varphi = H/K$:

Por definiĂ§ĂŁo, sabemos que:

\[
\ker \varphi = \{gK\in G/K : \varphi(gK) = H\} 
\]

Logo:

\[
gK\in \ker \varphi \iff \varphi(gK) = H \iff gH = H \iff g\in H \iff gK \subseteq H/K
\]

ConcluĂ­mos que $\ker \varphi = H/K$.

Portanto, aplicando o \hyperlink{primteoiso}{\textbf{Primeiro Teorema dos Isomorfismos}} deduzimos diretamente que:

\[
\frac{G/K}{H/K} \cong \frac{G}{H}
\]

\qed

\begin{proposition}

\end{array}

Ă© um automorfismo de $G$ denominado \textbf{automorfismo interno de $G$ associado a $g\in G$}.

\textbf{DemonstraĂ§ĂŁo:}

Mostremos que $\mathcal{I}_g$ Ă© de fato um automorfismo de $G$, dado $g\in G$.

Mostremos, primeiramente, que $\mathcal{I}_g$ Ă© um homomorfismo de grupos.

Sabemos, trivialmente, que $\mathcal{I}_g$ Ă© de fato uma funĂ§ĂŁo.

Sejam $x, y \in G$, temos, por definiĂ§ĂŁo:

\[
\mathcal{I}_g(xy) = gxyg^{-1} = gx(g^{-1}g)yg^{-1} = (gxg^{-1})(gyg^{-1}) = \mathcal{I}_g(x)\mathcal{I}_g(y)
\]

Portanto, concluĂ­mos que $\mathcal{I}_g$ Ă© um homomorfismo de grupos. 

Mostremos que $\mathcal{I}_g$ Ă© uma funĂ§ĂŁo bijetiva.

Sejam $x, y \in G$:

\[
\mathcal{I}_g(x) = \mathcal{I}_g(y) \Rightarrow gxg^{-1} = gyg^{-1} \Rightarrow g^{-1}gxg^{-1}g = g^{-1}gyg^{-1}g \Rightarrow x = y
\]

Logo, $\mathcal{I}_g$ Ă© injetiva.

Tomemos, ainda, $y \in G$ e $x = g^{-1}yg$. Temos, portanto:

Logo, $\mathcal{I}_g$ Ă© sobrejetiva.

\begin{proposition}

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos primeiramente que para $\varphi_1, \varphi_2 \in \operatorname{Inn}(G), \; \varphi_1\circ \varphi_{2}^{-1}\in \operatorname{Inn}(G)$, onde $\circ$ Ă© a composiĂ§ĂŁo usual de funĂ§Ăµes.

\[

\]

Temos:

\]

\[

Logo, concluĂ­mos que $\operatorname{Inn}(G) \leq \operatorname{Aut}(G)$.

\qed

\hypertarget{def171}{}
\begin{definition}
Seja $\mathfrak{G} = \{G_1, \ldots, G_n \}$ uma famĂ­lia de grupos.

Definimos \textbf{produto direto externo de $\mathfrak{G}$} como sendo o conjunto $G_1 \times \ldots \times G_n$ munido da operaĂ§ĂŁo: 

\[
(x_1,\ldots, x_n)\cdot (y_1,\ldots, y_n) \overset{def}{=} (x_1y_1, \ldots, x_ny_n).
\]

Denotamos, comumente:

\[
\prod_{i=1}^{n} G_i \overset{def}{=} G_1 \times \ldots \times G_n.
\]
\end{definition}

\textit{ObservaĂ§ĂŁo: por uma questĂŁo de convenĂ§ĂŁo, a operaĂ§ĂŁo "$\cdot$" serĂˇ omitida.}

\begin{proposition}

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Via a \hyperlink{def171}{\textbf{DefiniĂ§ĂŁo 1.7.1}} temos que a operaĂ§ĂŁo "$\cdot$" Ă© uma lei de composiĂ§ĂŁo interna por construĂ§ĂŁo. 

Temos tambĂ©m, de forma evidente, que o elemento neutro e o elemento inverso de $x \overset{def}{=}(x_1,\ldots,x_n)\in \prod_{i=1}^{n} G_i$ sĂŁo dados por, respectivamente:

\[
e \overset{def}{=} (e_1, \ldots, e_n) \in \prod_{i=1}^{n} G_i
\]

Onde $e_n$ representa o elemento neutro do $n$-Ă©simo grupo do produto direto externo.

\[
x^{-1} \overset{def}{=} (x_{1}^{-1},\ldots, x_{n}^{-1})\in \prod_{i=1}^{n} G_i
\]

Onde $x_{n}^{-1}$ representa o elemento inverso para $x_n$ no $n$-Ă©simo grupo do produto direto externo.

Temos tambĂ©m que o produto direto externo de $\mathfrak{G}$ Ă© evidentemente associativo, uma vez  que $G_1, \ldots, G_n$ sĂŁo grupos.

ConcluĂ­mos, portanto, que $(\prod_{i=1}^{n} G_i,\cdot)$ Ă© um grupo. 

\qed

\hypertarget{def172}{}
\begin{definition}
Sejam $G$ um grupo e $H_1, \ldots, H_n \leq G$.

Dizemos que $G$ Ă© um \textbf{produto direto interno} de $H_1, \ldots, H_n$ se, e somente se, as seguintes condiĂ§Ăµes sĂŁo satisfeitas:

\begin{itemize}
    \item $G = H_1 \ldots H_n$;
    \item $H_i \trianglelefteq G, \; \forall i \in \llbracket 1, n \rrbracket$;
    \item $\left\{
          \begin{aligned}
            &H_1 \cap H_2 \ldots H_n = \{e\} \\
            &H_i \cap H_1 \ldots H_{i-1}H_{i+1}\ldots H_n = \{e\},\; \forall i \in \llbracket 2, n-1 \rrbracket\\
            &H_n \cap H_1 \ldots H_{n-1} = \{e\}
          \end{aligned}
          \right.$
\end{itemize}

Caso $G$ e $H_1, \ldots, H_n$ satisfaĂ§am as condiĂ§Ăµes acima, denotamos:

\[
G \overset{def}{=} \bigodot_{i=1}^{n} H_i
\]

\end{definition}

\hypertarget{prop172}{}
\begin{proposition}
Sejam $G$ um grupo e $H_1, \ldots, H_n \leq G$.

Se $G$ for um produto direto interno de $H_1, \ldots, H_n$, entĂŁo quaisquer elementos de $H_1, \ldots, H_n$ distintos comutam entre si.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Sejam $(h_i, h_j) \in H_i \times H_j$, onde $i, j \in \llbracket 1, n \rrbracket$ e $i \neq j$.

Tomemos o elemento $g \overset{def}{=} [h_i, h_j] = h_ih_jh_{i}^{-1}h_{j}^{-1} \in G$.

Note que:

\[
h_ih_jh_{i}^{-1}h_{j}^{-1} = (h_ih_jh_{i}^{-1})h_{j}^{-1} \in H_j, \; \; \text{pois} \; \; H_j \trianglelefteq G
\]

\[

\begin{theorem}

\[

\end{theorem}

\[

Deduzimos, portanto, que $m_i \nmid x$ e $m_i \mid x$, para algum representante $x$ da classe lateral $\overline{x}$. Absurdo.

Como $\mathfrak{M}$ Ă© nĂŁo vazio, concluĂ­mos que $\mathfrak{M} = \{\overline{0}\}$.

Mostremos, por fim, que:

\[

\]

Note que pela \hyperlink{prop156}{\textbf{ProposiĂ§ĂŁo 1.5.6}} temos que:

\[
\left\vert \sum_{i=1}^{n} \langle \overline{M_i} \rangle \right\vert = \left\vert \langle \overline{M_1} \rangle + \sum_{i=2}^{n} \langle \overline{M_i} \rangle \right\vert = \frac{\vert \langle \overline{M_1} \rangle\vert \displaystyle\left\vert \sum_{i=2}^{n} \langle \overline{M_i} \rangle \right\vert}{\underbrace{\left\vert \langle \overline{M_1}\rangle \cap \sum_{i=2}^{n} \langle \overline{M_i} \rangle \right\vert}_{=1}} = m_1\left\vert \sum_{i=2}^{n}\langle \overline{M_i} \rangle \right\vert

Aplicando a fĂłrmula indutivamente em $i$, temos que:

Como $\sum_{i=1}^{n} \langle \overline{M_i} \rangle \subseteq \mathbb{Z}/M\mathbb{Z}$, deduzimos diretamente que:

ConcluĂ­mos, finalmente, que:

\qed

\section{Produto Semidireto de Grupos}

\begin{definition}
\color{red}

Seja $G$ um grupo e $H,K\leq G$. Dizemos que \textbf{$G$ Ă© o produto semidireto interno de $H$ e $K$ se:}

\begin{itemize}
  \item $G = HK$;
  \item $H \trianglelefteq G$;
  \item $H\cap K = \{e\}$.
\end{itemize}

Caso $G$, $H$ e $K$ satisfaĂ§am as condiĂ§Ăµes acima, denotamos:

\[
G = H \rtimes K
\]

\end{definition}

{\color{red}
\textit{ObservaĂ§Ăµes:}

\begin{itemize}
  \item \textit{Note que, diferentemente da \hyperlink{def172}{\textbf{DefiniĂ§ĂŁo 1.7.2}}, definimos o produto semidireto interno apenas entre $2$ subgrupos de $G$. Tal escolha Ă© motivada pela natureza assimĂ©trica do produto semidireto, ao contrĂˇrio da simetria do produto direto. Entretanto Ă© possĂ­vel estender essa estrutura de forma indutiva para $n$ subgrupos de $G$.   \item \textit{Caso $K\trianglelefteq G$, entĂŁo denotarĂ­amos $G = K\rtimes H$ ou $G = H\ltimes K$}.
\end{itemize}

}

\begin{example}
\color{red}

Tomemos os grupos $GL_2(\mathbb{R})$ e $SL_2(\mathbb{R})$. Sabemos, a partir dos exemplos \hyperlink{grupoespecial}{\textbf{1.3.3}} e \hyperlink{grupoespecial2}{\textbf{1.6.6}} que $SL_2(\mathbb{R}) \trianglelefteq GL_2(\mathbb{R})$. Denotemos $H\overset{def}{=} SL_2(\mathbb{R})$ \\[2.5pt]

\end{pmatrix},\; a\neq 0 \right\}$. Note que $K\leq G$, mas $K \not\trianglelefteq G$, pois:

\[

\begin{pmatrix}
  1 & 0 \\
  1 & 1
\end{pmatrix}}_{\overset{def}{=} M \in GL_2(\mathbb{R})}
\begin{pmatrix}
  2 & 0\\
  0 & 1
\end{pmatrix}

\begin{pmatrix}
  1 & 0\\
  -1 & 1
\end{pmatrix}}_{= M^{-1} \in GL_2(\mathbb{R})}

\begin{pmatrix}
  2 & 0\\
  1 & 1
\end{pmatrix}

\]

PorĂ©m, note que $GL_2(\mathbb{R}) = HK$, uma vez que:

Sendo $M\in GL_2(\mathbb{R})$ e $M\left(\frac{1}{a}\right)\in K$, temos,

\begin{pmatrix}
  1 & 0 \\
  0 & 1
\end{pmatrix}
\right\}

\end{example}

\begin{proposition}
\color{red}

\qed

Note que, diferentemente do produto direto interno, nĂŁo existe a comutatividade entre elementos de subgrupos distintos. \begin{definition}

\end{definition}

  \item \textit{Note que, caso $\theta$ seja trivial, entĂŁo $H\rtimes_\theta K = H\times K$}.

\begin{proposition}

\[

\begin{aligned}

\end{aligned}

\end{aligned}

ConcluĂ­mos, portanto, que $H\rtimes_{\theta} K$ Ă© um grupo.

\begin{definition}

\begin{proposition}

\begin{theorem}

& & \phantom{\Upsilon_g: \ } x \longmapsto gx
\end{array}

\end{theorem}

\begin{definition}

    &\sigma(x_2) = x_3 \\
    &\dots \\
    &\sigma(x_{k-1}) = x_k \\

\begin{itemize}

	\item \textit{Chamamos tambĂ©m um $k$-ciclo de \textbf{ciclo de comprimento $k$}};
	\item \textit{Caso $\sigma,\tau$ sejam ciclos com suportes dois a dois disjuntos, temos diretamente via a \hyperlink{prop213}{\textbf{ProposiĂ§ĂŁo 2.1.3}} que $\sigma\tau=\tau\sigma$};

=

	\textit{Evidentemente $\sigma$ nĂŁo Ă© um ciclo, porĂ©m $\sigma$ pode ser escrito como produto de $2$ ciclos de suportes disjuntos}.

\end{definition}

\hypertarget{teo221}{}
\begin{theorem}
Seja $X=\{x_1,\ldots,x_n\}$ para $n\geq 2$. EntĂŁo toda permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X\setminus \{\text{id}_X\}$ pode ser decomposta em um produto de ciclos de suportes dois a dois disjuntos.

\end{theorem}

\textbf{DemonstraĂ§ĂŁo:}

Tomemos de forma arbitrĂˇria um elemento $x_j$, com $j\in\llbracket 1,n\rrbracket$ tal que $\sigma(x_j)\neq x_j$. Note que Ă© garantido que tal elemento exista, uma vez que $\sigma \neq \text{id}_X$.

\]

Note que a restriĂ§ĂŁo $\sigma\upharpoonright_{\mathfrak{O}(x_j)}$ forma um $k$-ciclo e $\text{supp}(\sigma\upharpoonright_{\mathfrak{O}(x_j)}) = \mathfrak{O}(x_j)$, pela \hyperlink{prop212}{\textbf{ProposiĂ§ĂŁo 2.1.2}}.

\[

\]

deduzimos que $\sigma$ Ă© igual a composiĂ§ĂŁo de todo $k_x$-ciclo $\sigma\upharpoonright_{\mathfrak{O}(r)}$, onde $r\in \mathfrak{R}$.

Mostremos que tal decomposiĂ§ĂŁo Ă© Ăşnica. 

Suponha $\sigma = \rho_1\ldots \rho_h = \eta_1\ldots\eta_m$ duas decomposiĂ§Ăµes de $\sigma$ em ciclos de suportes dois a dois disjuntos.

Tomemos $x\in \text{supp}(\sigma)$. Como os ciclos dentro de cada decomposiĂ§ĂŁo possuem suportes disjuntos, $x$ pertence ao suporte de exatamente um ciclo da primeira decomposiĂ§ĂŁo, digamos $\rho_i$, e a exatamente um ciclo da segunda, digamos $\eta_j$.

Como todos os outros ciclos nas respectivas decomposiĂ§Ăµes fixam $x$, a aĂ§ĂŁo iterada de $\sigma$ sobre $x$ Ă© determinada unicamente por $\rho_i$ e $\eta_j$. Ou seja, para todo $k \in \mathbb{Z}$, temos:

\[
\sigma^k(x) = \rho_i^k(x) = \eta_j^k(x)
\]

Isso significa que $\rho_i$ e $\eta_j$ possuem o mesmo suporte, e, pelo fato de serem ciclos, temos portanto $\rho_i = \eta_j$.

Como isso vale para qualquer elemento $x \in \text{supp}(\sigma)$, todo ciclo que aparece na primeira decomposiĂ§ĂŁo necessariamente aparece na segunda, e vice-versa. Portanto, os conjuntos de ciclos $\{\rho_1, \ldots, \rho_h\}$ e $\{\eta_1, \ldots, \eta_m\}$ coincidem, provando que a decomposiĂ§ĂŁo Ă© Ăşnica exceto pela ordem em que os ciclos sĂŁo escritos.

\qed

\begin{proposition}
Seja $X$ um conjunto finito e nĂŁo vazio tal que $\text{Card}(X) \geq 3$. 

EntĂŁo $Z(\mathfrak{S}_X) = \{\text{id}_X\}$.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Sejam $\sigma,\tau\in \mathfrak{S}_X$ e $(x,y) \in \left(X\setminus\text{supp}(\sigma)\right) \times \text{supp}(\sigma)$. Tomemos $\sigma\neq\text{id}_X$ e $\tau \overset{def}{=} (x\;y)$.

Ora, temos entĂŁo $y\notin\text{supp}(\tau\sigma\tau^{-1})$, pois $\tau\sigma\tau^{-1}(y) = y$. Portanto, temos diretamente que $\tau\sigma\tau^{-1} \neq \sigma \iff \tau\sigma \neq \sigma\tau$.

Suponha, agora, que $X\setminus\text{supp}(\sigma) = \emptyset$. Procedemos por disjunĂ§ĂŁo de casos. 

Considere, inicialmente, que $\sigma$ possui um ciclo de comprimento maior ou igual a $3$ (como $\text{Card}(X)\geq 3$, entĂŁo podemos afirmar que tal caso existe). 

Tomemos, sem perda de generalidade, $\sigma \overset{def}{=} (x_1\;x_s\;x_k)\kappa$, onde $\kappa\in\mathfrak{S}_X$ e $\{x_1, x_s, x_k\}\cap\text{supp}(\kappa)=\emptyset$. Sabemos que $\sigma$ pode ser decomposto unicamente de tal forma pelo \hyperlink{teo221}{\textbf{Teorema 2.2.1}}.
 
Definimos, ainda, $\tau \overset{def}{=} (x_1 \; x_s)$. Logo, temos que $\tau\sigma\tau^{-1}(x_k) = x_s \neq x_1$ e, portanto, $\tau\sigma\tau^{-1} \neq \sigma \iff \tau\sigma\neq\sigma\tau$.

Suponha, agora, que $\sigma$ nĂŁo possui nenhum ciclo de comprimento maior ou igual a $3$. Isto Ă©, $\sigma$ Ă© apenas composto de transposiĂ§Ăµes de suportes dois a dois disjuntos (note que, no mĂ­nimo, duas transposiĂ§Ăµes, visto que $\text{Card}(X) \geq 3$).

Denotemos $\sigma \overset{def}{=} (x_1\; x_2)(x_3\; x_4)\kappa$, onde $\kappa\in\mathfrak{S}_X$ e $\{x_1, x_2, x_3, x_4\}\cap\text{supp}(\kappa)=\emptyset$.

Tomemos, neste caso, $\tau = (x_1 \; x_3)$. Temos, portanto, $\tau\sigma\tau^{-1}(x_3) = x_2 \neq x_4 \implies \tau\sigma\tau^{-1} \neq \sigma$.

Como para ambos os casos sempre podemos encontrar uma permutaĂ§ĂŁo $\tau$ tal que $\sigma$ nĂŁo comuta, entĂŁo deduzimos que $Z(\mathfrak{S}_X) = \{\text{id}_X\}$ para $\text{Card}(X) \geq 3$.

\qed

\begin{proposition}

\end{corollary}

\textbf{DemonstraĂ§ĂŁo:}

\[

Sabemos a partir da \hyperlink{prop221}{\textbf{ProposiĂ§ĂŁo 2.2.1}} que cada $k$-ciclo pode ser escrito como $k-1$ transposiĂ§Ăµes (de suportes nĂŁo necessariamente disjuntos).

\[
\sum_{i=1}^{m} k_i - 1 = \sum_{i=1}^{m} k_i - \sum_{i=1}^{m} 1 = \sum_{i=1}^{m} k_i - m
\]  

Como $\sum_{i=1}^m k_i \leq n$, pelo PrincĂ­pio da Casa de Pombos, e $m\in\llbracket 1,\lfloor \frac{n}{2}\rfloor\rrbracket$, $\sigma$ pode se fatorar em no mĂˇximo $n-1$ transposiĂ§Ăµes (de suportes nĂŁo necessariamente disjuntos).

\begin{proposition}

\[
\sigma (x_i\; x_j) \sigma^{-1} = (\sigma(x_i)\; \sigma(x_j)).
\]
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Seja $x \in X \setminus \{\sigma(x_i),\;\sigma(x_j)\}$. Temos diretamente que:

\[
\sigma^{-1}(x) = \underbrace{x'}_{\in X\setminus \{x_i,x_j\}} \implies \bigl((x_i\;x_j)\sigma^{-1}\bigr)(x) = x' \implies \bigl(\sigma(x_i\;x_j)\sigma^{-1}\bigr)(x) = \sigma(x') = x
\]

Tomemos agora $x = \sigma(x_i)$, temos que:

\[
\sigma^{-1}(\sigma(x_i)) = x_i \implies \bigl((x_i\;x_j)\sigma^{-1}\bigr)(\sigma(x_i)) = x_j \implies \bigl(\sigma(x_i\;x_j)\sigma^{-1}\bigr)(\sigma(x_i)) = \sigma(x_j)
\]

Analogamente temos que $x = \sigma(x_j) \implies \bigl(\sigma(x_i\;x_j)\sigma^{-1}\bigr)(\sigma(x_j)) = \sigma(x_i)$.


ConcluĂ­mos que $\sigma (x_i\; x_j) \sigma^{-1} = (\sigma(x_i)\; \sigma(x_j))$.

\qed

\begin{proposition}
\color{red}
Seja $X$ um conjunto finito, nĂŁo vazio e munido de uma ordem total. Logo, toda transposiĂ§ĂŁo de $\mathfrak{S}_X$ Ă© uma permutaĂ§ĂŁo Ă­mpar.
\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Seja $\tau = (x\;y) \in \mathfrak{S}_X$ uma transposiĂ§ĂŁo tal que, pela ordem total de $X$, $x < y$ e $\tau(x) > \tau(y)$.

Note que, como $x<y$ e $\tau(x) > \tau(y)$, entĂŁo temos diretamente que $\forall z_x\in X_{<x},\; \tau(x) > z$.

Simetricamente, temos que $\forall z_y\in X_{>y},\; \tau(y) < z$.

Portanto, resta contar o nĂşmero de inversĂµes para $z\in\llbracket x, y\rrbracket$.

Ora, para todo $z\in\llbracket x,y\rrbracket$ temos diretamente que $\tau(y) > z, \; \forall z\in \llbracket x,y\rrbracket$.

Seja $M$ o nĂşmero de inversĂµes $(\tau(y), z)$ calculado acima. 

Note que, de forma anĂˇloga, temos que $\tau(x) < z, \; \forall z\in \llbracket x, y \rrbracket$. 

Portanto, o nĂşmero de inversĂµes dos pares $(\tau(x), z)$ Ă© tambĂ©m $M$. 

Entretanto, como contabilizamos duas vezes a inversĂŁo $(\tau(x),\tau(y))$, temos que $\mathscr{I}(\tau) = 2M-1$. 

Portanto, $\varepsilon(\tau) = -1$ e $\tau$ Ă© Ă­mpar. 

\qed}

\begin{theorem}
\color{red}
Sejam $X$ um conjunto finito munido de uma ordem total tal que $\text{Card}(X) = n\in\mathbb{Z}_{\geq 2}$ e uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$.

Portanto, ao decompor $\sigma = \tau_1\ldots\tau_m$ para $m\in\mathbb{Z}_{+}^{*}$ onde $\tau_i$ Ă© uma transposiĂ§ĂŁo para todo $i\in\llbracket 1,m\rrbracket$, temos que:

\[
\varepsilon(\sigma) = (-1)^m
\]

\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Note, primeiramente, que podemos supor sem perda de generalidade que $\sigma$ Ă© decomposto em um nĂşmero finito $p\in\mathbb{Z}_{\geq m}$ de transposiĂ§Ăµes elementares. 

Isso se dĂˇ pois, pela \hyperlink{prop224}{\textbf{ProposiĂ§ĂŁo 2.2.4}}, temos diretamente via a construĂ§ĂŁo da demonstraĂ§ĂŁo que uma dada transposiĂ§ĂŁo se decompĂµe em um nĂşmero Ă­mpar de transposiĂ§Ăµes elementares. Portanto, $(-1)^m = (-1)^p$.

Sejam $\sigma' \in \mathfrak{S}_X$ uma permutaĂ§ĂŁo arbitrĂˇria e $\tau' = (x_d\; x_{d+1})\in \mathfrak{S}_X$ uma transposiĂ§ĂŁo elementar tal que $x_d < x_{d+1}$ via a ordem total de $X$.

Note que, ao tomar sem perda de generalidade o produto $\sigma'\tau'$, a transposiĂ§ĂŁo $\tau'$ atua trocando os elementos apenas nas posiĂ§Ăµes adjacentes $d$ e $d+1$. Consequentemente, a quantidade total de inversĂµes envolvendo qualquer outro par de posiĂ§Ăµes permanece idĂŞntica. Temos, portanto, apenas o par $(x_d, x_{d+1})$ a analisar:

Suponha que $\sigma'\tau'(x_d) > \sigma'\tau'(x_{d+1})$. Logo, a aplicaĂ§ĂŁo de $\tau'$ gerou uma nova inversĂŁo, de modo que $\mathscr{I}(\sigma'\tau') = \mathscr{I}(\sigma') + 1$.

Suponha, agora, que $\sigma'\tau'(x_d) < \sigma'\tau'(x_{d+1})$. Logo, simetricamente, $\mathscr{I}(\sigma'\tau') = \mathscr{I}(\sigma') - 1$.

Logo, concluĂ­mos que $\varepsilon(\sigma'\tau') = -\varepsilon(\sigma')$.

Como $\sigma = \tau_{1}'\ldots \tau_{p}'$ onde $\tau_{i}'$ Ă© transposiĂ§ĂŁo elementar para todo $i\in\llbracket 1,p\rrbracket$, temos diretamente que:

\[
\varepsilon(\sigma) = (-1)^p = (-1)^m
\]

\qed}

\begin{theorem}
\color{red}
Sejam $X$ um conjunto finito com $\text{Card}(X)\geq 2$ e a \textbf{funĂ§ĂŁo assinatura $\varepsilon$} tal que:

\[
\varepsilon: 
\begin{array}{r c l}
	\mathfrak{S}_X &\to &\{-1,1\} \\
	\sigma &\mapsto &\varepsilon (\sigma)
\end{array}
\]

Logo, $\varepsilon$ Ă© um epimorfismo de grupos.
\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Sejam $\sigma,\sigma' \in\mathfrak{S}_X$ tais que suas decomposiĂ§Ăµes em transposiĂ§Ăµes de $\mathfrak{S}_X$ sĂŁo dadas por:

\[
\sigma = \tau_1 \ldots \tau_p,\; p\in\mathbb{Z}_{+}^{*}
\]

\[
\sigma' = \tau_{1}' \ldots \tau_{q}',\; q\in\mathbb{Z}_{+}^{*}
\]

Logo, temos que:

\[
\varepsilon(\sigma\sigma') = (-1)^{p+q} = (-1)^p(-1)^q = \varepsilon(\sigma)\varepsilon(\sigma').
\]

Como $\varepsilon(\text{id}_X) = 1$ e $\varepsilon(\tau) = -1$ para qualquer $\tau\in\mathfrak{S}_X$ (cuja existĂŞncia Ă© garantida, uma vez que $\text{Card}(X)\geq 2$), entĂŁo concluĂ­mos que $\varepsilon$ Ă© um epimorfismo de grupos.

\qed

\textit{ObservaĂ§ĂŁo: note que $\forall\sigma,\tau\in\mathfrak{S}_X$ tal que $\tau$ Ă© uma transposiĂ§ĂŁo, temos que $\varepsilon(\sigma\tau)=\varepsilon(\tau\sigma)=-\varepsilon(\sigma)$}.

}

\section{Grupo Alternado $\mathfrak{A}_X$}

\begin{definition}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio. Definimos o conjunto $\mathfrak{A}_X$ como o \textbf{conjunto alternado de $\mathfrak{S}_X$}, tal que:

\[
\mathfrak{A}_X \overset{def}{=} \{\sigma\in\mathfrak{S}_X\; ; \; \varepsilon(\sigma) = 1\}.
\]

Onde $\varepsilon$ denota a funĂ§ĂŁo assinatura.
\end{definition}

\begin{proposition}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio, entĂŁo: 

\[
\mathfrak{A}_X \trianglelefteq \mathfrak{S}_X
\]
\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Sabemos que a funĂ§ĂŁo assinatura $\varepsilon$ Ă© um epimorfismo de grupos. Por definiĂ§ĂŁo, sabemos que o nĂşcleo de $\varepsilon$ Ă© definido por:

\[
\ker{\varepsilon} \overset{def}{=} \{\sigma\in\mathfrak{S}_X \; ; \; \varepsilon(\sigma)=1 \} = \mathfrak{A}_X
\]

Como $\ker{\varepsilon}\trianglelefteq \mathfrak{S}_X$, entĂŁo temos diretamente que:

\[
\mathfrak{A}_X \trianglelefteq \mathfrak{S}_X
\]

\qed}

\begin{example}
\color{red}
Enumeremos todos os elementos de $\mathfrak{S}_3$:

\[
(1\;2\;3),\;(1\;3\;2),\;(1\;2),\;(1\;3),\;(2\;3),\;\text{id}_{\{1,2,3\}}
\]

Logo, fica claro que os elementos de $\mathfrak{A}_3$ sĂŁo:

\[
(1\;2\;3),\;(1\;3\;2),\;\text{id}_{\{1,2,3\}}.
\]

\end{example}

\begin{proposition}
\color{red}
Seja $X$ um conjunto finito tal que $\text{Card}(X) = n\in\mathbb{Z}_{\geq 2}$. Logo:

\[
\vert \mathfrak{A}_X \vert = \frac{n!}{2}.
\]
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $\sigma,\tau \in \mathfrak{S}_X$, tal que $\tau$ Ă© uma transposiĂ§ĂŁo fixada.

Definimos a funĂ§ĂŁo $\varphi$ de forma que:

\[
\varphi: 
\begin{array}{r c l}
	\mathfrak{S}_X &\to &\mathfrak{S}_X \\
	\sigma &\mapsto &\sigma\tau
\end{array}
\]

Note que $\varphi$ Ă© de fato uma funĂ§ĂŁo e Ă© bijetiva. Sejam $\sigma_1,\sigma_2\in\mathfrak{S}_X$ tal que $\sigma_1=\sigma_2$, temos que:

\[
\sigma_1 = \sigma_2 \iff \sigma_1 \tau = \sigma_2 \tau \iff \varphi(\sigma_1) = \varphi(\sigma_2)
\]

Como a funĂ§ĂŁo $\varphi$ Ă© injetiva e tem como domĂ­nio e contradomĂ­nio o mesmo conjunto $\mathfrak{S}_X$, deduzimos que $\varphi$ Ă© de fato uma bijeĂ§ĂŁo.

Sabemos, tambĂ©m, que $\varepsilon(\sigma\tau) = -\varepsilon(\sigma)$ para qualquer $\sigma\in\mathfrak{S}_X$. Portanto, deduzimos diretamente que $\varphi(\mathfrak{A}_X) = \mathfrak{S}_X \setminus \mathfrak{A}_X$.

Como a funĂ§ĂŁo $\varphi$ Ă© bijetiva, entĂŁo tambĂ©m temos que $\vert \mathfrak{A}_X\vert = \text{Card}(\mathfrak{S}_X\setminus \mathfrak{A}_X)$.

Ă‰ evidente ainda que $\mathfrak{S}_X = \mathfrak{A}_X \sqcup \mathfrak{S}_X\setminus\mathfrak{A}_X$, logo:

\[
\vert \mathfrak{S}_X \vert = \vert \mathfrak{A}_X\vert + \text{Card}(\mathfrak{S}_X\setminus\mathfrak{A}_X) = 2\mathfrak{A}_X \iff \vert \mathfrak{A}_X \vert = \frac{n!}{2}
\]

\qed

\textit{ObservaĂ§ĂŁo: note que para o caso em que $\text{Card}(X) = 1$ temos evidentemente que $\mathfrak{A}_X = \{\text{id}_X\}\implies \vert \mathfrak{A}_X\vert = 1$}.

}

\begin{proposition}
\color{red}

Sejam $X$ um conjunto finito e nĂŁo vazio e $x,y,z\in X$, entĂŁo:

\[

\]

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Pela \hyperlink{prop232}{\textbf{ProposiĂ§ĂŁo 2.3.2}} temos ainda que $\varepsilon\left((x\;y\;z)^k\right) = 1,\; \forall k\in\mathbb{Z}$. 

Portanto, temos diretamente que $\langle (x\;y\;z)\rangle \subseteq \mathfrak{A}_X$.

Mostremos que $\mathfrak{A}_X \subseteq \langle(x\;y\;z)\rangle$.

Note, primeiramente, que $\text{id}_X = (x\;y\;z)^3 \in \langle(x\;y\;z)\rangle$.

Perceba tambĂ©m que qualquer par de transposiĂ§Ăµes de $\mathfrak{S}_X$ pode ser escrito como um $3$-ciclo. Sejam $a,b,c\in X$, temos que:

Como qualquer par de transposiĂ§Ăµes pode ser escrito como um $3$-ciclo, entĂŁo $\mathfrak{A}_X \subseteq \langle (x\;y\;z)\rangle$.

ConcluĂ­mos que:

\qed}

\begin{theorem}

\qed}

Suponha $H\trianglelefteq \mathfrak{A}_X$ tal que $H$ contĂ©m um $3$-ciclo. Ora, entĂŁo, por definiĂ§ĂŁo de normalidade, todo elemento conjugado do $3$-ciclo de $H$ por permutaĂ§Ăµes de $\mathfrak{A}_X$ pertence Ă  $H$.

\begin{theorem}
\color{red}
Seja $X$ um conjunto nĂŁo vazio e finito, onde $n\overset{def}{=} \text{Card}(X)$. 

Se $n\neq 4$, entĂŁo $\mathfrak{A}_X$ Ă© um grupo simples.
\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Note que, caso $n = 1$ ou $n = 2$, temos trivialmente que $\mathfrak{A}_X$ Ă© simples e, pelo \hyperlink{teolagrange}{\textbf{Teorema de Lagrange}}, sabemos que se $n=3$ entĂŁo $\mathfrak{A}_X$ Ă© simples (uma vez que $\vert \mathfrak{A}_X\vert = 3$ Ă© primo).

Suponhamos que $n\geq 5$. Para mostrar que $\mathfrak{A}_X$ Ă© simples basta mostrar que se existe um subgrupo normal nĂŁo trivial $H\trianglelefteq \mathfrak{A}_X$, entĂŁo $H$ contĂ©m um $3$-ciclo. Pela \hyperlink{prop246}{\textbf{ProposiĂ§ĂŁo 2.4.6}} temos $H=\mathfrak{A}_X$.

Sabemos que, para qualquer $\sigma\in\mathfrak{S}_X$, se $\text{Card}(\text{supp}(\sigma)) = 3$, entĂŁo $\sigma$ Ă© um $3$-ciclo. Mostremos que para qualquer permutaĂ§ĂŁo de $H$ podemos realizar um nĂşmero finito de operaĂ§Ăµes tal que a permutaĂ§ĂŁo resultante seja um $3$-ciclo de $H$. Mostremos via disjunĂ§ĂŁo de casos.

Suponha, primeiramente, que $\sigma\in H$ Ă© um produto de transposiĂ§Ăµes de suportes dois a dois disjuntos (nĂşmero par de transposiĂ§Ăµes, uma vez que $\varepsilon(\sigma) = 1$). Note que a existĂŞncia de $s\in X\setminus\text{supp}(\sigma)$ Ă© garantida, uma vez que $n\geq 5$.

Tomemos $\sigma' \overset{def}{=} [\sigma,\tau]$. Note que $\sigma'\in H$, uma vez que $H\trianglelefteq \mathfrak{A}_X$. Temos que:

\[
\sigma' = (x\;y)(z\;t)\gamma(x\;y)(z\;s)(x\;y)(z\;t)\gamma(x\;y)(z\;s) = (z\;t)(z\;s)(z\;t)(z\;s) = (z\;t\;s) \in H

Logo, existe um $3$-ciclo em $H$. 

Suponha, agora, que a decomposiĂ§ĂŁo de $\sigma\in H$ em ciclos de suportes dois a dois disjuntos possui um $k$-ciclo, tal que $k\in\mathbb{Z}_{\geq 3}$, mas $\sigma$ nĂŁo simplesmente um $3$-ciclo.

Note, primeiramente, que $\text{Card}(\text{supp}(\sigma)) \geq 5$, uma vez que um $4$-ciclo possui assinatura igual Ă  $-1$. 

Tomemos $\sigma = (x\;y\;z\;\ldots)\gamma$, onde $\gamma\in\mathfrak{A}_X$ tal que $\text{supp}\left((x\;y\;z\;\ldots)\right)\cap\text{supp}(\gamma) = \emptyset$.

Como $n\geq 5$, existem $p,q\in X\setminus\{x,y,z\}$ tal que $p,q\in\text{supp}(\sigma)$, uma vez que $\text{Card}(\text{supp}(\sigma)) \geq 5$.

Tomemos $\tau = (z\;p\;q)$ e $\sigma'\overset{def}{=} [\sigma,\tau]$. Analogamente ao caso anterior, temos que $\sigma'\in H$.

Note que, para $t\in X\setminus\text{supp}(\sigma)$, temos:

\[
\sigma'(t) = \sigma(\tau(\sigma^{-1}(\tau^{-1}(t)))) = \sigma(\tau(\sigma^{-1}(t))) = \sigma(\tau(t)) = \sigma(t) = t
\]

Note, tambĂ©m, que:

\[
\sigma'(y) = \sigma(\tau(\sigma^{-1}(\tau^{-1}(y)))) = \sigma(\tau(\sigma^{-1}(y))) = \sigma(\tau(x)) = \sigma(x) = y
\]

AlĂ©m disso, $\sigma'(p) = z$ e, portanto, concluĂ­mos que $\sigma'$ nĂŁo Ă© trivial e que tambĂ©m diminuiu estritamente o nĂşmero de elementos do suporte original de $\sigma$. Logo:

\[

\]

Podemos, portanto, realizar um nĂşmero finito de operaĂ§Ăµes em $\sigma$ tal que o resultado de tais operaĂ§Ăµes Ă© um elemento de $H$ com cardinal de suporte menor. Em particular, chegamos em um suporte de cardinal igual Ă  $3$. ConcluĂ­mos que $H$ contĂ©m um $3$-ciclo. 

Finalmente, deduzimos que $H$ possui $3$-ciclos para todos os casos e, portanto, $H = \mathfrak{A}_X$. 

Logo, $\mathfrak{A}_X$ Ă© um grupo simples.

\qed}


\begin{definition}

\end{definition}

\begin{proposition}
\color{red}

Sejam $X = \{x_1,x_2,x_3,x_4\}$ e $\varphi : \mathfrak{K} \to \mathcal{K}$ uma funĂ§ĂŁo tal que:

Finalmente, concluĂ­mos que $\varphi$ Ă© um isomorfismo de grupos e temos:

\qed

\begin{definition}

\[

\end{array}

\[
\begin{cases}
\varphi(e,x) = x, & \forall x \in X \\
\varphi(g, \varphi(h,x)) = \varphi(gh, x), & \forall (g,h,x) \in G\times G\times X
\end{cases}

Caso $\varphi$ seja uma aĂ§ĂŁo Ă  esquerda de $G$ sobre $X$ dizemos que \textbf{$G$ age Ă  esquerda sobre $X$} e denotamos:

\end{definition}

{\color{red}
\textit{ObservaĂ§Ăµes:}

\begin{itemize}
  \item \textit{Podemos, de forma anĂˇloga, definir a \textbf{aĂ§ĂŁo Ă  direita de $G$ sobre $X$}. Neste caso denotamos $X \curvearrowleft G$};
    \item \textit{Quando nĂŁo houver ambiguidade, a aĂ§ĂŁo Ă  esquerda de $G$ sobre $X$ serĂˇ denominada apenas como aĂ§ĂŁo de $G$ sobre $X$}.

\begin{example}
\color{red}

\begin{itemize}
  \item $\varphi(g,h) = gh \in G, \; \forall g,h \in G$;
  \item $\varphi(e, h) = eh = h, \; \forall h\in G$;
  \item $\varphi(g,\varphi(g',h)) = \varphi(g, g'h) = gg'h = \varphi(gg',h),\; \forall g,g',h\in G$.
\end{itemize}

A aĂ§ĂŁo $\varphi$ Ă© denominada \textbf{aĂ§ĂŁo por translaĂ§ĂŁo Ă  esquerda de $G$}.

\end{example}


\begin{example}

  \item $\varphi(e, h) = ehe^{-1} = h, \; \forall h\in G$;
  \item $\varphi(g_1,\varphi(g_2,h)) = \varphi(g_1, g_2hg_{2}^{-1}) = g_1g_2hg_2^{-1}g_1^{-1} = (g_1g_2)h(g_1g_2)^{-1}=\varphi(g_1g_2,h), \; \forall g_1,g_2,h\in G$.

\end{example}

\end{example}

\begin{proposition}
\color{red}
Sejam $G$ um grupo e $X$ um conjunto nĂŁo vazio tal que $G\curvearrowright X$. EntĂŁo:

\begin{enumerate}
  \item $X = \bigcup_{x\in X}\mathscr{O}(x);$
  \item $\text{Ou}\;\; \mathscr{O}(x) = \mathscr{O}(y) \;\; \text{ou}\;\; \mathscr{O}(x)\cap\mathscr{O}(y) = \emptyset,\; \forall x,y\in X.$
\end{enumerate}

\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Mostremos primeiramente o item $1.$. Note que $\bigcup_{x\in X} \mathscr{O}(x) \subseteq X$ Ă© trivial por definiĂ§ĂŁo.
AlĂ©m disso, dado $x\in X$, sempre teremos $x\in \mathscr{O}(x)$, pois $\varphi(e,x) = x\in \mathscr{O}(x)$. Logo, $X\subseteq \bigcup_{x\in X} \mathscr{O}(x)$.

Mostremos, agora, o item $2.$. Suponha que $\mathscr{O}(x)\cap\mathscr{O}(y) \neq \emptyset$, para $x,y\in X$. Consideremos ainda $\varphi$ a aĂ§ĂŁo de $G$ em $X$.

\]

\[

\]

ConcluĂ­mos, portanto, que $\mathscr{O}(x) = \mathscr{O}(y)$.

\qed}
