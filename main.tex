\section{OperaĂ§Ăľes BinĂĄrias}

\begin{definition}

Sejam $G$ e $E$ conjuntos nĂŁo-vazios e $\oplus$ uma funĂ§ĂŁo tal que:

\[
\oplus: 
\begin{array}{r c l}
G\bigtimes G &\to &E \\
(a, b) &\mapsto &\oplus (a,b)
\end{array}
\]

Definimos a funĂ§ĂŁo acima como a \textbf{operaĂ§ĂŁo binĂĄria de dois elementos de $G$ em $E$} e a escrevemos comumente como: $a\oplus b$.
\end{definition}

\begin{example}
    A adiĂ§ĂŁo usual $+$ ĂŠ uma operaĂ§ĂŁo binĂĄria de dois elementos de $\mathbb{I}$ em $\mathbb{R}$. Onde $\mathbb{I}$ denota o conjunto dos nĂşmeros irracionais.
\end{example}

\begin{example}
    Sejam $(a,b)\in \mathbb{R}^2$, a funĂ§ĂŁo que define a distĂ˘ncia cartesiana entre dois pontos $a$ e $b$:
    \[
    \text{dist}(a,b): 
    \begin{array}{c}
    \mathbb{R}\bigtimes \mathbb{R} \to \mathbb{R}^+ \\
    (a, b) \mapsto \sqrt{a^2 + b^2}
    \end{array}
    \]
    representa uma operaĂ§ĂŁo binĂĄria de dois elementos de $\mathbb{R}$ em $\mathbb{R}^+$.
    
\end{example}

\begin{definition}
A partir das notaĂ§Ăľes acima, definimos \textbf{lei de composiĂ§ĂŁo interna de $G\times G \to G$ se $E = G$}.
\end{definition}

\textit{ObservaĂ§ĂŁo: caso nĂŁo haja ambiguidade, denotaremos simplesmente \textbf{lei de composiĂ§ĂŁo interna em $G$} para representar a lei de composiĂ§ĂŁo interna de $G\times G \to G$.}

\begin{example}
    A operaĂ§ĂŁo usual $+$ em $\mathbb{N}$ ĂŠ uma lei de composiĂ§ĂŁo interna em $\mathbb{N}$, ao contrĂĄrio da operaĂ§ĂŁo usual $-$ de $\mathbb{N}$ em $\mathbb{Z}$.
\end{example}

\section{Grupos}

\begin{definition}
    Seja $G$ um conjunto nĂŁo-vazio. \textbf{Dizemos que $(G, \cdot)$ ĂŠ um grupo} se, e somente se, $\cdot$ ĂŠ uma lei de composiĂ§ĂŁo interna em $G$ tal que:
    \begin{enumerate}
    \item $\exists \,e \in G \; , \forall \,x \in G \; : x\cdot e = e\cdot x = x $;
    \item $\forall x \in G, \exists \hat{x}\in G: \; x\cdot \hat{x} = \hat{x} \cdot x = e$;
    \item $\forall \, x,y,z \in G : (x\cdot y)\cdot z = x\cdot (y\cdot z)$.
    \end{enumerate}
\end{definition}

\textit{ObservaĂ§Ăľes:}

\textit{Levando em consideraĂ§ĂŁo as notaĂ§Ăľes acima, temos:}
\begin{enumerate}
	\item \textit{Primeiramente, notamos que $e$ e $\hat{x}$ sĂŁo Ăşnicos, uma vez que:}

    \textit{Supondo que existam $e$ e $e'$ pertencentes Ă  $G$ que satisfazem o item 1, temos:}
    \[
        x \cdot e = x = x \cdot e' 
        \;\;\;\implies\;\;\;
        \hat{x}\cdot x \cdot e 
        = \hat{x} \cdot x \cdot e' 
        \;\;\;\implies\;\;\;
        e = e' 
        \quad \qedsymbol
    \]

    \textit{Supondo agora que existam $\hat{x}$ e $\hat{x}'$ que satisfaĂ§am o item 2, temos:}
    \[
        \hat{x}\cdot x 
        = e 
        = \hat{x}'\cdot x 
        \implies
        \hat{x}\cdot x \cdot \hat{x} 
        = \hat{x}'\cdot x \cdot \hat{x} 
        \implies
        \hat{x}\cdot e 
        = \hat{x}'\cdot e 
        \implies
        \hat{x} = \hat{x}' 
        \quad \qedsymbol
    \]
    \item \textit{Notamos por convenĂ§ĂŁo $x^{-1}$ no lugar de $\hat{x}$ no \textbf{item 2} (dada sua unicidade).}
    
    \item \textit{Caso $\forall \,(x,y) \in G \times G:\; x \cdot y = y \cdot x$, dizemos que $G$ ĂŠ um grupo \emph{abeliano} (ou \emph{comutativo}).}
    
    \item \textit{Caso $G$ seja um grupo abeliano, entĂŁo} 
    \[
      (x\cdot y)^n = x^n \cdot y^n, 
      \quad \forall n \in \mathbb{Z}.
    \]
\end{enumerate}

Seja, neste exemplo, para fins de simplificaĂ§ĂŁo, $\mathbb{Z} / n\mathbb{Z} \overset{def}{=} \mathbb{Z}_n$, para $n\in \mathbb{Z}$.

Seja a operaĂ§ĂŁo $\odot$ em $\mathbb{Z}_n$ definida da seguinte forma:

\[
\odot: 
\begin{array}{c}
\mathbb{Z}_n \bigtimes \mathbb{Z}_n \to \mathbb{Z}_n \\
(\overline{a}, \overline{b}) \mapsto \overline{a} \odot \overline{b} = \overline{a\cdot b}
\end{array}
\] 

onde $\cdot$ ĂŠ a operaĂ§ĂŁo usual de produto nos inteiros.

Temos que $(\mathbb{Z}_{p}^{*}, \odot)$, onde $p$ ĂŠ um nĂşmero primo, ĂŠ um grupo abeliano.

\end{example}
\textbf{DemonstraĂ§ĂŁo:}

Por construĂ§ĂŁo, temos que $\overline{a}\odot \overline{b} \in \mathbb{Z}_{p}^{*}$.

Para mostrar a associatividade, sejam $\overline{a}, \overline{b}, \overline{c} \in \mathbb{Z}_{p}^{*}$.

Temos que:

\begin{align*}
	&\overline{a}\odot (\overline{b}\odot \overline{c}) = \overline{a}\odot (\overline{b\cdot c}) =\\[10pt]
	&= \overline{a\cdot (b\cdot c)} = \overline{(a\cdot b)\cdot c} = (\overline{a}\odot \overline{b})\odot \overline{c}.
\end{align*}

O elemento neutro ĂŠ evidentemente o elemento $\overline{1}\in \mathbb{Z}_{p}^{*}$, pois:

\[
\overline{a}\odot \overline{1} = \overline{a\cdot 1} = \overline{a}, \; \forall \overline{a} \in \mathbb{Z}_{p}^{*}.
\]

TambĂŠm temos que para todo elemento de $\mathbb{Z}_{p}^{*}$, existe elemento inverso, pois, sabemos que:

\[
\forall \overline{a}\in \mathbb{Z}_{p}^{*} \implies \text{mdc}(a, p) = 1.
\]

Logo, pelo Teorema de BĂŠzout, temos que existem $x$ e $y$ inteiros tais que:

\[
ax - py = 1
\]

Ora mas isso ĂŠ a mesma coisa que afirmar que existe uma soluĂ§ĂŁo para a equaĂ§ĂŁo: 

\[
a \cdot x \equiv 1 \pmod{p} \iff \overline{a} \odot \overline{x} = \overline{1}. 
\]

Logo, deduzimos que $\forall \overline{a} \in \mathbb{Z}_{p}^{*}, \; \exists \overline{a}^{-1} \in \mathbb{Z}_{p}^{*}$.

AlĂŠm disso, ĂŠ evidente que a operaĂ§ĂŁo $\odot$ ĂŠ comutativa.

Portanto, provamos que $(\mathbb{Z}_{p}^{*}, \odot)$ ĂŠ um grupo abeliano.

\qed


\begin{example}
     Seja $G = \;]-1, 1[, \; (G, \star)$ tal que 
     \[
     \forall x,y \in G \; : x\star y = \frac{x + y}{1 + xy}
     \]
     define um grupo abeliano.
\end{example}

\textbf{DemonstraĂ§ĂŁo:}

Provemos primeiramente que $\forall x, y \in G, x\star y \in G$.

Fixando $y \in G$ temos a seguinte funĂ§ĂŁo de $x \in G$:

\[
f(x) = \frac{x + y}{1 + xy}
\]

A funĂ§ĂŁo ĂŠ derivĂĄvel em $G$. Tomando sua derivada temos:

\[
f'(x) = \frac{1 - y^2}{(1+xy)^2} 
\]

Temos evidentemente $\forall (x, y) \in G \times G, f'(x) > 0$.

(De forma simĂŠtrica podemos mostrar o mesmo escrevendo $f$ como uma funĂ§ĂŁo de $y$).

Logo, deduzimos que a funĂ§ĂŁo $f$ ĂŠ estritamente crescente.

Portanto:

\[
f(-1)<x \star y < f(1) \iff \frac{y-1}{1-y} < x \star y < \frac{1+y}{1+y} \iff -1 <x\star y < 1
\]

Logo, provamos que $x\star y \in G$.

Provemos os outros axiomas:

\textit{ExistĂŞncia do neutro}:

Tomando $y = 0$ temos:

\[
x\star 0 = \frac{x + 0}{1 + 0\cdot x} = x
\]

Portanto, deduzimos que o elemento neutro do grupo $G$ ĂŠ dado por $e = 0$.

\textit{ExistĂŞncia do inverso}:

Tomando $y = -x$ temos:

\[
x\star -x = \frac{x -x}{1 - (-x)x} = 0
\]

Portanto, deduzimos que o elemento inverso do grupo $G$ existe e ĂŠ dado por $x^{-1} = -x$.

\textit{Associatividade}:

Sejam $x, y, z \in G$, mostremos que $(x\star y)\star z = x\star (y\star z)$

Temos:

\begin{align*}
(x\star y)\star z &= \frac{(x\star y) + z}{1+(x\star y)z} = \frac{\frac{x+y}{1+xy} + z}{1 + z\frac{x+y}{1 +xy}} = \frac{x + y+z+xyz}{1+xy+xz+yz} = \\[10pt]
&= \frac{x(1+yz) + (y +z)}{(1+yz)+x(y+z)} = \frac{x+\frac{y+z}{1+yz}}{1+x\frac{y+z}{1+yz}} = x\star (y\star z)
\end{align*}

Mostrando, assim, a associatividade.

Ainda, temos que o grupo ĂŠ evidentemente abeliano.\qed

\section{Subgrupos}

\begin{definition}
    Seja $(G, \cdot)$ um grupo. \textbf{Um subconjunto $H \subseteq G$ ĂŠ chamado de subgrupo de $G$} (denotamos $H\leq G$) se, e somente se, $(H, \cdot)$ ĂŠ um grupo.
\end{definition}
\textit{ObservaĂ§ĂŁo: temos ainda que se $H \subset G$, temos entĂŁo $H$ ĂŠ chamado de subgrupo prĂłprio de $G$ e denotamos como $H< G$.}

\begin{proposition} \hypertarget{prop131}
    Seja $H \subseteq G$ tal que $H \ne \emptyset$ e $(G, \cdot)$ ĂŠ um grupo. $H\leq G$ ĂŠ equivalente Ă  satisfazer as seguintes condiĂ§Ăľes:
    \begin{enumerate}
    \item $h_1 \cdot h_2 \in H,\; \forall (h_1,h_2) \in H \times H$;
    \item $h^{-1} \in H, \; \forall h\in H$.
    \end{enumerate}
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Ă necessĂĄrio mostrarmos as duas implicaĂ§Ăľes da equivalĂŞncia:

\begin{align}
    H\leq G \Longrightarrow (1.)\; \text{e} \;(2.)
\end{align}

\begin{align}
    (1.)\; \text{e} \;(2.) \Longrightarrow H\leq G
\end{align}

A implicaĂ§ĂŁo $(1.1)$ ĂŠ trivial. Ora, se $H\leq G$, entĂŁo pela definiĂ§ĂŁo de subgrupo temos que $h_1\cdot h_2 \in H$ e $h^{-1}\in H$, isto ĂŠ $\exists h^{-1} \in H : h\cdot h^{-1} = h^{-1} \cdot h = h$.

Para a implicaĂ§ĂŁo $(1.2)$:

Sabemos que $H \subseteq G$, logo, se $h_1 \cdot h_2 \in H \Longrightarrow h_1 \cdot h_2 \in G$. Ora, sabemos que $(G, \cdot)$ ĂŠ um grupo. Logo, a associatividade ĂŠ satisfeita. Para demonstrar que $e \in H$, basta tomarmos $h_2 = h^{-1}$ a partir de $(2.)$. Logo, temos $h\cdot h^{-1} = e \in H$. Com isso mostramos todos os axiomas necessĂĄrios e deduzimos que $H\leq G$. \qed

\textbf{DemonstraĂ§ĂŁo:}

Ă evidente que $(n\mathbb{Z}, +)$ sĂŁo subgrupos de $(\mathbb{Z}, +)$. Mostremos que sĂŁo os Ăşnicos! 

Seja $(H, +)$ um subgrupo qualquer de $(\mathbb{Z}, +)$. Se $H = \{0\}$, entĂŁo $H = 0\mathbb{Z}$.

Logo, como $n \in H$ e $H \leq \mathbb{Z}$, temos que $n\mathbb{Z} \subseteq H$. 

De maneira inversa, seja $h \in H$. Logo, pelo Algoritmo de Euclides, existem $q, r \in \mathbb{Z}$ tais que:

\[
h = qn + r \; \; (0 \leq r < n)
\] 

PorĂŠm, note que, como $h \in H$, temos:

\[
r = h - qn \in H  
\]

PorĂŠm, sabemos que $0 \leq r < n$.

Ora, como $n$ ĂŠ o elemento mĂ­nimo de $H$ estritamente maior que $0$, deduzimos que apenas podemos ter $r = 0$.

Logo:

\[
h = qn \implies h \in n\mathbb{Z} \implies H \subseteq n\mathbb{Z}.
\]

Portanto deduzimos que $H = n\mathbb{Z}$. \qed

\begin{example}

\textbf{DemonstraĂ§ĂŁo:}

Como visto na \hyperlink{prop131}{\textbf{ProposiĂ§ĂŁo 1.3.1}}, mostremos que:

\begin{enumerate}
    \item $\forall\; x_1, x_2 \in \bigcap_{i\in I} H _i \implies x_1\cdot x_2 \in \bigcap_{i\in I} H _i$;
    \item $\forall x \in \bigcap_{i \in I} H_i \implies \exists \; x^{-1} \in \bigcap_{i\in I} H _i$.
\end{enumerate}

Provemos o \textbf{item 1}:

Sejam,

\[
x_1, x_2 \in \bigcap_{i\in I} H_i
\]

Logo:

\[
\forall i \in I, \; x_1, x_2 \in H_i
\]

Sabemos tambĂŠm que:

\[
\forall i \in I, \; H_i \leq G 
\]

Portanto, deduzimos que:

\[
\forall i \in I, \; x_1 \cdot x_2 \in H_i
\]

Mas isso ĂŠ a mesma coisa que dizer:

\[
\forall x_1, x_2 \in \bigcap_{i \in I} H_i \implies x_1 \cdot x_2 \in \bigcap_{i\in I} H_i
\]

Provemos o \textbf{item 2}:

Analogamente ao \textbf{item 1}, sabemos que:

\[
x_0 \in \bigcap_{i \in I} H_i \iff \forall i \in I, \; x_0 \in H_i
\]

PorĂŠm, sabemos que:

\[
\forall i \in I, \; H_i \leq G
\]

Logo, deduzimos que:

\[
\forall i \in I, \; x_0 \in H_i, \; \exists x^{-1}_{0} \in H_i   
\]

Mas isso ĂŠ a mesma coisa que:

\[
\forall x \in \bigcap_{i \in I} H_i \implies \exists  x^{-1} \in \bigcap_{i \in I} H_i
\]

Portanto, provamos que:

\[
\bigcap_{i \in I} H_i \leq G
\]  

\qed 

\begin{definition}
    Seja $G$ um grupo. O subconjunto $Z(G)$ tal que: 
    \[
    Z(G) = \{x \in G : xg = gx, \; \forall g \in G\}
    \]
    define um subgrupo de $G$ chamado \textit{centro} de $G$.
\end{definition}


\textbf{DemonstraĂ§ĂŁo:}

Como visto na \hyperlink{prop131}{\textbf{ProposiĂ§ĂŁo 1.3.1}}, para mostrar que $Z(G) \leq G$ ĂŠ necessĂĄrio mostrar que $x\cdot x^{-1} \in Z(G), \; \forall x \in Z(G)$.

Nota: $Z(G)$ ĂŠ claramente nĂŁo vazio uma vez que o elemento neutro comuta com todos elementos de $G$ e, portanto, estĂĄ em $Z(G)$.

Temos que:

Se:
\[
x \in Z(G) \Rightarrow x\cdot g = g\cdot x, \; \forall g\in G.
\]

Logo, teremos:

\[
xgx^{-1} = g \Longrightarrow x^{-1}xgx^{-1} = x^{-1}g \Longrightarrow gx^{-1} =x^{-1}g, \; \forall g \in G 
\]

Portanto: 

\[
x^{-1} \in Z(G)
\]

Temos tambĂŠm que:

\[
x_1 \in Z(G) \Longrightarrow x_1g = gx_1, \; \forall g \in G \;\; \text{(I)}
\]

\[
x_2 \in Z(G) \Longrightarrow x_2g = gx_2, \; \forall g \in G \; \; \text{(II)}
\]

Deduzimos de (I):

\[
x_1g = gx_1 \Longrightarrow g = x^{-1}_1gx_1
\]

Substituindo em (II):

\[
x_2x^{-1}_1gx_1 = x^{-1}_1gx_1x_2 \Longrightarrow x_2x^{-1}_1x_1g = x^{-1}_1gx_1x_2 \Longrightarrow 
\]

\[
\Longrightarrow x_2g = x^{-1}_1gx_1x_2 \Longrightarrow (x_1x_2)g = g(x_1x_2)
\]

Logo, deduzimos que:

\[
(x_1, x_2)\in Z(G)\times Z(G) \Longrightarrow x_1\cdot x_2 \in Z(G)
\]

Portanto, $Z(G) \leq G$. \qed

\textit{ObservaĂ§ĂŁo: O subgrupo centro serve o propĂłsito de "medir a comutatividade" de um dado grupo. Por exemplo, observamos que $Z(\mathbb{Z}) = \mathbb{Z}$, $Z(GL_2 (\mathbb{R})) = \{\lambda I: \lambda \in \mathbb{R}^*\}$ e  $Z(S_n) = \{e\},\; n \geq 3$}.

\begin{definition} \hypertarget{def133}
    Seja $(G, \cdot)$ um grupo e $X$ um conjunto nĂŁo-vazio tal que $X\subseteq G$. \textbf{Chamamos de subgrupo gerado por um subconjunto a interseĂ§ĂŁo de todos os subgrupos de $G$ que contĂŠm $X$.}  Denotamos-o como $\langle X\rangle$.

Matematicamente temos:

\[
\langle X \rangle = \bigcap \{H : H \leq G \; \; \text{e} \; \; X \subseteq H\}
\]

\end{definition}

\begin{proposition}
A partir das notaĂ§Ăľes da \hyperlink{def133}{\textbf{DefiniĂ§ĂŁo 1.3.3}}, temos que $\langle X \rangle$ ĂŠ o menor subgrupo de $G$ que contĂŠm $X$.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo}:

Suponha que $J \leq G$ seja o menor subgrupo de $G$ tal que $X \subseteq J$.

Ora, como $J \leq G$ e $X \subseteq J$, entĂŁo: $\langle X \rangle \subseteq J$.

Entretanto, tambĂŠm sabemos que $J$ ĂŠ o menor subgrupo de $G$ tal que $X \subseteq J$.

Portanto, deduzimos que $J \subseteq H,\; \forall H : H \leq G \; \; \text{e} \; \; X \subseteq H$.

PorĂŠm, para todo $H$ subgrupo de $G$ temos que $X\subseteq H$, logo, deduzimos que $J \subseteq \langle X \rangle$.

Portanto, $J = \langle X \rangle$. 

\qed  


\begin{proposition}
A partir das notaĂ§Ăľes da \hyperlink{def133}{\textbf{DefiniĂ§ĂŁo 1.3.3}}, temos que:
\[
\langle X \rangle = \{x_1 x_2 ... x_n : x_i \in X\cup X^{-1}, \; n \geq 1 \}
\]
\end{proposition}

\textbf{DemonstraĂ§ĂŁo}:

Sejam:

\[
\dot{X} \overset{def}{=} \bigcap \{H : H \leq G \; \; \text{e} \; \; X \subseteq H\}
\]

\[
\bar{X} \overset{def}{=} \{x_1 x_2 ... x_n : x_i \in X\cup X^{-1}, \; n \geq 1 \}
\]

Queremos mostrar que: $\dot{X} = \bar{X}$.

Realizemos, primeiramente, algumas convenĂ§Ăľes de notaĂ§ĂŁo:

\[
\bar{x}_p \overset{def}{=} x_1 x_2 ... x_p, \; p\in \mathbb{Z}^{*}_{+}
\]

\[
\bar{x}_{p}^{-1} \overset{def}{=} x_{1}^{-1} x_{2}^{-1} ... x_{p}^{-1}, \; p \in \mathbb{Z}_{+}^{*}
\]

Ă evidente que $\bar{x}_p,\; \bar{x}_{p}^{-1} \in \bar{X}$. Assim como $\bar{x}_p \bar{x}_{p}^{-1} \in \bar{X}$, o que nos mostra que $\bar{X} \leq G$.

Mostremos que $\dot{X} \subseteq \bar{X}$:

Sabemos que: 

\[
\bar{X} = \{\bar{x}_p : x_i \in X \cup X^{-1}, p\in \mathbb{Z}_{+}^{*} \; \; \text{e} \; \; 1 \leq i \leq p \}
\]

Evidentemente temos que: 

\[
\forall x \in X \implies x \in \bar{X}
\]

Uma vez que $\bar{X} \leq G$, temos diretamente que $\dot{X} \subseteq \bar{X}$.

Isso se dĂĄ pelo fato de que $\dot{X}$ ĂŠ o menor subgrupo de $G$ contendo $X$, e, como $\bar{X}$ ĂŠ um subgrupo de $G$ contendo $X$, realizamos tal deduĂ§ĂŁo.

Mostremos agora que $\bar{X} \subseteq \dot{X}$:

Seja $H \leq G$ tal que: 

\[
H \leq G \; \; \text{e} \; \; X \subseteq H.
\]

Ora, temos evidentemente que:

\[
\forall \bar{x}_p \in \bar{X} \implies \bar{x}_p \in H.
\]

Logo:

\[
\bar{x}_p \in H \implies \bar{x}_p \in \bigcap_{i \in I} H_i
\]

Onde $I$ ĂŠ um conjunto nĂŁo-vazio de Ă­ndices.

Evidentemente temos entĂŁo que $\bar{x}_p \in \dot{X}$.

Logo, $\bar{X} \subseteq \dot{X}$.

Portanto, mostramos que: $\bar{X} = \dot{X}$. 

\qed

    De forma genĂŠrica, para um grupo $G$ e um elemento $a\in G$, temos:

\begin{definition}

\end{definition}

\begin{example}

\begin{proposition} \hypertarget{prop132}
    Se $G$ ĂŠ um grupo cĂ­clico, entĂŁo $G$ ĂŠ um grupo abeliano.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Seja $a \in G$ tal que $G = \langle a\rangle$. Podemos representar $G$ como:

\[
G = \{...,\; (a^{-1})^{r},\; ...,\; (a^{-1})^2, \; a^{-1},\; e, \; a,\; a^2, \; ..., a^r, \; ...\} 
\]

Onde $r \in \mathbb{Z}$.

Sejam $(x,y)\in G \times G$, queremos mostrar que $x\cdot y = y\cdot x$.

Sabemos que:

\[
x = a^{r_1}, \; r_1 \in \mathbb{Z}
\]

\[
y = a^{r_2}, \; r_2 \in \mathbb{Z}
\]

Logo:

\[
x\cdot y = a^{r_1} \cdot a^{r_2} = a^{r_1 + r_2} \stackrel{(*)}{=} a^{r_2+ r_1} = a^{r_2}\cdot a^{r_1} = y\cdot x
\]

\textit{$(*)$ : deduz-se que $r_1 + r_2 = r_2 + r_1$ pois estamos trabalhando dentro do grupo abeliano $(\mathbb{Z}, +)$}.

Portanto, $G$ ĂŠ um grupo abeliano. \qed

	Podemos tambĂŠm denotar $G' \overset{def}{=} [G,G]$, onde, para $x,y \in G$, temos:

	\[
	[x,y] \overset{def}{=} xyx^{-1}y^{-1}.
	\]
\end{definition}

\textit{ObservaĂ§Ăľes}:

\item \textit{Note que, para um dado grupo $G$ e $x,y\in G$, \textbf{nĂŁo} necessariamente temos [x,y]=[y,x].}

\end{itemize}

\begin{definition}
    Seja $(G, \cdot)$ um grupo. \textbf{Definimos ordem do grupo $(G, \cdot)$ a quantidade de elementos no conjunto $G$} e a denotamos por $\vert G \vert$. \\
    Se $\alpha \in G$, \textbf{a ordem de $\alpha$ ĂŠ a ordem do subgrupo gerado por $\alpha$}, denotada por $\mathcal{O}(\alpha)$, isto ĂŠ,  $ \mathcal{O}(\alpha) = \vert \langle \alpha \rangle \vert$.
\end{definition}

\begin{example}
    $\vert \mathbb{Z} \vert = \infty, \; \vert\mathbb{Z}/n\mathbb{Z}\vert = n, \; \vert(\mathbb{Z}/p\mathbb{Z})^* \vert = p-1, \; \vert S_n \vert = n!$
\end{example}

\begin{proposition} \hypertarget{prop135}
	Seja $G$ um grupo finito e $\alpha$ um elemento de $G$. 

	Logo, $\mathcal{O}(\alpha) < \infty$.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Provemos a \hyperlink{prop135}{\textbf{ProposiĂ§ĂŁo 1.3.5}} via absurdo.

Suponha que $\mathcal{O}(\alpha)$ seja nĂŁo finito, logo podemos gerar $n$ valores distintos a partir de potĂŞncias de $\alpha$, onde $n \in \mathbb{Z}$.

Ora, a partir da geraĂ§ĂŁo de infinitos valores distintos de potĂŞncias de $\alpha$, sabemos que, para dado valor inteiro $k$, teremos $\alpha ^k \notin G$. Ora, mas $\langle \alpha \rangle$ ĂŠ um subgrupo de $G$. Absurdo.

Portanto, temos que $\mathcal{O}(\alpha) < \infty$. 

\qed  

\begin{proposition} \hypertarget{prop136}
Seja $G$ um grupo e $\alpha$ um elemento de $G$. EntĂŁo, as seguintes proposiĂ§Ăľes sĂŁo equivalentes:

$(i)$ A ordem $\mathcal{O}(\alpha)$ ĂŠ finita. Isto ĂŠ, $\mathcal{O}(\alpha) < \infty$;

$(ii)$ $\exists t \in \mathbb{Z}_{+}^{*} : \alpha ^t = e$, onde $t = \min{\{k \in G : k > 0\}}$.

Note que, caso a proposiĂ§ĂŁo seja satisfeita, entĂŁo temos $\mathcal{O}(\alpha) = t$.

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Queremos provar que: $(i) \iff (ii)$.

Comecemos provando a implicaĂ§ĂŁo $(i) \Longrightarrow (ii):$

Temos, por definiĂ§ĂŁo, que $\langle \alpha \rangle = \{\alpha ^m \; \vert \; m \in \mathbb{Z}\}$. 

Como $\mathcal{O}(\alpha) < \infty$, temos que $\exists p, q \in \mathbb{Z} : p > q$ e $\alpha ^p = \alpha ^q$.

Deduzimos diretamente que: $\alpha ^{p-q} = e$. Como $p-q \in \mathbb{Z}^{*}_{+}$, mostramos $(i) \Longrightarrow (ii)$.

Note que a escolha do valor $p-q$ ocorre sem perda de generalidade, uma vez que o conjunto $\mathbb{Z}_{+}^{*}$ ĂŠ enumerĂĄvel e sempre podemos garantir a minimalidade de $p-q$. 

Provemos $(ii) \Longrightarrow (i):$

Seja $\alpha \in G$ tal que $\langle \alpha \rangle = \{ \alpha ^n : n\in\mathbb{Z}\}$, sabemos via o Algoritmo de Euclides que: 

\[
n = qt + r, \; q,t\in \mathbb{Z} \; \; \text{e} \; \; r\in \llbracket 0, t-1 \rrbracket
\]

Temos, portanto:

\[
\alpha ^n = \alpha ^{qt + r} = \alpha ^{qt} \alpha ^{r} = (\alpha ^{t})^{q} \alpha ^{r} = e^q \alpha ^r = \alpha ^r
\]

Portanto $\alpha ^n = \alpha ^r, \; \forall n\in \mathbb{Z}$ e $r\in \llbracket 0, t-1\rrbracket$.

Logo, deduzimos que:

\[
\langle \alpha \rangle = \{\alpha^0, \ldots, \alpha ^{t-1} \} \Longrightarrow \mathcal{O}(\alpha) < \infty
\]

Para mostrar que de fato temos $\mathcal{O}(\alpha) = t$, mostremos que $\alpha ^{r_1} \neq \alpha ^{r_2}, \; \forall r_1,r_2 \in \llbracket 0,t-1 \rrbracket$.

Suponha, por absurdo, que $\exists r_1, r_2 \in \llbracket 0,t-1 \rrbracket$ tal que $\alpha ^{r_1} = \alpha ^{r_2}$.

Temos, portanto:

\[
\alpha ^{r_1} = \alpha ^{r_2} \iff \alpha ^{r_1 - r_2} = e
\]

Absurdo, pois $r_1 - r_2 < t$, porĂŠm $t$ ĂŠ minimal.

Logo, como todos elementos de $\langle \alpha \rangle$ sĂŁo distintos, deduzimos que $\mathcal{O}(\alpha) = t$.

\qed

\section{Teorema de Lagrange}

\begin{definition}
    Seja $G$ um grupo e $H$ um subgrupo de $G$. \textbf{Definimos classe lateral Ă  esquerda de $H$ em $G$ que contĂŠm $x$ o subconjunto $xH$} de $G$ tal que $\forall x \in G$:
    \[
    xH = \{xh \; \vert \; h\in H \}
    \]
    Analogamente definimos \textbf{classe lateral Ă  direita de $H$ em $G$ que contĂŠm $x$ o subconjunto $Hx$} de $G$ tal que $\forall x \in G$:
    \[
    Hx = \{ hx \; \vert \; h \in H\}
    \]
\end{definition}


\textit{ObservaĂ§Ăľes:}
    \begin{itemize}
        \item \textit{As classes laterais de $G$ nĂŁo sĂŁo necessariamente subgrupos de $G$;}
        \item \textit{Quando nĂŁo houver confusĂŁo possĂ­vel, podemos denominar as classes laterais Ă  esquerda/direita de $H$ em $G$ que contĂŠm $x$ como simplesmente: classe lateral Ă  esquerda/direita de $H$}.
    \end{itemize}

\begin{definition} \hypertarget{def142}
	A cardinalidade do conjunto das classes laterais Ă  esquerda \textit{ou} Ă  direita ĂŠ definida como \textbf{o Ă­ndice de $H$ em $G$}, e serĂĄ denotada por $[G:H]$.
\end{definition} 

\textit{ObservaĂ§ĂŁo: note que o nĂşmero de classes laterais Ă  direita de $H$ ĂŠ igual ao nĂşmero de classes laterais Ă  esquerda de $H$ (por mais que as classes laterais sejam diferentes).} 

\textit{Isto se dĂĄ pelo fato de que a funĂ§ĂŁo:} 

\[
\phi: 
\begin{array}{c}
\{\text{classes lat. Ă  esquerda}\} \to \{\text{classes lat. Ă  direita} \} \\
xH \mapsto Hx^{-1}
\end{array}
\]

\textit{ĂŠ claramente uma bijeĂ§ĂŁo.}

\begin{theorem} \hypertarget{lagrange}
{\bfseries Teorema de Lagrange (Grupos)}\normalfont

    Seja $G$ um grupo finito e $H$ um subgrupo de $G$.

    Logo, $\vert H\vert$ divide $\vert G \vert$.
\end{theorem}

\textbf{DemonstraĂ§ĂŁo:}


%A ideia da demonstraĂ§ĂŁo se baseia em, a partir de um subgrupo $H$ de $G$, formar partiĂ§Ăľes de $G$ duas-a-duas disjuntas tais que a cardinalidade dessas partiĂ§Ăľes sejam iguais Ă  ordem de $H$.

%Isto ĂŠ, se $\{X_i \}_{i\in\mathbb{N}}$ representa uma famĂ­lia de subconjuntos de $G$ e $H$ ĂŠ um subgrupo de $G$, entĂŁo queremos provar que:

%\[
%\bigcup_{i \in \mathbb{N}}X_i = G; \; \; \bigcap_{i \in \mathbb{N}}X_i = \emptyset\; \; \text{e} \; \; \text{Card}(X_i) = \vert H\vert, \; \forall i \in \mathbb{N}.
%\]

Seja $x\in G \backslash H$, consideremos o conjunto das classes laterais Ă  esquerda de $H$:

\[
xH = \{xh \; \vert \; h \in H\}
\]

Mostremos que $H\cap xH = \emptyset$:

Supondo $\alpha \in H\cap xH$:

\[
\alpha \in H\cap xH \iff \alpha = xh \in H.
\]

Como $\alpha =xh \in H$, logo $\exists h^{-1}\in H$ tal que $hh^{-1} \in H$

Portanto:

\[
\alpha h^{-1} = xhh^{-1} \in H \iff x \in H \Longrightarrow \text{Absurdo, pois   } x\in G\backslash H.
\]

Logo, $H\cap xH = \emptyset$.

Agora mostremos que $\text{Card}(xH) = \vert H\vert$:

Seja $\zeta$ a funĂ§ĂŁo definida abaixo:

\[
\zeta: 
\begin{array}{c}
H \to xH \\
h \mapsto xh
\end{array}
\]

A funĂ§ĂŁo $\zeta$ ĂŠ claramente sobrejetiva por definiĂ§ĂŁo. 

$\zeta$ tambĂŠm ĂŠ injetiva pois se $(xh_1, xh_2) \in (xH)^2$:

\[
xh_1 = xh_2 \Longrightarrow x^{-1}xh_1 = x^{-1}xh_2 \Longrightarrow h_1 = h_2.
\]

Portanto, deduzimos que $\text{Card}(xH) = \vert H\vert$.

Consideremos agora o conjunto $yH$ das classes laterais Ă  esquerda de $H$ em $G$ que contĂŠm $y$ tal que $y \notin H\cup xH$.

JĂĄ mostramos anteriormente que $y \notin H$.

Mostremos que $yH \cap xH = \emptyset$

Supondo $\beta \in yH \cap xH$:

EntĂŁo $\beta$ pode ser escrito de duas formas:

\[
\beta = yh_1
\]

\[
\beta = xh_2
\]

Logo, temos:

\[
yh_1 = xh_2 \Longrightarrow y = xh_2h_1^{-1} \in xH \Longrightarrow \text{Absurdo, pois  } y \notin H\cup xH.
\]

Analogamente ao passo anterior podemos provar que $\text{Card}(yH) = \text{Card}(xH) = \vert H\vert$.

Portanto, realizando os passos acima sucessivamente, criamos partiĂ§Ăľes de $G$.

Como $G$ ĂŠ finito, o processo terĂĄ finalizado apĂłs $n$ etapas.

Portanto, temos: $\vert G\vert = n\vert H\vert$. \qed

\textit{ObservaĂ§Ăľes:} 

\begin{enumerate}
	\item \textit{Segue como consequĂŞncia direta do \textbf{Teorema de Lagrange} que caso $G$ seja um grupo finito e $\alpha \in G$, entĂŁo $\mathcal{O}(\alpha)$ divide $\vert G\vert$.}
	\item \textit{Temos diretamente pela \hyperlink{def142}{\textbf{DefiniĂ§ĂŁo 1.4.2}} que: $\vert G\vert = \vert H\vert [G:H]$.}
\end{enumerate}

\begin{corollary}
	Seja $G$ um grupo nĂŁo finito e $H \leq G$. 

	EntĂŁo vale o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}}. 
\end{corollary}

\textbf{DemonstraĂ§ĂŁo:}

Demonstraremos novamente o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} de forma que o corolĂĄrio acima possa ser justificado de forma clara. 

Seja $G$ um grupo e $H \leq G$.

Ora, sabemos que:

\[
\text{Ou} \; \; xH = yH \; \; \text{ou} \; \; xH \cap yH = \emptyset, \; \forall (x,y) \in G \times G.
\]

Sabemos tambĂŠm que, sendo $I$ um conjunto nĂŁo vazio de Ă­ndices tal que Card($I$) $= [G:H]$:

\[
\bigsqcup_{i \in I} x_i H = G.
\]

Como $\vert G\vert$ ĂŠ nĂŁo finito, entĂŁo se $\vert H\vert$ ou $[G:H]$ sĂŁo nĂŁo finitos, vale que $\vert G\vert = \vert H\vert[G:H]$.

Suponhamos, agora, que $\vert H\vert < \infty$ e $[G:H] < \infty$.

Como $[G:H] < \infty$, entĂŁo $\vert I\vert < \infty$.

Logo, podemos escrever $I$ como:

\[
I = \{i_1, i_2, \dots, i_n, \; n\in \mathbb{N} \}.
\]

Logo:

\[
G = \bigsqcup_{i_1 \leq i \leq n} x_i H.
\]

Portanto, podemos escrever:

\[
\vert G\vert = \sum_{k = 1}^{n} \vert x_{i_k} H\vert.
\]

Ora, deduzimos na demonstraĂ§ĂŁo do \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} que $\vert xH\vert = \vert H\vert, \; \forall x \in G$.

Portanto temos que:

\[
\vert G\vert = n\vert H\vert.
\]

Ora, mas $\vert G\vert$ ĂŠ nĂŁo finito e $\vert H\vert < \infty$. Absurdo !

Portanto, deduzimos que $\vert H\vert$ ou $[G:H]$ sĂŁo nĂŁo finitos.

Assim, provamos que o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} vale tambĂŠm para $\vert G\vert$ nĂŁo finito. Isto ĂŠ:

\[
\vert G\vert = \vert H\vert [G:H].
\]

\qed

\hypertarget{prop141}{}
\begin{proposition}
    Seja $G$ um grupo finito de ordem $p\in \mathbb{N}^*$. \\
    Se $p$ for primo, entĂŁo $G$ ĂŠ um grupo cĂ­clico.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Pelo \hyperlink{lagrange}{Teorema de Lagrange} sabemos que se $H$ ĂŠ subgrupo de um grupo finito $G$, entĂŁo $\vert H\vert$ divide $\vert G\vert$.

Como $\vert G\vert = p$ primo, entĂŁo os Ăşnicos subgrupos possĂ­veis de $G$ sĂŁo seus subgrupos triviais.

Seja $x\in G$ tal que $x \neq e$, onde $e$ ĂŠ o elemento neutro de $G$.

Logo, o Ăşnico subgrupo gerado por $x$ ĂŠ o prĂłprio $G$, $\langle x\rangle = G$ \qed

\textit{ObservaĂ§ĂŁo: como visto na \hyperlink{prop132}{\textbf{ProposiĂ§ĂŁo 1.3.2}}, $G$ tambĂŠm ĂŠ abeliano!}

\hypertarget{teoremadeeuler}{}

\begin{theorem} 
	\textbf{Teorema de Euler (Grupos)} \\
	Seja $(G, \cdot)$ um grupo finito tal que $\vert G \vert = n, \; n\in \mathbb{Z}$.
	EntĂŁo:
	\[
	\forall g \in G, \; g^n = 1.
	\]
\end{theorem}

\textbf{DemonstraĂ§ĂŁo:}

Seja $g$ um elemento do grupo finito $G$. Sabemos que $\langle g \rangle \leq G$. Sabemos tambĂŠm, pelo \hyperlink{lagrange}{\textbf{Teorema de Lagrange}} que $\mathcal{O}(g)$ divide a ordem de $G$. 

Ora, podemos entĂŁo escrever:

\[
\vert G\vert = k\mathcal{O}(g), \; k\in \mathbb{Z} 
\]

PorĂŠm, pela \hyperlink{prop136}{\textbf{ProposiĂ§ĂŁo 1.3.6}}, deduzimos:

\[
g^{n} = g^{\vert G \vert} = g^{k\mathcal{O}(g)} = \left(g^{\mathcal{O}(g)}\right)^{k} = e^k = e
\]

Ora, demonstramos, com o argumento acima, sem perda de generalidade, tal fato para qualquer elemento de $G$.

\qed

\hypertarget{pequenofermat}{}

\begin{theorem} 
    \textbf{Pequeno Teorema de Fermat} \\
   	Seja $p$ um nĂşmero primo e $a\in \mathbb{Z} \setminus p\mathbb{Z}$, entĂŁo:
	\[
	a^{p-1} \equiv 1 \pmod{p}.
	\] 
\end{theorem}

\textbf{DemonstraĂ§ĂŁo:}

O \hyperlink{pequenofermat}{\textbf{Pequeno Teorema de Fermat}} ĂŠ evidentemente o caso especĂ­fico do \hyperlink{teoremadeeuler}{\textbf{Teorema de Euler}} em que $(G, \cdot) = ((\mathbb{Z}/p\mathbb{Z})^*, \odot)$.

\qed

\begin{proposition}
	Seja $G$ um grupo e sejam $K < H < G$.

	Logo $[G : K] = [G : H][H : K].$
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Basta aplicar sucessivamente o \hyperlink{lagrange}{\textbf{Teorema de Lagrange}}:

\begin{align*}
& H < G &&\Rightarrow && \lvert G \rvert = \lvert H \rvert \cdot [G : H] \; \; \; \text{(I)} \\
& K < H &&\Rightarrow && \lvert H \rvert = \lvert K \rvert \cdot [H : K] \; \; \; \text{(II)} \\
& K < G &&\Rightarrow && \lvert G \rvert = \lvert K \rvert \cdot [G : K] \; \; \; \text{(III)}
\end{align*}

Combinando as expressĂľes (I) e (II), obtemos:

\[
\lvert G \rvert = \lvert K \rvert \cdot [H : K] \cdot [G : H] = \lvert K \rvert \cdot [G : K]
\]

Portanto:

\[
[G : K] = [G : H] \cdot [H : K]
\]

\qed

\begin{definition}
	Seja $G$ um grupo e $H\leq G$. 

	\textbf{Definimos a seguinte operaĂ§ĂŁo entre as classes laterais Ă  esquerda de $H$ em $G$:}

\begin{array}{c}

\end{array}

\end{definition}

\textit{ObservaĂ§Ăľes:}

\begin{itemize}
	\item \textit{Note que, por convenĂ§ĂŁo, estamos tratando das classes laterais Ă  esquerda de $H$ em $G$. Entretanto, a definiĂ§ĂŁo ĂŠ vĂĄlida para classes laterais Ă  direita tambĂŠm.}
	\item \textit{A operaĂ§ĂŁo definida ĂŠ uma \textbf{lei de composiĂ§ĂŁo interna} por construĂ§ĂŁo.}
	\item \textit{NĂŁo mostramos ainda que a operaĂ§ĂŁo estĂĄ de fato bem definida, isto ĂŠ:}
	
	\[
	\begin{cases}
	(x_1 H, y_1 H) \in G/H \times G/H \\	
	(x_2 H, y_2 H) \in G/H \times G/H
	\end{cases}
	\]
	\[
	\text{\textit{Se}} \; \; (x_1 H, y_1 H) = (x_2 H, y_2 H) \implies x_1 y_1 H = x_2 y_2 H.
	\]
	\textit{Tal fato serĂĄ destacado na \hyperlink{prop151}{\textbf{ProposiĂ§ĂŁo 1.5.1}}.}
\end{itemize}

\hypertarget{prop151}{}

\begin{proposition}
Seja $G$ um grupo e $H\leq G$. 

As afirmaĂ§Ăľes a seguir sĂŁo equivalentes:

$(i)$ A operaĂ§ĂŁo definida na \hyperlink{def152}{\textbf{DefiniĂ§ĂŁo 1.5.2}} estĂĄ bem definida;

$(ii)$ $gHg^{-1} \subseteq H, \; \forall g \in G$;

$(iii)$ $gHg^{-1} = H, \; \forall g \in G$;

$(iv)$ $gH = Hg, \; \forall g \in G$.

\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos as equivalĂŞncias:

\[
(i) \iff (ii) \; \; \; \text{(I)}
\]

\[
(ii) \iff (iii) \; \; \; \text{(II)}
\]

\[
(iii) \iff (iv) \; \; \; \text{(III)}
\]

Comecemos mostrando a equivalĂŞncia (I):

Ora, perceba que para $(x,y) \in G\times G$ e $(h, h')\in H\times H$, temos que:

$x$ e $xh$ sĂŁo representantes distintos para a mesma classe lateral $xH$.

$y$ e $yh'$ sĂŁo representantes distintos para a mesma classe lateral $yH$.

Portanto, podemos deduzir que a operaĂ§ĂŁo "$\bullet$" definida na \hyperlink{def152}{\textbf{DefiniĂ§ĂŁo 1.5.2}} sĂł estarĂĄ bem definida se, e somente se:

\[
xyH = xhyh'H, \; \forall (x,y)\in G\times G \; \; \; \text{e} \; \; \; \forall (h,h')\in H\times H.
\]

Logo:

\[
xyH = xhyh'H \iff y^{-1}x^{-1}xyH = y^{-1}x^{-1}xhyh'H \iff H = y^{-1}hyh'H.
\]

Ora, mas isso ĂŠ equivalente Ă  dizer que a operaĂ§ĂŁo sĂł estarĂĄ bem definida se, e somente se:

\[
ghg^{-1} \in H,\; \forall g\in G, \; \forall h\in H.
\]

Com isso mostramos a equivalĂŞncia (I).

Mostremos a equivalĂŞncia (II):

Para a implicaĂ§ĂŁo $(ii) \implies (iii)$ mostremos que:

\[
gHg^{-1} \subseteq H \implies H \subseteq gHg^{-1}, \; \forall g \in G.
\]

Ora, temos diretamente da hipĂłtese:

\[
gHg^{-1} \subseteq H \implies g^{-1}Hg \subseteq H \implies g(g^{-1}Hg)g^{-1} \subseteq gHg^{-1}
\]

\[
\implies H \subseteq gHg^{-1}
\]

Logo, podemos concluir que:

\[
gHg^{-1} \subseteq H \implies H = gHg^{-1}.
\]

A implicaĂ§ĂŁo $(iii) \implies (ii)$ ĂŠ evidente.

Com isso mostramos a equivalĂŞncia (II).

Mostremos a equivalĂŞncia (III):

\[
gHg^{-1} = H \iff gHg^{-1}g = Hg \iff gH = Hg.
\]

Assim mostramos a equivalĂŞncia (III).

Tendo mostrado as equivalĂŞncias (I), (II) e (III), mostramos que todas as afirmaĂ§Ăľes sĂŁo duas a duas equivalentes. 

\qed

\begin{proposition}

\begin{proposition}

\vert H\vert = \vert H\cap K\vert \cdot [H : H\cap K] \iff \frac{\vert H \vert}{\vert H\cap K\vert} = [H : H\cap K]
\]

Pela \hyperlink{prop156}{\textbf{ProposiĂ§ĂŁo 1.5.6}}, temos diretamente:

\[
\vert HK\vert = \frac{\vert H\vert \vert K\vert}{\vert H\cap K\vert} \iff \frac{\vert HK\vert}{\vert K\vert} = \frac{\vert H\vert}{\vert H\cap K\vert}
\]

Logo, deduzimos que:

\[
[HK : K] = [H : H\cap K]
\]

\qed

\begin{definition}
Seja $G$ um grupo e $H\trianglelefteq G$.

\textbf{EntĂŁo $(G/H, \bullet)$ ĂŠ um grupo chamado de grupo quociente.}
\end{definition}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos que $(G/H, \bullet)$ ĂŠ de fato um grupo.

Ora, pela \hyperlink{def152}{\textbf{DefiniĂ§ĂŁo 1.5.2}} sabemos que a operaĂ§ĂŁo "$\bullet$" ĂŠ uma lei de composiĂ§ĂŁo interna por construĂ§ĂŁo. Sabemos tambĂŠm, pela \hyperlink{prop151}{\textbf{ProposiĂ§ĂŁo 1.5.1}} que caso $H\trianglelefteq G$, entĂŁo "$\bullet$" estĂĄ bem definida. 

Nos resta mostrar que $(G/H, \bullet)$ satisfaz os axiomas de grupo.

De fato, $G/H$ ĂŠ associativo em relaĂ§ĂŁo Ă  operaĂ§ĂŁo "$\bullet$" pois:

Sejam $xH, yH, zH \in G/H$, temos:

\[
(xH\bullet yH)\bullet zH = (xy)H \bullet zH = (xy)zH \overset{(*)}{=} x(yz)H = xH\bullet(yH\bullet zH).
\]

TambĂŠm temos evidentemente que o elemento neutro ĂŠ dado por $H \in G/H$.

Ora, seja $xH\in G/H$, temos evidentemente:

\[
xH\bullet H = xH = H\bullet xH.
\]

Por fim, temos que para dado $xH \in G/H$, seu elemento inverso ĂŠ dado por $x^{-1}H \in G/H$. 

De fato:

\[
xH\bullet x^{-1}H = xx^{-1}H = H = x^{-1}xH = x^{-1}H\bullet xH.
\]

Portanto, $(G/H, \bullet)$ ĂŠ de fato um grupo.

\qed

\textit{ObservaĂ§ĂŁo: muitas vezes iremos denotar $\bar{x} \overset{def}{=} xH$ para elementos de $G/H$ por uma questĂŁo de simplificaĂ§ĂŁo de escrita.}

\begin{example}
O grupo $\mathbb{Z}/n\mathbb{Z}$ ĂŠ um grupo quociente formado pelo quociente entre $\mathbb{Z}$ e $n\mathbb{Z}\trianglelefteq \mathbb{Z}$.
\end{example}

\begin{proposition}
	Seja $G$ um grupo e $Z(G)$ seu centro. 

	Se o grupo $G/Z(G)$ for cĂ­clico, entĂŁo $G = Z(G)$.
\end{proposition}
	

\textbf{DemonstraĂ§ĂŁo:}

Seja $G/Z(G)$ um grupo cĂ­clico. 

Mostremos que $G = Z(G)$, isto ĂŠ, que $G$ ĂŠ um grupo abeliano.

Como $G/Z(G)$ ĂŠ cĂ­clico, entĂŁo existe $x\in G/Z(G)$ tal que $\langle x\rangle = G$.

Portanto, sabemos que, para determinado inteiro $k\in \mathbb{Z}$, temos:

\[
gz' = x^k, \; (g,z')\in G\times Z(G)
\]

Podemos reescrever a expressĂŁo acima como:

\[
g = x^kz, \; (g,z)\in G\times Z(G)
\]

Tomemos $g_1, g_2 \in G$ tal que:

\[
g_1 \overset{def}{=} x^{k_1}z_1, \; (k_1, z_1) \in \mathbb{Z}\times Z(G)
\]

\[
g_2 \overset{def}{=} x^{k_2}z_2, \; (k_2, z_2) \in \mathbb{Z}\times Z(G)
\]

Portanto temos:

\[
g_1g_2 = x^{k_1}z_1x^{k_2}z_2 = x^{k_1 + k_2}z_1z_2 = x^{k_2}x^{k_1}z_1z_2 = x^{k_2}z_2x^{k_1}z_1 = g_2g_1. 
\]

Isso se dĂĄ pelo fato de qualquer elemento de $z\in Z(G)$ comutar com elementos de $G$ e, portanto, elementos de $G/Z(G)$.

Com isso, mostramos que $G$ ĂŠ um grupo abeliano e, portanto, $G = Z(G)$. 

\qed

\begin{proposition}
Seja $G$ um grupo e $G'$ o seu subgrupo de comutadores. 

O grupo quociente $G/G'$ ĂŠ abeliano.

Este grupo ĂŠ denominado como \textit{abelianizaĂ§ĂŁo} de $G$ e ĂŠ denotado como $G^{ab}$.

Note que $G'$ ĂŠ o menor subgrupo normal de $G$ com essa propriedade e $G^{ab}$ representa o maior quociente abeliano de $G$.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Note, primeiramente, que $G^{ab}$ ĂŠ de fato um grupo, uma vez que $G' \trianglelefteq G$.

Sejam $\bar{x} = xG', \bar{y} = yG'$ elementos de $G^{ab}$, com $x,y\in G$, sabemos que:

\[
xyx^{-1}y^{-1} \in G' \iff xy(yx)^{-1} \in G' \iff xyG' = yxG' \iff \bar{x}\bar{y}=\bar{y}\bar{x}
\]

ConcluĂ­mos, portanto, que $G^{ab}$ ĂŠ um grupo abeliano.

Para mostrar que $G'$ ĂŠ o menor subgrupo normal de $G$ tal que $G/G'$ ĂŠ abeliano, basta mostrar que para quaisquer $H\trianglelefteq G$, se $G/H$ for abeliano, entĂŁo $G'\subseteq H$.

Suponha, por absurdo, que $G/H$ ĂŠ um grupo abeliano e $G'\nsubseteq H$.

Ora, entĂŁo temos, para $x,y\in G$:

\[
xyH = yxH \iff xy(yx)^{-1} \in H \iff xyx^{-1}y^{-1} \in H
\]

Sabemos que $xyx^{-1}y^{-1}\in G'$, mas $G' \nsubseteq H$, portanto induzimos um absurdo.

ConcluĂ­mos que $G' \subseteq H$.

\qed

\section{Homomorfismos de Grupos}

\begin{definition}
Sejam $(G, \cdot)$ e $(\mathcal{G}, *)$ dois grupos. \textbf{Uma funĂ§ĂŁo $\varphi : G \rightarrow \mathcal{G}$ ĂŠ chamada de homomorfismo de grupos se:}

\[
\varphi(a\cdot b) = \varphi(a)*\varphi(b), \; \forall a,b \in G
\]
\end{definition}

\textit{ObservaĂ§Ăľes:}

\textit{Note que, se $\varphi : G \rightarrow \mathcal{G}$} for um homomorfismo de grupos, entĂŁo:

\begin{enumerate}
	\item $\varphi(e_G) = e_\mathcal{G}$;
	\item $\varphi(x^{-1}) = \varphi(x)^{-1}.$
\end{enumerate}

\textit{De fato, note que, para o item 1., temos que}:

\[
\varphi(e_G) = \varphi(e_G \cdot e_G) = \varphi(e_G)*\varphi(e_G) \implies \varphi(e_G) = e_\mathcal{G}
\]

\textit{E para o item 2., temos:}

\[
e_\mathcal{G} = \varphi(e_G) = \varphi(x\cdot x^{-1}) = \varphi(x)*\varphi(x^{-1}) \implies \varphi(x^{-1}) = \varphi(x)^{-1}
\]

\textit{$*$ Note tambĂŠm que muitas vezes as operaĂ§Ăľes de $G$ e de $\mathcal{G}$ serĂŁo confundidas para fins de simplificaĂ§ĂŁo, isto ĂŠ: $\varphi(ab) = \varphi(a)\varphi(b)$.}

\begin{definition}
Sejam $G$ e $\mathcal{G}$ dois grupos. Definimos $\operatorname{Hom}(G,\mathcal{G})$ como o \textbf{conjunto de todos os homomorfismos de $G$ em $\mathcal{G}$}.
\end{definition}

\begin{example}
$\text{Id}: (G, \cdot) \rightarrow (G, \cdot), \; \text{Id}(g) = g$ ĂŠ um homomorfismo chamado \textit{identidade}.
\end{example}

\begin{example}
$\theta: G \rightarrow \mathcal{G},\; \theta (g) = e_{\mathcal{G}}, \; \forall g\in G$, ĂŠ um homomorfismo chamado \textit{homomorfismo trivial}.
\end{example}

\begin{example}
A funĂ§ĂŁo:

\[
\text{det} \colon \begin{cases}
GL_n(\mathbb{K}) \to \mathbb{K}^* \\
A \mapsto \text{det}(A)
\end{cases}
\]

ĂŠ um homomorfismo de grupos, pois sabemos que $\forall A, B\in GL_n(\mathbb{K})$ temos:

\[
\text{det}(AB) = \text{det}(A)\text{det}(B).
\]
\end{example}

\begin{example}
Seja $G$ um grupo e $N \trianglelefteq G$, logo a funĂ§ĂŁo:

\[
\pi \colon \begin{cases}
G \to G/N \\
x \mapsto xN
\end{cases}
\]

ĂŠ um homomorfismo chamado de \textit{projeĂ§ĂŁo canĂ´nica}.
\end{example}

\textbf{DemonstraĂ§ĂŁo:}

Sejam $x,y \in G$, ora:

\[
\pi(xy) = xyN = xN\bullet yN = \pi(x)\bullet \pi(y).
\]
\qed

\begin{example}
Seja $G = \mathbb{R}_{+}^{*}\times \mathbb{R}$ munido da operaĂ§ĂŁo $\heartsuit$ definida por:

\[
(a,b) \heartsuit (a', b') = (aa', b+b'), 
\]
\[
(a,b), (a',b') \in G
\]

A funĂ§ĂŁo:

\[
\spadesuit \colon \begin{cases}
(G, \heartsuit) \to (\mathbb{C}^*, \cdot) \\
(r, \theta) \mapsto re^{i\theta}
\end{cases}
\] 

ĂŠ um homomorfismo de grupos, onde $e$ ĂŠ o nĂşmero de Euler, $i^2 = -1$ e:
\[
e^{i\theta} \overset{def}{=} \cos \theta + i \sin \theta, \; \forall \theta \in \mathbb{R} 
\]
\end{example}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos primeiramente que $(G, \heartsuit)$ ĂŠ de fato um grupo.

Por construĂ§ĂŁo temos diretamente que $\heartsuit$ ĂŠ uma lei de composiĂ§ĂŁo interna em $G$.

TambĂŠm sabemos que, pelo fato de $\mathbb{R}_{+}^{*}$ ser um grupo com o produto usual e $\mathbb{R}$ ser um grupo com a adiĂ§ĂŁo usual, temos diretamente que a dupla formada pela operaĂ§ĂŁo $\heartsuit$ ĂŠ um grupo.

Portanto, mostramos que $(G, \heartsuit)$ ĂŠ de fato um grupo.

Mostremos que $\spadesuit$ ĂŠ um homomorfimo de grupos.

Note, primeiramente, que $\spadesuit$ ĂŠ claramente uma funĂ§ĂŁo de $G$ em $\mathbb{C}^*$.

Sejam $(r, \theta), (r', \theta') \in G$, temos que:

\[
\spadesuit((r, \theta)\heartsuit(r', \theta')) = \spadesuit(rr',\theta + \theta') = rr'e^{i(\theta + \theta')} = \spadesuit(r, \theta)\cdot \spadesuit(r', \theta')
\]

Logo, $\spadesuit$ ĂŠ de fato um homomorfismo de grupos.

\qed

\begin{definition}
Definimos o \textbf{nĂşcleo de um homomorfismo $\varphi$ o subconjunto $\ker \varphi \subseteq G$, tal que:}

\[
\ker \varphi \overset{def}{=} \{x \in G,\; \varphi(x) = e_\mathcal{G}\}.
\]
\end{definition}

\begin{proposition}
O nĂşcleo de um homomorfismo $\varphi$ ĂŠ um subgrupo normal de $G$.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Note, primeiramente, que $\ker \varphi$ ĂŠ nĂŁo-vazio, uma vez que sempre teremos $\varphi(e_G) = e_\mathcal{G}$.

Mostremos, portanto, que $\ker \varphi \leq G$. 

Para isso, mostremos que $x,y \in \ker \varphi \implies xy^{-1} \in \ker \varphi$. 

Sejam $x, y \in \ker \varphi$, temos:

\[
\varphi(x\cdot y^{-1}) = \varphi(x) * \varphi(y^{-1}) = \varphi(x)*\varphi(y)^{-1} = e_\mathcal{G} * e_\mathcal{G} = e_\mathcal{G};\]

Deduzimos que $\ker \varphi \leq G$, mostremos, por fim, que $\ker \varphi \trianglelefteq G$:

Sejam $g, x \in G\times \ker \varphi$, mostremos que $gxg^{-1} \in \ker \varphi$.

\[
\varphi(gxg^{-1}) = \varphi(g) * \varphi(x) * \varphi(g^{-1}) = \varphi(g) * e_\mathcal{G} * \varphi(g^{-1}) = \varphi(g) * \varphi(g^{-1}) = \varphi(g) * \varphi(g)^{-1} = e_\mathcal{G}. 
\]

Portanto, deduzimos que:

\[
\ker \varphi \trianglelefteq G
\]

\qed

\begin{example}
Seja \( n \in \mathbb{N}^* \). O nĂşcleo do homomorfismo 
\[
\det : GL_n(\mathbb{K}) \to \mathbb{K}^*, \quad A \mapsto \det(A)
\]
ĂŠ dado por 
\[
\ker(\det) = \{A\in GL_n(\mathbb{K}),\; \det(A) = 1\} = SL_n(\mathbb{K}).
\]
\end{example}


\begin{definition}
Definimos a \textbf{imagem de um homomorfismo $\varphi$ o subconjunto $\operatorname{Im} \varphi \subseteq \mathcal{G}$ tal que:}

\[
\operatorname{Im} \varphi \overset{\text{def}}{=} \{ y \in \mathcal{G} \mid \exists x \in G,\, y = \varphi(x) \}
\]
\end{definition}

\begin{proposition}
A imagem de um homomorfismo $\varphi$ ĂŠ um subgrupo de $\mathcal{G}$.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Note, primeiramente, que $\operatorname{Im} \varphi$ ĂŠ nĂŁo vazio uma vez que $e_{\mathcal{G}}\in \operatorname{Im} \varphi$.

Mostremos agora que $\forall x, y \in \operatorname{Im} \varphi, \; xy^{-1} \in \operatorname{Im} \varphi$:

Temos que:

\[
x, y \in \operatorname{Im} \varphi \implies \exists a, b\in G, \; x = \varphi(a) \; \; \text{e} \; \; y = \varphi(b)
\]

Ora, mas entĂŁo temos:

\[
\exists b^{-1} \in G \implies y^{-1} = \varphi(b^{-1}) = \varphi(b)^{-1}
\]

TambĂŠm temos que:

\[
ab^{-1} \in G, \; \; \text{logo,} \; \; \varphi(ab^{-1}) = \varphi(a)\varphi(b)^{-1} = xy^{-1}
\]

Portanto, via a definiĂ§ĂŁo de imagem de um homomorfismo de grupos deduzimos que:

\[
xy^{-1} \in \operatorname{Im} \varphi
\]

Logo, mostramos que $\operatorname{Im} \varphi \leq \mathcal{G}$. 

\qed

\begin{proposition}
Sejam $G$ e $\mathcal{G}$ grupos tais que a funĂ§ĂŁo $\varphi : G \to \mathcal{G}$ define um homomorfismo de grupos. Logo: 
\[
\ker \varphi = \{e_G\} \iff \varphi \; \; \text{injetiva}.
\]
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos a implicaĂ§ĂŁo $(\Rightarrow)$:

Para mostrar que $\varphi$ ĂŠ injetiva, ĂŠ necessĂĄrio mostrar a seguinte implicaĂ§ĂŁo:

\[
\varphi(x) = \varphi(y) \implies x = y, \; \forall x,y \in G
\]

Suponhamos que $\varphi(x) = \varphi(y)$, logo:

\[
\varphi(x) = \varphi(y) \implies \varphi(x)\varphi(y)^{-1} = e_{\mathcal{G}} \implies \varphi(x)\varphi(y^{-1}) = e_{\mathcal{G}} \implies \varphi(xy^{-1}) = e_{\mathcal{G}}
\]

Deduzimos, portanto, que $xy^{-1} \in \ker \varphi$.

Ora, por hipĂłtese, temos que $\ker \varphi = \{e_G\}$, logo deduzimos que:

\[
x = y 
\]

Portanto, concluĂ­mos que $\varphi$ ĂŠ injetiva.

Mostremos a implicaĂ§ĂŁo $(\Leftarrow)$:

Agora, supondo que $\varphi$ ĂŠ injetiva, mostremos que $\ker \varphi = \{e_G \}$.

Para mostrar que $\ker \varphi = \{e_G\}$ ĂŠ necessĂĄrio mostrar:

\[
\text{Se} \; \; \varphi(x) = e_{\mathcal{G}} \implies x = e_G, \; \forall x \in G
\]

Sabemos que $\varphi(e_G) = e_{\mathcal{G}}$.

Suponha que $\varphi(x) = e_{\mathcal{G}}$, para $x\in G$.

PorĂŠm, sabemos que $\varphi$ ĂŠ injetiva, logo:

\[
\varphi(x) = \varphi(e_G) \implies x = e_G
\]

Logo, concluĂ­mos que:

\[
\ker \varphi = \{e_G\}
\]

\qed

\begin{proposition}
Sejam $\varphi : G \to \mathcal{G}$ e $\xi : \mathcal{G} \to \mathfrak{G}$ homomorfismos de grupos.

Logo, $(\xi \circ \varphi) : G \to \mathfrak{G}$ ĂŠ um homomorfismo de grupos. 
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Sejam $a,b \in G$.

Queremos mostrar que:

\[
(\xi \circ \varphi) (ab) = (\xi \circ \varphi) (a)(\xi \circ \varphi) (b)
\]

Sabemos que $\varphi$ e $\xi$ sĂŁo homomorfismos de grupos, portanto:

\[
(\xi \circ \varphi) (ab) = \xi(\varphi(ab)) = \xi(\varphi(a)\varphi(b)) = \xi(\varphi(a))\xi(\varphi(b)) = (\xi \circ \varphi)(a)(\xi\circ \varphi) (b)
\]

Portanto, mostramos que $(\xi \circ \varphi)$ ĂŠ de fato um homomorfismo de grupos.

\begin{definition}
Sejam $G$ e $\mathcal{G}$ grupos. Um homomorfismo $\varphi : G \to \mathcal{G}$ \textbf{ĂŠ chamado de isomorfismo de grupos se $\varphi$ ĂŠ bijetivo}. 

Caso $\varphi$ seja um isomorfismo, dizemos que $G$ e $\mathcal{G}$ sĂŁo \textbf{isomorfos} e denotamos-os como:

\[
G \cong \mathcal{G}
\]
\end{definition}

\textit{ObservaĂ§ĂŁo: note que para que dois grupos sejam isomorfos basta mostrar que existe um isomorfismo entre eles.}

\begin{example}
$(\mathbb{R}^{*}_{+}, \cdot) \cong (\mathbb{R}, +)$, onde $+$ e $\cdot$ denotam as operaĂ§Ăľes usuais de soma e produto, respectivamente, no conjunto $\mathbb{R}$.
\end{example}

\textbf{DemonstraĂ§ĂŁo}:

Seja $\varphi : \mathbb{R}^{*}_{+} \to \mathbb{R}, \; x \mapsto \log(x)$.

Mostremos que $\varphi$ ĂŠ um homomorfismo de grupos bijetivo. 

Primeiramente, note que $(\mathbb{R}_{+}^{*}, \cdot)$ e $(\mathbb{R}, +)$ sĂŁo de fato grupos.

Sabemos tambĂŠm que a funĂ§ĂŁo:

\[
\varphi(x) = \log(x), \; x\in \; ]0, \infty[\;  = \mathbb{R}_{+}^{*}
\] 

ĂŠ uma bijeĂ§ĂŁo com imagem igual Ă  $\mathbb{R}$.

Mostremos que $\varphi$ ĂŠ um homomorfismo de grupos.

Sejam $x,y\in \mathbb{R}_{+}^{*}$

\[
\varphi(xy) = \log(xy) = \log(x) + \log(y) = \varphi(x) + \varphi(y)
\]

Logo, deduzimos que $\varphi$ ĂŠ um isomorfismo de grupos e, portanto, temos:

\[
(\mathbb{R}^{*}_{+}, \cdot) \cong (\mathbb{R}, +)
\]

\qed

\begin{proposition}

Logo, $\varphi^{-1} : \mathcal{G} \to G$ ĂŠ um isomorfismo de grupos.

Logo:

\]

Portanto, temos que:

\hypertarget{primteoiso}{}
\begin{theorem}
\textbf{Primeiro Teorema dos Isomorfismos}

Sejam $G$ e $\mathcal{G}$ grupos tais que $\varphi : G \to \mathcal{G}$ ĂŠ um homomorfismo de grupos. EntĂŁo, a funĂ§ĂŁo $\psi$ tal que:

\[
\psi: 
\begin{array}{c}
G/\ker \varphi \to \operatorname{Im}\varphi \\
g\ker\varphi \mapsto \varphi(g)
\end{array}
\]

ĂŠ um isomorfismo de grupos.

Isto ĂŠ:

\[
\frac{G}{\ker \varphi} \cong \operatorname{Im} \varphi
\]
\end{theorem}

\textbf{DemonstraĂ§ĂŁo}:

Mostremos que $\psi$ ĂŠ uma funĂ§ĂŁo bem definida e que se trata de uma funĂ§ĂŁo injetiva:

Sejam $\bar{x},\bar{y}\in G/\ker\varphi$, temos:

\[
\bar{x} = \bar{y} \iff xy^{-1} \in \ker\varphi \iff \varphi(xy^{-1}) = e_{\mathcal{G}} \iff
\]

\[
\iff \varphi(x)\varphi(y)^{-1} = e_{\mathcal{G}} \iff \varphi(x) = \varphi(y) \iff \psi(\bar{x}) = \psi(\bar{y})
\]

Logo, a funĂ§ĂŁo $\psi$ estĂĄ de fato bem definida e ĂŠ injetiva.

Mostremos que $\psi$ ĂŠ uma funĂ§ĂŁo sobrejetora. 

Seja $y\in \operatorname{Im}\varphi$, entĂŁo:

\[
\exists x\in G ;\; \varphi(x) = y
\]

Temos, portanto:

\[
\psi(\bar{x}) = \varphi(x) = y
\] 

Logo, $\psi$ ĂŠ sobrejetora.

Mostremos que $\psi$ ĂŠ um homomorfismo de grupos.

Primeiramente, note que $G/\ker\varphi$ e $\operatorname{Im}\varphi$ sĂŁo de fato grupos (jĂĄ demonstrado).

Sejam $\bar{x},\bar{y} \in G/\ker\varphi$, logo:

\[
\psi(\bar{x}\bar{y}) = \psi(\overline{xy}) = \varphi(xy) = \varphi(x)\varphi(y) = \psi(\bar{x})\psi(\bar{y})
\]

Com isso, mostramos que $\psi$ se trata se um isomorfismo de grupos e deduzimos que:

\[
\frac{G}{\ker\varphi} \cong \operatorname{Im}\varphi
\]

\qed

\begin{theorem}
\textbf{Segundo Teorema dos Isomorfismos}

Seja $G$ um grupo,\; $H\trianglelefteq G$ e $K\leq G$. Temos:

\[
H \trianglelefteq HK
\]

e,

\[
\frac{K}{H\cap K} \cong \frac{HK}{H}
\]
\end{theorem}

\textbf{DemonstraĂ§ĂŁo}:

Note que $HK$ ĂŠ de fato um grupo, pois $H\trianglelefteq G$ e $K\leq G$.

Tal fato foi mostrado na \hyperlink{prop154}{\textbf{ProposiĂ§ĂŁo 1.5.4}}. 

Mostremos, primeiramente, que $H \trianglelefteq HK$:

Seja $x\in HK$, logo, ĂŠ suficiente mostrar que $xHx^{-1} \subseteq H$.

Como $H\trianglelefteq G$, temos que:

\[
gH = Hg, \; \; \forall g\in G
\]

Portanto:

\[
xHx^{-1} = Hxx^{-1} = H
\]

Logo, deduzimos que $H \trianglelefteq HK$.

Seja $\varphi$ o homomorfismo de grupos definido a seguir:

\[
\varphi: 
\begin{array}{c}
K \to \frac{HK}{H} \\
x \mapsto xH
\end{array}
\]

Note que $\frac{HK}{H}$ ĂŠ de fato um grupo, uma vez que $H\trianglelefteq HK$.

Note tambĂŠm que $\varphi$ se trata mesmo de um homomorfismo (restriĂ§ĂŁo da projeĂ§ĂŁo canĂ´nica).

Mostremos que $\varphi$ ĂŠ um homomorfismo sobrejetivo.

Tome $x\in \frac{HK}{H}$, logo:

\[
x = \underbrace{hk}_{\in HK}H = Hhk = Hk = kH = \varphi(k), \; k\in K
\]

Logo, mostramos que $\varphi$ ĂŠ sobrejetiva, isto ĂŠ:

\[
\operatorname{Im}(\varphi) = \frac{HK}{H}
\]

Mostremos agora que $\ker \varphi = H\cap K$.

Sabemos que, por definiĂ§ĂŁo:

\[
\ker \varphi = \{k\in K : \varphi(k) = H\}
\]

Deduzimos portanto que $k\in \ker\varphi \iff k\in H\cap K$.

Logo, deduzimos que $\ker \varphi = H\cap K$.

Aplicando o \hyperlink{primteoiso}{\textbf{Primeiro Teorema do Isomorfismo}} deduzimos diretamente que:

\[
\frac{K}{H\cap K} \cong \frac{HK}{H}
\]

\qed

\begin{theorem}
\textbf{Terceiro Teorema dos Isomorfismos}

Seja $G$ um grupo e $H,K \trianglelefteq G$ tal que $K\subseteq H$.

Logo, 

\[
\frac{H}{K} \trianglelefteq \frac{G}{K}
\]

e,

\[
\frac{G/K}{H/K} \cong \frac{G}{H}
\]

\end{theorem}


\textbf{DemonstraĂ§ĂŁo}:

Mostremos, primeiramente, que $H/K \trianglelefteq G/K$:

Seja $gK \in G/K$ e $hK \in H/K$, mostremos que $(gK)(hK)(gK)^{-1} \subseteq H/K$:

\[
(gK)(hK)(gK)^{-1} = gKhKKg^{-1} = gKhKg^{-1} = ghKKg^{-1} = 
\]

\[
= ghKg^{-1} = \underbrace{ghg^{-1}}_{\in H}K \in H/K 
\]

Logo, concluĂ­mos que:

\[
\frac{H}{K} \trianglelefteq \frac{G}{K}
\]

Seja a funĂ§ĂŁo $\varphi$ definida a seguir:

\[
\varphi: 
\begin{array}{c}
\frac{G}{K} \to \frac{G}{H} \\[0.3cm]
gK \mapsto gH
\end{array}
\]

Note que, como $K\subseteq H$, todo elemento de $G/K$ ĂŠ evidentemente elemento de $G/H$ e a funĂ§ĂŁo ĂŠ bem definida, sobrejetora e um homomorfismo de grupos.

Logo, deduzimos que: 

\[
\operatorname{Im} \varphi = \frac{G}{H}
\]

Mostremos que $\ker \varphi = H/K$:

Por definiĂ§ĂŁo, sabemos que:

\[
\ker \varphi = \{gK\in G/K : \varphi(gK) = H\} 
\]

Logo:

\[
gK\in \ker \varphi \iff \varphi(gK) = H \iff gH = H \iff g\in H \iff gK \subseteq H/K
\]

ConcluĂ­mos que $\ker \varphi = H/K$.

Portanto, aplicando o \hyperlink{primteoiso}{\textbf{Primeiro Teorema dos Isomorfismos}} deduzimos diretamente que:

\[
\frac{G/K}{H/K} \cong \frac{G}{H}
\]

\qed

\begin{definition}
Sejam $G$ e  $\mathcal{G}$ grupos.

\textbf{Um epimorfismo de $G$ em $\mathcal{G}$ ĂŠ um homomorfismo $\varphi$ tal que $\varphi$ ĂŠ sobrejetivo}.

Denotamos o \textbf{conjunto de todos os epimorfismos de $G$ em $\mathcal{G}$ por $\operatorname{Epi}(G, \mathcal{G})$}.
\end{definition}

\begin{definition}
Sejam $G$ e $\mathcal{G}$ grupos.

\textbf{Um monomorfismo de $G$ em $\mathcal{G}$ ĂŠ um homomorfismo $\varphi$ tal que $\varphi$ ĂŠ injetivo}.

Denotamos o \textbf{conjunto de todos os monomorfismos de $G$ em $\mathcal{G}$ por $\operatorname{Mon}(G, \mathcal{G})$}.
\end{definition}

\begin{definition}
Seja $G$ um grupo. 

\textbf{Um automorfismo de $G$ ĂŠ um isomorfismo $\varphi$} tal que:

\[
\varphi: G \to G
\]

Denotamos o \textbf{conjunto de todos os automorfismos de $G$ por $\operatorname{Aut}(G)$.}
\end{definition}

\begin{proposition}
Seja $G$ um grupo, entĂŁo $(\operatorname{Aut}(G), \circ)$ ĂŠ um grupo, onde $\circ$ denota a composiĂ§ĂŁo usual de funĂ§Ăľes.
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Note que $\operatorname{Aut}(G)$ nĂŁo ĂŠ vazio, uma vez que o isomorfismo $\operatorname{id}: G\to G, \; \operatorname{id}(g) = g$ estĂĄ em $\operatorname{Aut}(G)$.

Evidentemente a funĂ§ĂŁo $\operatorname{id}$ ĂŠ o elemento neutro de $\operatorname{Aut}(G)$.

TambĂŠm sabemos que a operaĂ§ĂŁo $\circ$ ĂŠ associativa via associatividade de composiĂ§ĂŁo usual de funĂ§Ăľes.

Pela \hyperlink{prop166}{\textbf{ProposiĂ§ĂŁo 1.6.6}}, sabemos que o homomorfismo inverso, $\varphi^{-1}$, existe e ĂŠ um isomorfismo. Logo, concluĂ­mos que $\varphi^{-1} \in \operatorname{Aut}(G)$.

Por fim, sabemos que a composiĂ§ĂŁo de funĂ§Ăľes bijetivas resulta em uma funĂ§ĂŁo bijetiva e pela \hyperlink{prop164}{\textbf{ProposiĂ§ĂŁo 1.6.4}} sabemos que a composiĂ§ĂŁo de homomorfismo de $G$ em $G$ ĂŠ um homomorfismo de $G$ em $G$.

Portanto, concluĂ­mos que $\circ$ ĂŠ uma lei de composiĂ§ĂŁo interna em $\operatorname{Aut}(G)$.

Logo, deduzimos que $(\operatorname{Aut}(G), \circ)$ ĂŠ um grupo. 

\qed

\begin{example}
A funĂ§ĂŁo:

\[
\mathcal{I}_g: 
\begin{array}{c}
G \to G \\
x \mapsto gxg^{-1}
\end{array}
\]

ĂŠ um automorfismo de $G$ denominado \textbf{automorfismo interno de $G$ associado a $g\in G$}.

O automorfismo $\mathcal{I}_g$ ĂŠ tambĂŠm chamado de \textbf{conjugaĂ§ĂŁo por $g$}.
\end{example}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos que $\mathcal{I}_g$ ĂŠ de fato um automorfismo de $G$, dado $g\in G$.

Mostremos, primeiramente, que $\mathcal{I}_g$ ĂŠ um homomorfismo de grupos.

Sabemos, trivialmente, que $\mathcal{I}_g$ ĂŠ de fato uma funĂ§ĂŁo.

Sejam $x, y \in G$, temos, por definiĂ§ĂŁo:

\[
\mathcal{I}_g(xy) = gxyg^{-1} = gx(g^{-1}g)yg^{-1} = (gxg^{-1})(gyg^{-1}) = \mathcal{I}_g(x)\mathcal{I}_g(y)
\]

Portanto, concluĂ­mos que $\mathcal{I}_g$ ĂŠ um homomorfismo de grupos. 

Mostremos que $\mathcal{I}_g$ ĂŠ uma funĂ§ĂŁo bijetiva.

Sejam $x, y \in G$:

\[
\mathcal{I}_g(x) = \mathcal{I}_g(y) \Rightarrow gxg^{-1} = gyg^{-1} \Rightarrow g^{-1}gxg^{-1}g = g^{-1}gyg^{-1}g \Rightarrow x = y
\]

Logo, $\mathcal{I}_g$ ĂŠ injetiva.

Tomemos, ainda, $y \in G$ e $x = g^{-1}yg$. Temos, portanto:

\[
\mathcal{I}_g(x) = g(g^{-1} yg)g^{-1} = y
\]

Logo, $\mathcal{I}_g$ ĂŠ sobrejetiva.

ConcluĂ­mos, portanto, que $\mathcal{I}_g$ ĂŠ um automorfismo de $G$.

\qed

\begin{definition}
Seja $G$ um grupo, definimos:

\[
\operatorname{Inn}(G) = \{\mathcal{I}_g \in \operatorname{Aut}(G) : g \in G \}
\]

o \textbf{conjunto de todos os automorfismos internos de $G$ associados Ă  $g\in G$}.
\end{definition}

\begin{proposition}
Seja $G$ um grupo, temos:

\[
\operatorname{Inn}(G) \trianglelefteq \operatorname{Aut}(G).
\]
\end{proposition}

\textbf{DemonstraĂ§ĂŁo:}

Sabemos que $\operatorname{Inn}(G) \subseteq \operatorname{Aut}(G)$, por construĂ§ĂŁo.

Mostremos primeiramente que para $\varphi_1, \varphi_2 \in \operatorname{Inn}(G), \; \varphi_1\circ \varphi_{2}^{-1}\in \operatorname{Inn}(G)$, onde $\circ$ ĂŠ a composiĂ§ĂŁo usual de funĂ§Ăľes.

Como o domĂ­nio e contradomĂ­nio das funĂ§Ăľes em $\operatorname{Inn}(G)$ sĂŁo iguais para as funĂ§Ăľes em $\operatorname{Aut}(G)$, mostremos apenas a igualdade de suas leis de correspondĂŞncia.

Ora, como $\varphi_1, \varphi_2 \in \operatorname{Inn}(G)$, podemos escrevĂŞ-los, sem perda de generalidade, para um dado $x\in G$, como:

\[
\varphi_1(x) = g_1xg_{1}^{-1},\; g_1 \in G
\]

\[
\varphi_2(x) = g_2xg_{2}^{-1}, \; g_2 \in G
\]

Temos:

\[
\varphi_1\circ \varphi_{2}^{-1} (x) = \varphi_1(\varphi^{-1}_2(x)) = g_1(g_{2}^{-1}xg_2)g_{1}^{-1} = (g_1g_{2}^{-1})x(g_2g_{1}^{-1}) = 
\]

\[
= (g_1g_{2}^{-1})x(g_1g_{2}^{-1})^{-1} \in \operatorname{Inn}(G)
\]

Logo, concluĂ­mos que $\operatorname{Inn}(G) \leq \operatorname{Aut}(G)$.

Provemos agora que para $(\varphi, \phi)\in \operatorname{Inn}(G) \times \operatorname{Aut}(G), \; \phi \circ \varphi \circ \phi^{-1} \in \operatorname{Inn}(G)$. Seja $x\in G$ tal que:

\[
\varphi(x) = gxg^{-1},\; g\in G
\]

Temos:

\[
(\phi \circ \varphi \circ \phi^{-1}) (x) = \phi(\varphi(\phi^{-1}(x))) = \phi(g\phi^{-1}(x)g^{-1}) = \phi(g)x\phi(g)^{-1} \in \operatorname{Inn}(G)
\]

Logo, concluĂ­mos que:

\[
\operatorname{Inn}(G) \trianglelefteq \operatorname{Aut}(G)
\]

\qed

\begin{definition}
Seja G um grupo e $\varphi : G \to G$ um homomorfismo de grupos. 

Dizemos que $X\subseteq G$ ĂŠ \textbf{invariante por $\varphi$} se $\varphi(X) \subseteq X$.

Ou ainda, se $\forall x \in X \implies \varphi(x) \in X$.
\end{definition}

\begin{definition}
Sejam $G$ um grupo e $H\leq G$.

$H$ ĂŠ dito \textbf{subgrupo caracterĂ­stico de $G$} se $H$ ĂŠ invariante para todo $\varphi \in \operatorname{Aut}(G)$ e o denotamos como:

\[
H \trianglechareq G
\]
\end{definition}

\textit{ObservaĂ§Ăľes:}

\begin{itemize}
	\item \textit{A partir da definiĂ§ĂŁo acima temos que, se  $H<G$ e $H$ ĂŠ subgrupo caracterĂ­stico de $G$, denotamos $H\trianglechar G$};
	\item \textit{A definiĂ§ĂŁo de subgrupo caracterĂ­stico ĂŠ considerada mais forte que a definiĂ§ĂŁo de subgrupo normal, uma vez que enquanto um subgrupo normal ĂŠ invariante apenas para automorfismos internos, um subgrupo caracterĂ­stico ĂŠ invariante para todo automorfismo de $G$}:
\end{itemize}

\begin{center}
\begin{tikzpicture}[scale=1.2]
  
  \node (subgroup) at (0,2) {$H \leq G$};
  \node (normal) at (0,0) {$H \trianglelefteq G$};
  \node (char) at (0,-2) {$H \trianglechareq G$};
  
  \draw[->, thick] (char) -- (normal);
  \draw[->, thick] (normal) -- (subgroup);
  
  \node[right] at (0.75,-2) {\footnotesize invariante por automorfismos};
  \node[right] at (0.75,0) {\footnotesize invariante por conjugaĂ§ĂŁo};
  \node[right] at (0.75,2) {\footnotesize operaĂ§ĂŁo bem definida};
  
\end{tikzpicture}
\end{center}

\begin{itemize}
	\item \textit{Temos, portanto, diretamente que $H \trianglechareq G \Longrightarrow H \trianglelefteq G$.}
\end{itemize}

\begin{example}
Os subgrupos triviais de $G$ sĂŁo subgrupos caracterĂ­sticos de $G$.
\end{example}

\begin{example}
Seja $G$ um grupo e $Z(G)$ seu centro, logo:

\[
Z(G) \trianglechareq G
\]
\end{example}

\textbf{DemonstraĂ§ĂŁo:}

Seja $z\in Z(G)$, temos por definiĂ§ĂŁo que:

\[
zg = gz, \; \forall g\in G
\]

Seja $\varphi \in \operatorname{Aut}(G)$, temos:

\[
\varphi(zg) = \varphi(gz) \iff \varphi(z)\varphi(g) = \varphi(g)\varphi(z)
\]

Como $\varphi$ ĂŠ um automorfismo, entĂŁo sabemos que se trata de uma funĂ§ĂŁo bijetora. 

Logo, $\varphi(G) = G$ e podemos escrever $\varphi(g) = h, \; h\in G$.

Temos, portanto:

\[
\varphi(z)h = h\varphi(z), \; \forall h\in G
\]

ConcluĂ­mos que $\varphi(z) \in Z(G)$.

Logo:

\[
Z(G) \trianglechareq G
\]

\qed

\begin{proposition}
Seja $K\trianglechareq H \trianglelefteq G$, entĂŁo $K\trianglelefteq G$.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $\varphi \in \operatorname{Aut}(H)$ tal que:

\[
\varphi(K) \overset{def}{=} gKg^{-1}
\]

Note que a funĂ§ĂŁo acima ĂŠ bem definida uma vez que $K\subseteq H$ e $gHg^{-1}\in H$.

Como $K \trianglechareq H$, entĂŁo $\varphi(K) \subseteq K$.

ConcluĂ­mos que:

\[
gKg^{-1} \in K \Longrightarrow K \trianglelefteq G.
\]

\qed

\textit{ObservaĂ§Ăľes: note que $K \trianglelefteq H \trianglelefteq G \nRightarrow K\trianglelefteq G$.}

\textit{Tomemos como contraexemplo a seguinte cadeia de normalidade:}

\[
\underbrace{\{e,s\}}_{\overset{def}{=} K} \trianglelefteq \underbrace{\{e, r^2, s, r^2s\}}_{\overset{def}{=} H} \trianglelefteq \underbrace{D_4}_{\overset{def}{=} G}.
\]

\textit{Onde $D_4$ ĂŠ o Grupo Diedral $D_4$, $s$ representa uma reflexĂŁo/espelhamento, $r$ representa uma rotaĂ§ĂŁo de $90^\circ$ no sentido antihorĂĄrio e $e$ representa o elemento neutro.} 


\textit{Note que de fato a cadeia de normalidade ĂŠ verificada, uma vez que:}

\begin{enumerate}
	\item \textit{Temos que $K \leq H \leq G$};
	\item \textit{$K$ ĂŠ evidentemente abeliano e portanto $K\trianglelefteq H$;}
	\item \textit{$\frac{\vert G\vert}{\vert H\vert} = 2$ e, pela \hyperlink{prop152}{\textbf{ProposiĂ§ĂŁo 1.5.2}} temos diretamente que $H \trianglelefteq G$}.
\end{enumerate}

\textit{Mostremos que o item $1.$ ĂŠ de fato verdadeiro, validando os itens $2.$ e $3.$.}

\textit{Note que $H \subseteq D_4$ e ĂŠ evidentemente nĂŁo vazio. AlĂŠm disso, cada elemento de $H$ ĂŠ seu prĂłprio elemento inverso.}

\textit{Temos tambĂŠm para os resultados nĂŁo triviais gerados a partir das operaĂ§Ăľes entre os elementos de $H$:}

\[
r^2r^2s = r^4s = s, \; sr^2 = r^2s, \; sr^2s = r^2,\; r^2sr^2 = s, \; r^2s^2 = r^2 
\]

\textit{Note que todos os resultados pertencem ao conjunto $H$}.

\textit{Portanto, $H \leq G$. Note que $K \leq H$, trivialmente.}

\textit{ConcluĂ­mos que $K \leq H \leq G$ e, pelos itens $2.$ e $3.$, $K\trianglelefteq H \trianglelefteq G$.}

\textit{Note, porĂŠm, que $K$ nĂŁo ĂŠ normal a $G$, uma vez que:}

\[
\underbrace{r}_{\in G} \cdot \underbrace{s}_{\in K} \cdot \underbrace{r^{-1}}_{\in G} = r^2s \notin K
\]

\textit{ConcluĂ­mos, portanto, via contraexemplo, que a normalidade nĂŁo ĂŠ transitiva.}

\section{Produto Direto de Grupos}

Nesta seĂ§ĂŁo inteira a variĂĄvel $n$ representa um nĂşmero inteiro positivo.

\hypertarget{def171}{}
\begin{definition}
\color{red}
Seja $\mathfrak{G} = \{G_1, \ldots, G_n \}$ uma famĂ­lia de grupos.

Definimos \textbf{produto direto externo de $\mathfrak{G}$} como sendo o conjunto $G_1 \times \ldots \times G_n$ munido da operaĂ§ĂŁo: 

\[
(x_1,\ldots, x_n)\cdot (y_1,\ldots, y_n) \overset{def}{=} (x_1y_1, \ldots, x_ny_n).
\]

Denotamos, comumente:

\[
\prod_{i=1}^{n} G_i \overset{def}{=} G_1 \times \ldots \times G_n.
\]
\end{definition}

{\color{red}
\textit{ObservaĂ§ĂŁo: por uma questĂŁo de convenĂ§ĂŁo, a operaĂ§ĂŁo "$\cdot$" serĂĄ omitida.}
}

\begin{proposition} 
\color{red}
A partir da \hyperlink{def171}{\textbf{DefiniĂ§ĂŁo 1.7.1}}, temos que o produto direto externo de $\mathfrak{G}$ ĂŠ um grupo.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Via a \hyperlink{def171}{\textbf{DefiniĂ§ĂŁo 1.7.1}} temos que a operaĂ§ĂŁo "$\cdot$" ĂŠ uma lei de composiĂ§ĂŁo interna por construĂ§ĂŁo. 

Temos tambĂŠm, de forma evidente, que o elemento neutro e o elemento inverso de $x \overset{def}{=}(x_1,\ldots,x_n)\in \prod_{i=1}^{n} G_i$ sĂŁo dados por, respectivamente:

\[
e \overset{def}{=} (e_1, \ldots, e_n) \in \prod_{i=1}^{n} G_i
\]

Onde $e_n$ representa o elemento neutro do $n$-ĂŠsimo grupo do produto direto externo.

\[
x^{-1} \overset{def}{=} (x_{1}^{-1},\ldots, x_{n}^{-1})\in \prod_{i=1}^{n} G_i
\]

Onde $x_{n}^{-1}$ representa o elemento inverso para $x_n$ no $n$-ĂŠsimo grupo do produto direto externo.

Temos tambĂŠm que o produto direto externo de $\mathfrak{G}$ ĂŠ evidentemente associativo, uma vez  que $G_1, \ldots, G_n$ sĂŁo grupos.

ConcluĂ­mos, portanto, que $(\prod_{i=1}^{n} G_i,\cdot)$ ĂŠ um grupo. 

\qed}

\hypertarget{def172}{}
\begin{definition}
\color{red}
Sejam $G$ um grupo e $H_1, \ldots, H_n \leq G$.

Dizemos que $G$ ĂŠ um \textbf{produto direto interno} de $H_1, \ldots, H_n$ se, e somente se, as seguintes condiĂ§Ăľes sĂŁo satisfeitas:

\begin{itemize}
	\item $G = H_1 \ldots H_n$;
	\item $H_i \trianglelefteq G, \; \forall i\in \llbracket 1, n \rrbracket$;
	\item $H_i \cap H_1 \ldots H_{i-1}H_{i+1}\ldots H_n = \{e\},\; \forall i \in \llbracket 2, n-1 \rrbracket$.
\end{itemize}

Caso $G$ e $H_1, \ldots, H_n$ satisfaĂ§am as condiĂ§Ăľes acima, denotamos:

\[
G \overset{def}{=} \bigodot_{i=1}^{n} H_i
\]

\end{definition}

\hypertarget{prop172}{}
\begin{proposition}
\color{red}
Sejam $G$ um grupo e $H_1, \ldots, H_n \leq G$.

Se $G$ for um produto direto interno de $H_1, \ldots, H_n$, entĂŁo quaisquer elementos de $H_i, \ldots, H_n$ distintos comutam entre si.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Sejam $(h_i, h_j) \in H_i \times H_j$, onde $i, j \in \llbracket 1, n \rrbracket$ e $i \neq j$.

Tomemos o elemento $g \overset{def}{=} [h_i, h_j] = h_ih_jh_{i}^{-1}h_{j}^{-1} \in G$.

Note que:

\[
h_ih_jh_{i}^{-1}h_{j}^{-1} = (h_ih_jh_{i}^{-1})h_{j}^{-1} \in H_j, \; \; \text{pois} \; \; H_j \trianglelefteq G
\]

\[
h_ih_jh_{i}^{-1}h_{j}^{-1} = h_i(h_jh_{i}^{-1}h_{j}^{-1}) \in H_i, \; \; \text{pois} \; \; H_i \trianglelefteq G
\]

Temos, portanto, que $[h_i, h_j] \in H_i \cap H_j = \{e\}$.

Logo:

\[
h_ih_jh_{i}^{-1}h_{j}^{-1} = e \iff h_ih_j = h_jh_i
\]

Logo, quaisquer fatores de $H_i, \ldots, H_n$ distintos comutam entre si.

\qed}

\hypertarget{teo171}{}
\begin{theorem}
\color{red}
Sejam $G$ um grupo e $H_1, \ldots, H_n \leq G$.

Temos, a partir das notaĂ§Ăľes nas definiĂ§Ăľes \hyperlink{def171}{1.7.1} e \hyperlink{def172}{1.7.2}, que:

\[
G = \bigodot_{i=1}^{n} H_i \Longrightarrow G \cong \prod_{i=1}^{n} H_i
\]
\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Suponha que $G = \bigodot_{i=1}^{n} H_i$.

Tomemos a funĂ§ĂŁo $\mu$ tal que:

\[
\mu:
\begin{array}{r c l}
  G & \to & \displaystyle \prod_{i=1}^{n} H_i \\[10pt]
  h_1 \ldots h_n & \mapsto & (h_1, \ldots, h_n)
\end{array}
\]

Note que a funĂ§ĂŁo estĂĄ de fato bem definida e ĂŠ trivialmente bijetiva: 

A sobrejetividade de $\mu$ segue diretamente da primeira condiĂ§ĂŁo da \hyperlink{def172}{\textbf{DefiniĂ§ĂŁo 1.7.2}}, pois $G = H_1 \ldots H_n$. AlĂŠm disso, a terceira condiĂ§ĂŁo da mesma definiĂ§ĂŁo garante que a representaĂ§ĂŁo de qualquer elemento $g \in G$ na forma $h_1 \ldots h_n$ ĂŠ Ăşnica. Essa unicidade assegura que $\mu$ estĂĄ bem definida e ĂŠ injetiva. Logo, $\mu$ ĂŠ uma bijeĂ§ĂŁo.

Mostremos, por fim, que $\mu$ ĂŠ um homomorfismo de grupos.

Sejam $g \overset{def}{=} h_1\ldots h_n$ e $g'\overset{def}{=} h_{1}' \ldots h_{n}'$.

Temos:

\begin{align*}
    \mu(gg') &= \mu\bigl( (h_1 \ldots h_n)(h'_{1} \ldots h'_{n}) \bigr) \\
             &= \mu(h_1 h'_{1} \ldots h_n h'_{n}) \; \; \; \text{(\hyperlink{prop172}{\textbf{ProposiĂ§ĂŁo 1.7.2}})} \\
             &= (h_1 h'_{1}, \ldots, h_n h'_{n}) \\
             &= (h_1, \ldots, h_n)(h'_{1}, \ldots, h'_{n}) \\
             &= \mu(g)\mu(g')
\end{align*}

Logo, $\mu$ ĂŠ um isomorfismo de grupos e temos que:

\[
G \cong \prod_{i=1}^{n} H_i
\]

\qed

\textit{ObservaĂ§ĂŁo: note que pela \hyperlink{prop164}{\textbf{ProposiĂ§ĂŁo 1.6.4}} e pelo fato da composiĂ§ĂŁo usual de bijeĂ§Ăľes ser uma bijeĂ§ĂŁo temos tambĂŠm diretamente que:}

\[
G \cong \bigodot_{i=1}^{n} H_i \implies G \cong \prod_{i=1}^{n} H_i
\]

}

\hypertarget{teochi}{}
\begin{theorem}
\color{red}
\textbf{Teorema ChinĂŞs do Resto}

Seja $M \overset{def}{=} m_1\ldots m_n \in \mathbb{Z}_{+}^{*}$ tal que $\text{mdc}(m_i, m_j) = 1, \; \forall i,j\in \llbracket 1,n \rrbracket$ e $i\neq j$.

EntĂŁo:

\[
\mathbb{Z}/M\mathbb{Z} \cong \prod_{i=1}^{n} \mathbb{Z}/m_i\mathbb{Z}
\]  

\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $M_i \overset{def}{=} \frac{M}{m_i}, \; \forall i\in \llbracket 1, n\rrbracket$. Note que $\overline{M_i} \in \mathbb{Z}/M\mathbb{Z}$, uma vez que $M_i \in \llbracket 1, M\rrbracket$.

Note tambĂŠm que, evidentemente, temos $\mathcal{O}(M_i) = m_i$, uma vez que $m_iM_i = M$.

A partir da \hyperlink{prop165}{\textbf{ProposiĂ§ĂŁo 1.6.5}} temos portanto $\mathbb{Z}/m_i\mathbb{Z} \cong \langle M_i \rangle \cong \langle \overline{M_i} \rangle$.  

Mostremos que $\mathbb{Z}/M\mathbb{Z} = \bigodot_{i=1}^{n} \langle \overline{M_i} \rangle$.

Ă evidente que $\langle \overline{M_i} \rangle \leq \mathbb{Z}/M\mathbb{Z}$, uma vez que, por definiĂ§ĂŁo, ele ĂŠ um subgrupo gerado. AlĂŠm disso, por estar munido da soma usual dos inteiros, tambĂŠm ĂŠ um grupo abeliano. Temos diretamente que $\langle \overline{M_i} \rangle \trianglelefteq \mathbb{Z}/M\mathbb{Z}$.

Portanto, devido a operaĂ§ĂŁo de soma usual de inteiros, denotemos:

\[
\langle \overline{M_1} \rangle \ldots \langle \overline{M_n} \rangle \overset{def}{=} \sum_{i=1}^{n} \langle \overline{M_i} \rangle
\]

Mostremos que:

\[
\mathfrak{M} \overset{def}{=} \langle \overline{M_i} \rangle \cap \sum_{\substack{j\neq i \\ j\in\llbracket 1, n\rrbracket}} \langle \overline{M_j} \rangle = \{0\}, \; \forall i\in \llbracket 1,n\rrbracket
\]

Tomemos $\overline{x}\in \mathfrak{M}$.

Seja $\overline{x}\in \langle \overline{M_i} \rangle \setminus \{0\}$, temos por definiĂ§ĂŁo (e pela coprimalidade entre $m_1,\ldots, m_n$) que $m_i \nmid x$, onde $x$ ĂŠ qualquer representante da classe lateral $\overline{x}$.

Seja $\overline{x}\in \langle \overline{M_i} \rangle \cup \mathfrak{M}$, logo $x$ ĂŠ expresso como combinaĂ§ĂŁo linear de elementos de $\langle \overline{M_1} \rangle,\ldots, \langle \overline{M_{i-1}}\rangle,\langle \overline{M_{i+1}}\rangle,\ldots,\langle \overline{M_n}\rangle$. Perceba que, por definiĂ§ĂŁo, todos os representantes de classe dos elementos de tais conjuntos sĂŁo divisĂ­veis por $m_i$, portanto $m_i \mid x$.

Como para $\overline{x}\in \mathfrak{M}$ temos $m_i \mid x$ e $m_i \nmid x$, concluĂ­mos que $\overline{x} = \overline{0}$.

Mostremos, por fim, que:

\[
\sum_{i=1}^{n} \langle \overline{M_i} \rangle = \mathbb{Z}/M\mathbb{Z}
\]

Note que pela \hyperlink{prop156}{\textbf{ProposiĂ§ĂŁo 1.5.6}} temos que:

\[
\left\vert \sum_{i=1}^{n} \langle \overline{M_i} \rangle \right\vert = \left\vert \langle \overline{M_1} \rangle + \sum_{i=2}^{n} \langle \overline{M_i} \rangle \right\vert = \frac{\vert \langle \overline{M_1} \rangle\vert \displaystyle\left\vert \sum_{i=2}^{n} \langle \overline{M_i} \rangle \right\vert}{\underbrace{\left\vert \langle \overline{M_1}\rangle \cap \sum_{i=2}^{n} \langle \overline{M_i} \rangle \right\vert}_{=1}} = m_1\left\vert \sum_{i=2}^{n}\langle \overline{M_i} \rangle \right\vert
\]

Aplicando a fĂłrmula indutivamente em $i$, temos que:

\[
\left\vert \sum_{i=1}^{n} \langle \overline{M_i}\rangle \right\vert = m_1\left\vert \sum_{i=2}^{n}\langle \overline{M_i} \rangle \right\vert = \ldots = m_1 \ldots m_n = M
\]

Como $\sum_{i=1}^{n} \langle \overline{M_i} \rangle \subseteq \mathbb{Z}/M\mathbb{Z}$, deduzimos diretamente que:

\[
\sum_{i=1}^{n} \langle \overline{M_i} \rangle = \mathbb{Z}/M\mathbb{Z}
\]

ConcluĂ­mos, finalmente, que:

\[
\mathbb{Z}/M\mathbb{Z} = \bigodot_{i=1}^{n} \langle \overline{M_i} \rangle \implies \mathbb{Z}/M\mathbb{Z} \cong \prod_{i=1}^{n} \mathbb{Z}/m_i\mathbb{Z}
\]

\qed}

\begin{definition}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio. A funĂ§ĂŁo bijetiva $\sigma : X\to X$  ĂŠ chamada de \textbf{permutaĂ§ĂŁo de $X$}.

\end{definition}

{\color{red}

\begin{definition}
\color{red}
O conjunto $\mathfrak{S}_X$ munido da operaĂ§ĂŁo usual de composiĂ§ĂŁo de funĂ§Ăľes ĂŠ chamado de \textbf{grupo de permutaĂ§Ăľes de $X$}, ou \textbf{grupo simĂŠtrico de $X$}. 
\end{definition}

{\color{red}
\textit{ObservaĂ§Ăľes:}

\begin{itemize}
	\item \textit{Note que o conjunto $\mathfrak{S}_X$ com a composiĂ§ĂŁo usual de funĂ§Ăľes forma de fato um grupo, uma vez que a composiĂ§ĂŁo de funĂ§Ăľes bijetivas retorna de fato uma funĂ§ĂŁo bijetiva, a composiĂ§ĂŁo ĂŠ associativa, toda funĂ§ĂŁo bijetiva admite uma funĂ§ĂŁo inversa e o elemento neutro ĂŠ a funĂ§ĂŁo identidade, qual denotaremos $e \overset{def}{=} \text{id}_X$}.
	\item \textit{Seja $X = \{1,\ldots, n\}$ e $\sigma\in\mathfrak{S}_n$, utilizamos comumente a notaĂ§ĂŁo a seguir para descrever a permutaĂ§ĂŁo $\sigma$:}

\[
\sigma = \begin{pmatrix}
1 & 2 & \ldots & n \\

\end{pmatrix}
\]

\end{itemize}
}

\color{red}

\begin{proposition}
\color{red}

\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Como $\sigma$ ĂŠ uma funĂ§ĂŁo bijetiva, entĂŁo para um elemento $a_i \in X$ ĂŠ possĂ­vel realizar $n$ escolhas de imagem de $a_i$ para $a_j \in X$, com $i,j\in \llbracket 1,n\rrbracket$.

Uma vez tomada essa relaĂ§ĂŁo entre os dois elementos de $X$ tomamos novamente um $a_k \in X \setminus \{a_i\}$ e escolhemos uma imagem no conjunto $X\setminus \{a_j\}$. Como $X\setminus\{a_j\}$ possui $\text{Card}(X)-1$ elementos, possuimos $n-1$ escolhas de bijeĂ§ĂŁo.

Portanto, $\vert \mathfrak{S}_X \vert = n!$.

\qed}

\hypertarget{cayley}{}
\begin{theorem}
\color{red}
\textbf{Teorema de Cayley}

Sejam $G$ um grupo finito tal que $\vert G\vert = n\in \mathbb{Z}_{+}^{*}$ e $G_0$ seu conjunto subjacente (isto ĂŠ, o conjunto $G$ sem a estrutura de grupo).

A funĂ§ĂŁo $\Upsilon$:

\[
\begin{array}{r@{\;}c@{\;}l}
\Upsilon: G & \longrightarrow & \mathfrak{S}_{G_0} \\
g & \longmapsto & \Upsilon_g: G_0 \longrightarrow G_0 \\
& & \phantom{\Upsilon_g: \ } x \longmapsto gx
\end{array}
\]
 
ĂŠ um monomorfismo de grupos. 
\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Mostremos que $\Upsilon$ ĂŠ de fato funĂ§ĂŁo e ĂŠ injetiva.

Sejam $g_1,g_2 \in G$ tais que $g_1 = g_2$, temos que:

\[
g_1 = g_2 \iff g_1x = g_2x \iff \Upsilon_{g_1}= \Upsilon_{g_2} \iff \Upsilon(g_1) = \Upsilon(g_2) 
\]

Portanto, $\Upsilon$ ĂŠ funĂ§ĂŁo injetiva.

Mostremos que $\Upsilon$ ĂŠ um homomorfismo de grupos.

Sejam $g_1, g_2 \in G$, temos que:

\[
\Upsilon(g_1g_2) = \Upsilon_{g_1g_2}(x) = g_1g_2x = g_1(g_2x) = \Upsilon_{g_1}\Upsilon_{g_2} = \Upsilon(g_1)\Upsilon(g_2)   
\]

Portanto, $\Upsilon$ ĂŠ um homomorfismo injetivo.

\qed

\begin{itemize}
	\item \textit{Note que $G$ nĂŁo ĂŠ isomorfo a $\mathfrak{S}_{G_0}$, uma vez que $\vert G\vert = n \neq \vert \mathfrak{S}_{G_0}\vert = n!$, como mostrado na \hyperlink{prop211}{\textbf{ProposiĂ§ĂŁo 2.1.1}};}
	\item \textit{Como mostramos que se trata de um homomorfismo de grupos injetivo, entĂŁo a estrutura de grupo ĂŠ preservada e ĂŠ possĂ­vel mapear cada elemento de $G$ para algum elemento de $\mathfrak{S}_{G_0}$, em outras palavras:}

\textbf{\textit{"Todo grupo finito ĂŠ isomorfo a um subgrupo de um grupo simĂŠtrico".}}

\end{itemize}

}

\hypertarget{prop212}{}
\begin{proposition}
\color{red}
Sejam $X$ um conjunto finito nĂŁo vazio e $\sigma \in \mathfrak{S}_X$.

Como $X$ ĂŠ um conjunto finito, observamos que quaisquer potĂŞncias negativas de $\sigma$ equivalem Ă  potĂŞncias positivas, pois a ordem de $\sigma$ ĂŠ finita. Portanto, o resultado vale para qualquer $n\in\mathbb{Z}$.

\hypertarget{prop213}{}
\begin{proposition}
\color{red}
Sejam $X$ um conjunto finito e nĂŁo vazio tal que $\text{Card}(X) = n\in\mathbb{Z}_{+}^{*}$ e $\tau_1,\ldots,\tau_k \in \mathfrak{S}_X$ de suporte dois a dois disjuntos, onde $k\in\llbracket 1, n\rrbracket$.

Seja $\sigma\in\mathfrak{S}_X$ tal que $\sigma = \tau_1\ldots\tau_k$, entĂŁo para $x\in X$ temos:

\[
\sigma(x) = 
\begin{cases} 
   \tau_i(x) & \text{se } x \in \text{supp}(\tau_i) \;\; \text{para } i\in \llbracket 1,k\rrbracket,\\
   x & \text{se } x \notin \text{supp}(\tau_i) \;\; \text{para } i\in \llbracket 1,k\rrbracket.
\end{cases}
\]

Temos tambĂŠm que os elementos $\tau_1,\ldots,\tau_k$ comutam entre si e:

\[
\text{supp}(\sigma) = \bigsqcup_{i=1}^{k} \text{supp}(\tau_i).
\]

\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Seja $x\in \text{supp}(\tau_i)$, para algum $i\in\llbracket 1,k\rrbracket$. Pela \hyperlink{prop212}{\textbf{ProposiĂ§ĂŁo 2.1.2}} temos que $\tau_i(x) \in \text{supp}(\tau_i)$. 

Como $\forall j\in\llbracket 1,k\rrbracket$ tal que $i\neq j$ temos que $\text{supp}(\tau_j)\cap\text{supp}(\tau_i) = \emptyset$, entĂŁo $\sigma(x) = \tau_i(x)$.

Caso $x\notin \text{supp}(\tau_i)$, para algum $i\in\llbracket 1,k\rrbracket$, temos evidentemente que $\sigma(x) = x$, uma vez que $\sigma = \tau_1\ldots\tau_k$.

Note que a composiĂ§ĂŁo entre os elementos $\tau_1,\ldots,\tau_k$ tambĂŠm comuta. 

Sejam $i,j\in\llbracket 1,k\rrbracket$ tais que $i\neq j$. Suponha que $x\in\text{supp}(\tau_i)$. Como $\text{supp}(\tau_i)\cap\text{supp}(\tau_j) = \emptyset$, temos que:

\[
\tau_i(\tau_j(x)) = \underbrace{\tau_i(x)}_{\in\text{supp}(\tau_i)} = \tau_j(\tau_i(x))
\]

Simetricamente o mesmo ĂŠ vĂĄlido para $x\in\text{supp}(\tau_j)$. Suponha, agora, que $x\notin \text{supp}(\tau_i)\sqcup\text{supp}(\tau_j)$, temos trivialmente que:

\[
\tau_i(\tau_j(x)) = \tau_i(x) = x = \tau_j(x) = \tau_j(\tau_i(x))
\]

Portanto temos que $\tau_i\tau_j=\tau_j\tau_i, \; \forall i,j\in\llbracket 1,k\rrbracket$ tais que $i\neq j$.

A partir das observaĂ§Ăľes feitas acima temos diretamente que:

\[
\text{supp}(\sigma) = \bigsqcup_{i=1}^k \text{supp}(\tau_i).
\]

\qed}

\section{Ciclos e transposiĂ§Ăľes}

\begin{definition}
\color{red}
Seja $X = \{x_1,\ldots,x_n\}$ tal que $\text{Card}(X)\geq 2$.
Uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$ ĂŠ chamada de $k$\textbf{-ciclo} (onde $k\in\llbracket 2,n\rrbracket$) se:
\begin{align*}
    &\sigma(x_1) = x_2 \\
    &\sigma(x_2) = x_3 \\
    &\dots \\
    &\sigma(x_{k-1}) = x_k \\
    &\sigma(x_k) = x_1 \\
    &\sigma(y) = y, \; y\in X\setminus\{x_1,\ldots,x_k\} 
\end{align*}

Denotamos tal $k$-ciclo como:

\[
\sigma \overset{def}{=} (x_1\;\dots\;x_k).
\]

\end{definition}

{\color{red}
\textit{ObservaĂ§Ăľes:}

\begin{itemize}
	\item \textit{A notaĂ§ĂŁo $x_1,\ldots,x_k$ utilizada nĂŁo implica que o ciclo deve atuar necessariamente nos primeiros $k$ elementos indexados do conjunto $X$. De maneira rigorosa, um $k$-ciclo ĂŠ determinado por uma escolha ordenada de $k$ elementos distintos $a_1,\ldots,a_k\in X$. A definiĂ§ĂŁo aplica-se fazendo a correspondĂŞncia $\sigma(a_i) = a_{i+1}$ (com $a_{k+1} = a_1$ e fixando todos os elementos em $X\setminus\{a_1,\ldots,a_k\}$};
	\item \textit{Note que, por definiĂ§ĂŁo, se $\sigma = (x_1\;x_2\;\ldots\;x_k)$, entĂŁo $\text{supp}(\sigma) = \{x_1\;x_2\;\ldots\;x_k\}$};
	\item \textit{Perceba que a escolha de $x_1$ como ponto de partida para o $k$-ciclo $\sigma$ ĂŠ arbitrĂĄria, ĂŠ possĂ­vel comeĂ§ar com qualquer outro $x_i\in\text{supp}(\sigma)$:}

\[
(x_1\;x_2\;\ldots\;x_k) = (x_{i}\;x_{i+1}\;\ldots\;x_k\;x_1\;\ldots\;x_{i-1});
\]

	\item \textit{Chamamos tambĂŠm um $k$-ciclo de \textbf{ciclo de largura $k$}};
	\item \textit{Caso $\sigma,\tau$ sejam ciclos com suportes dois a dois disjuntos, temos diretamente via a \hyperlink{prop213}{\textbf{ProposiĂ§ĂŁo 2.1.3}} que $\sigma\tau=\tau\sigma$};

	\item \textit{Note que nem toda permutaĂ§ĂŁo ĂŠ um ciclo. Tomemos $\sigma\in\mathfrak{S}_4$ tal que:}

\[
\sigma =
\begin{pmatrix}
1 & 2 & 3 & 4 \\
2 & 1 & 4 & 3
\end{pmatrix}
=
(1\;2)(3\;4)
\]

	\textit{Evidentemente $\sigma$ nĂŁo ĂŠ um ciclo, porĂŠm $\sigma$ pode ser escrito como produto de $2$ ciclos de suportes disjuntos}.
\end{itemize}

}

\begin{definition}
\color{red}
Uma \textbf{transposiĂ§ĂŁo} ĂŠ um $2$-ciclo.
\end{definition}

\hypertarget{prop221}{}
\begin{proposition}
\color{red}
Sejam $X =\{x_1,\ldots,x_n\}$ para $n\geq 2$. Logo, todo $k$-ciclo de $\mathfrak{S}_X$ ĂŠ igual a um produto de $k-1$ transposiĂ§Ăľes.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $\sigma\in\mathfrak{S}_X$ tal que $\sigma = \underbrace{(x_1\;x_k)(x_1\;x_{k-1})\ldots(x_1\;x_2)}_{k-1\;\;\text{transposiĂ§Ăľes}}$.

Sabemos tambĂŠm que se $x\notin\{x_1,\ldots,x_k\}$ entĂŁo $\sigma(x) = x$.

Perceba pela definiĂ§ĂŁo de $\sigma$ que $\sigma(x_1) = x_2,\;\sigma(x_2)=x_3,\;\ldots,\;\sigma(x_k)=x_1$.

Logo temos diretamente que $\sigma = (x_1\;\ldots\;x_k)$.

\qed}

\begin{example}
\color{red}
Tomemos alguns exemplos de permutaĂ§Ăľes em $\mathfrak{S}_3$:

\[
\begin{pmatrix}
1 & 2 & 3 \\
2 & 1 & 3
\end{pmatrix}
= (1\; 2) = (2\; 1)
\]

\[
\begin{pmatrix}
1 & 2 & 3 \\
1 & 3 & 2
\end{pmatrix}
= (2\; 3) = (3\; 2)
\]

\[
\begin{pmatrix}
1 & 2 & 3 \\
2 & 3 & 1
\end{pmatrix}
= (1\; 2\; 3) = (2\; 3\; 1) = (3\;1\;2) = (1\; 2)(2\; 3).
\]
\end{example}

\hypertarget{prop222}{}
\begin{proposition}
\color{red}
Sejam $X = \{x_1,\ldots,x_n\}$ para $n\geq 2$ e $\sigma\in\mathfrak{S}_X$ um $k$-ciclo, onde $k\in\llbracket 2,n\rrbracket$. EntĂŁo $\mathcal{O}(\sigma) = k$.
\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Note primeiramente que $\mathcal{O}(\sigma) \geq k$, pois $\forall i\in\llbracket 1, k-1\rrbracket$ temos $\sigma^{i}(x_1) \neq x_1$.

Perceba que $\sigma^{k-1}(x_1) = x_k \implies \sigma^{k}(x_1) = x_1$. 

Analogamente, realizamos o mesmo procedimento $\forall i\in\llbracket 2,k\rrbracket$.

Deduzimos que $\sigma^{k} = \text{id}_X \implies \mathcal{O}(\sigma) = k$.

\qed}

\hypertarget{prop223}{}
\begin{proposition}
\color{red}
Seja $X = \{x_1,\ldots,x_n\}$ para $n\geq 2$. Tomemos $\sigma\in\mathfrak{S}_X$ tal que: 

\[
\sigma = \gamma_1 \ldots \gamma_t, \; t\in\llbracket 1,n\rrbracket
\]

Onde $\gamma_i$ ĂŠ um $k_i$-ciclo, $\forall i \in \llbracket 1, t\rrbracket$ e $\gamma_i,\gamma_j$ tem suportes disjuntos $\forall i,j\in\llbracket 1,t\rrbracket, \; i\neq j$.

EntĂŁo, $\mathcal{O}(\sigma) = \text{mmc}(k_1,\ldots,k_t)$.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $h\in\mathbb{Z}_{+}^{*}$ tal que $\sigma^h = \text{id}_X$.

Note que, como $\gamma_i\gamma_j=\gamma_j\gamma_i,\;\forall i,j\in\llbracket 1, t\rrbracket$ tal que $i\neq j$, pela \hyperlink{prop213}{\textbf{ProposiĂ§ĂŁo 2.1.3}} temos:

\[
\sigma^h = (\gamma_1\ldots\gamma_t)^h = \gamma_{1}^{h}\ldots\gamma_{t}^{h} = \text{id}_X
\]

Como os suportes de cada fator $\gamma$ de $\sigma$ sĂŁo dois a dois disjuntos, ĂŠ necessĂĄrio que $h$ seja um mĂşltiplo de cada ordem $k_i$ para cada $\gamma_i, \; \forall i\in\llbracket 1,t\rrbracket$ simultĂ˘neamente.

Por definiĂ§ĂŁo, temos que $\text{min}(h) = \text{mmc}(k_1,\ldots,k_t)$.

Como $\mathcal{O}(\sigma) = \text{min}(h)$, deduzimos que $\mathcal{O}(\sigma) = \text{mmc}(k_1,\ldots,k_t)$.

\qed}

\begin{theorem}
\color{red}
Seja $X=\{x_1,\ldots,x_n\}$ para $n\geq 2$. EntĂŁo toda permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X\setminus \{\text{id}_X\}$ pode ser decomposta em um produto de ciclos de suportes dois a dois disjuntos.

Tal decomposiĂ§ĂŁo ĂŠ Ăşnica, exceto pela ordem em que os ciclos sĂŁo escritos.
\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $\sigma\in\mathfrak{S}_X\setminus\{\text{id}_X\}$.

Tomemos de forma arbitrĂĄria um elemento $x_j$, com $j\in\llbracket 1,n\rrbracket$ tal que $\sigma(x_j)\neq x_j$. Note que ĂŠ garantido que tal elemento exista, uma vez que $\sigma \neq \text{id}_X$.

Definimos o seguinte conjunto:

\[
\mathfrak{O}(x) \overset{def}{=} \{\sigma^k(x)\; ; \;k\in\mathbb{Z} \}
\]

Note que a restriĂ§ĂŁo $\sigma\upharpoonright_{\mathfrak{O}(x_j)}$ forma um $k$-ciclo e $\text{supp}(\sigma\upharpoonright_{\mathfrak{O}(x_j)}) = \mathfrak{O}(x_j)$, pela \hyperlink{prop212}{\textbf{ProposiĂ§ĂŁo 2.1.2}}.

Realizamos o mesmo procedimento para todo $x\in X\setminus \mathfrak{O}(x_j)$ tal que $\sigma(x)\neq x$ e assim sucessivamente. Como $X$ ĂŠ um conjunto finito, realizamos tal procedimento em um nĂşmero finito de passos.

Pela \hyperlink{prop213}{\textbf{ProposiĂ§ĂŁo 2.1.3}} temos que a famĂ­lia $\{\mathfrak{O}(x)\}_{x\in X}$ forma uma partiĂ§ĂŁo de $X\setminus I$, onde $I\overset{def}{=} \{ x\in X ; \sigma(x) = x \}$.

Sendo $\mathfrak{R}$ o conjunto dos representantes de cada $k_x$-ciclo $\sigma \upharpoonright_{\mathfrak{O}(x)}, \; \forall x\in X\setminus I$, temos:

\[
\text{supp}(\sigma) = \bigsqcup_{r\in \mathfrak{R}} \text{supp}(\sigma\upharpoonright_{\mathfrak{O}(r)})
\]

deduzimos que $\sigma$ ĂŠ igual a composiĂ§ĂŁo de todo $k_x$-ciclo $\sigma\upharpoonright_{\mathfrak{O}(r)}$, onde $r\in \mathfrak{R}$.

Mostremos que tal decomposiĂ§ĂŁo ĂŠ Ăşnica. 

Sejam $\sigma_1 = \rho_1\ldots \rho_h$ e $\sigma_2 = \eta_1\ldots\eta_m$ duas decomposiĂ§Ăľes em ciclos de suportes dois a dois disjuntos de $\sigma$, onde $h,m\in\llbracket 1,\lfloor\frac{n}{2}\rfloor\rrbracket$.

Perceba que, como os suportes de cada ciclo sĂŁo dois a dois disjuntos, necessariamente $\rho_i(x) = \eta_j(x)$, para todo $x\in X$ e $i,j\in \llbracket 1,\lfloor\frac{n}{2}\rfloor \rrbracket$.

Utilizando a mesma lĂłgica de forma indutiva em todos os elementos de $X$, concluĂ­mos que para todo $x\in X$ iremos obter as mesmas partiĂ§Ăľes $\mathfrak{O}(x)$ nas quais podemos escrever $\text{supp}(\sigma)$.

Como o produto entre quaisquer ciclos de $\sigma$ ĂŠ comutativo, deduzimos diretamente que $\sigma_1 = \sigma_2$.

\qed

\textit{ObservaĂ§ĂŁo: note que para calcular a ordem de uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$ basta decompor $\sigma$ em ciclos de suportes dois a dois disjuntos e calcular sua ordem via a regra vista na \hyperlink{prop223}{\textbf{ProposiĂ§ĂŁo 2.2.3}}}.

}

\hypertarget{corol221}{}
\begin{corollary}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio tal que $\text{Card}(X) = n\in \mathbb{Z}_{\geq 2}$.

EntĂŁo, todo elemento de $\mathfrak{S}_X$ ĂŠ igual a um produto de no mĂĄximo $n-1$ transposiĂ§Ăľes.
\end{corollary}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $\sigma\in\mathfrak{S}_X$ uma permutaĂ§ĂŁo tal que:

\[
\sigma = \gamma_1\ldots\gamma_m,\; m\in\llbracket 1, \left\lfloor \frac{n}{2}\right\rfloor\rrbracket 
\]

Definimos $\mathcal{O}(\gamma_i) = k_i, \; \forall i\in\llbracket 1,m\rrbracket$ (de tal forma que $\gamma_i$ ĂŠ um $k_i$-ciclo).

Sabemos a partir da \hyperlink{prop221}{\textbf{ProposiĂ§ĂŁo 2.2.1}} que cada $k$-ciclo pode ser escrito como $k-1$ transposiĂ§Ăľes (de suportes nĂŁo necessariamente disjuntos).

Como possuĂ­mos $m$ ciclos dois a dois disjuntos, temos:

\[
\sum_{i=1}^{m} k_i - 1 = \sum_{i=1}^{m} k_i - \sum_{i=1}^{m} 1 = \sum_{i=1}^{m} k_i - m
\]  

Como $\sum_{i=1}^m k_i \leq n$ e $m\in\llbracket 1,\lfloor \frac{n}{2}\rfloor\rrbracket$, $\sigma$ pode se fatorar em no mĂĄximo $n-1$ transposiĂ§Ăľes (de suportes nĂŁo necessariamente disjuntos).

\qed

\textit{ObservaĂ§ĂŁo: diferentemente do \hyperlink{teo221}{\textbf{Teorema 2.2.1}}, a decomposiĂ§ĂŁo de uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$ em transposiĂ§Ăľes nĂŁo ĂŠ necessariamente Ăşnica. SerĂĄ mostrado na \hyperlink{prop224}{\textbf{ProposiĂ§ĂŁo 2.2.4}} que uma permutaĂ§ĂŁo pode tambĂŠm ser decomposta em \textbf{transposiĂ§Ăľes elementares}.}

}

\begin{definition}
\color{red}
Seja $X = \{x_1,\ldots,x_n\}$.

Uma \textbf{transposiĂ§ĂŁo elementar} ĂŠ uma transposiĂ§ĂŁo $\tau\in\mathfrak{S}_X$ da forma:

\[
\tau = (x_i\;x_{i+1}), \; i\in\{1,\ldots, n-1\}
\]

Isto ĂŠ, uma transposiĂ§ĂŁo cujas imagens sĂŁo elementos consecutivos de $X$.
\end{definition}

\hypertarget{prop224}{}
\begin{proposition}
\color{red}
Toda transposiĂ§ĂŁo de um conjunto finito e nĂŁo vazio $X$ pode ser decomposta em transposiĂ§Ăľes elementares de $X$.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $(x_i\;x_j)$ uma transposiĂ§ĂŁo de $X = \{x_1,\ldots, x_n\}$, onde $i<j$.

Note que podemos escrever $(x_i\;x_j)$ como:

\[
(x_i\;x_j) = (x_{i}\;x_{i+1})(x_{i+1}\;x_{i+2})\ldots(x_{j-2}\;x_{j-1})(x_{j-1}\;x_j)(x_{j-2}\;x_{j-1})\ldots(x_{i+1}\;x_{i+2})(x_i\;x_{i+1})
\]

Note que de fato, ao realizar o produto das transposiĂ§Ăľes elementares acima, temos que:

\[
x_i \rightarrow x_{i+1} \rightarrow \ldots \rightarrow x_{j-1} \rightarrow x_{j}
\]

e,

\[
x_j \rightarrow x_{j-1} \rightarrow \ldots \rightarrow x_{i+1} \rightarrow x_{i}.
\]

AlĂŠm disso, temos tambĂŠm diretamente que:

\[
x_k \rightarrow x_k,\; \forall k\in \rrbracket i,j\llbracket
\]

Isto conclui a prova e mostra que de fato podemos escrever quaisquer 
transposiĂ§Ăľes de um conjunto finito e nĂŁo vazio $X$ em transposiĂ§Ăľes elementares de $X$.

\qed

\textit{ObservaĂ§ĂŁo: note que a decomposiĂ§ĂŁo de uma dada transposiĂ§ĂŁo em transposiĂ§Ăľes elementares nĂŁo ĂŠ Ăşnica. Temos, por exemplo, em $\mathfrak{S}_3$:}

\[
(1\;3) = (1\;2)(2\;3)(1\;2) = (2\;3)(1\;2)(2\;3)
\]

}

\hypertarget{prop225}{}
\begin{proposition}
\color{red}
Sejam $X = \{x_1,\ldots,x_n\}$ tal que $n\geq 2$ e $(x_i\;x_j)$ uma transposiĂ§ĂŁo arbitrĂĄria de $X$.

EntĂŁo, $\forall \sigma \in \mathfrak{S}_X$, temos:

\[
\sigma (x_i\; x_j) \sigma^{-1} = (\sigma(x_i)\; \sigma(x_j)).
\]
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $x \in X \setminus \{\sigma(x_i),\;\sigma(x_j)\}$. Temos diretamente que:

\[
\sigma^{-1}(x) = \underbrace{x'}_{x'\in X\setminus \{x_i,x_j\}} \implies \bigl((x_i\;x_j)\sigma^{-1}\bigr)(x) = x' \implies \bigl(\sigma(x_i\;x_j)\sigma^{-1}\bigr)(x) = \sigma(x') = x
\]

Tomemos agora $x = \sigma(x_i)$, temos que:

\[
\sigma^{-1}(\sigma(x_i)) = x_i \implies \bigl((x_i\;x_j)\sigma^{-1}\bigr)(\sigma(x_i)) = x_j \implies \bigl(\sigma(x_i\;x_j)\sigma^{-1}\bigr)(\sigma(x_i)) = \sigma(x_j)
\]

Analogamente temos que $x = \sigma(x_j) \implies \bigl(\sigma(x_i\;x_j)\sigma^{-1}\bigr)(\sigma(x_j)) = \sigma(x_i)$.


ConcluĂ­mos que $\sigma (x_i\; x_j) \sigma^{-1} = (\sigma(x_i)\; \sigma(x_j))$.

\qed

\textit{ObservaĂ§ĂŁo: perceba que a proposiĂ§ĂŁo vale tambĂŠm para quaisquer $r$-ciclos tal que $r\leq n$. Basta realizar de forma indutiva em $r$ o mesmo procedimento construtivo. Logo, temos:}

\[
\sigma(x_1\;x_2\;\ldots\;x_r)\sigma^{-1} = (\sigma(x_1)\;\sigma(x_2\;\ldots\;\sigma(x_r))
\]

\textit{Isto ĂŠ, quaisquer ciclos em $\mathfrak{S}_X$ sĂŁo conjugados a outro ciclo se eles tĂŞm o mesmo comprimento.}

}



\begin{theorem}
\color{red}
Seja $X = \{x_1, x_2,\ldots,x_n\}$ tal que $n \geq 3$.

EntĂŁo, o grupo $\mathfrak{S}_X$ ĂŠ gerado por $\tau = (x_1\;x_2)$ e $\gamma = (x_1\;x_2\;\ldots\;x_n)$.

Isto ĂŠ, $\forall \sigma \in \mathfrak{S}_X, \; \exists a_1,\;\ldots, a_p,\;b_1,\ldots,b_p\in \mathbb{Z}$ tal que:

\[

\]

\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

A partir do \hyperlink{corol222}{\textbf{CorolĂĄrio 2.2.2}}, basta mostrar que o produto de $\tau$ e $\gamma$ formam uma transposiĂ§ĂŁo elementar.

Ora, pela \hyperlink{prop225}{\textbf{ProposiĂ§ĂŁo 2.2.5}} temos diretamente que:

\[
\gamma^{k}\tau\gamma^{-k} = (x_{k+1}\; x_{k+2}),\; \forall k\in \llbracket 0, n-2 \rrbracket
\]

Portanto, concluĂ­mos que $\mathfrak{S}_X$ pode ser gerado a partir de apenas $\tau$ e $\gamma$.

\qed

\textit{ObservaĂ§ĂŁo: a partir do resultado acima deduzimos que $\mathfrak{S}_X$ ĂŠ um grupo dicĂ­clico.}

\begin{definition}

Sejam $X$ um conjunto ordenado, finito e nĂŁo vazio e uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$.

Chamamos de \textbf{inversĂŁo em $\sigma$} o par ordenado $(x,y)\in X^2$ tq. $x<y$ e $\sigma(x) > \sigma(y)$.

Denotamos o nĂşmero de inversĂľes de uma dada permutaĂ§ĂŁo $\sigma$ como $\mathscr{I}(\sigma)$.
\end{definition}

Seja $\sigma\in\mathfrak{S}_{9}$ tal que:

\end{example}

\begin{definition}
\color{red}
Sejam $X$ um conjunto ordenado, finito e nĂŁo vazio e uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$.

Definimos a \textbf{assinatura de $\sigma$} como o resultado da funĂ§ĂŁo $\varepsilon(\sigma) \overset{def}{=} (-1)^{\mathscr{I}(\sigma)}$.

\end{definition}

{\color{red}

\textit{ObservaĂ§ĂŁo: note que, embora a definiĂ§ĂŁo de inversĂŁo dependa de uma relaĂ§ĂŁo de ordem em $X$, a assinatura $\varepsilon(\sigma)$ ĂŠ uma propriedade intrĂ­nseca da permutaĂ§ĂŁo. Se $X$ for um conjunto finito qualquer, podemos sempre munir $X$ de uma ordem total arbitrĂĄria para calcular $\varepsilon(\sigma)$; o valor obtido serĂĄ independente da ordem escolhida.}

}

\begin{definition}
\color{red}
A partir das notaĂ§Ăľes da \hyperlink{def232}{\textbf{DefiniĂ§ĂŁo 2.3.2}}, uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$ ĂŠ dita \textbf{par} se $\varepsilon(\sigma) = 1$.

Simetricamente, uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$ ĂŠ dita \textbf{Ă­mpar} se $\varepsilon(\sigma) = -1$.
\end{definition}

\begin{example}
\color{red}
Retomando o \hyperlink{ex231}{\textbf{Exemplo 2.3.1}} temos diretamente que $\sigma$ ĂŠ uma permutaĂ§ĂŁo de assinatura igual Ă  $-1$ e ĂŠ, portanto, Ă­mpar:

\[
\varepsilon(\sigma) = (-1)^{\mathscr{I}(\sigma)} = (-1)^{17} = -1.
\]
\end{example}

\hypertarget{prop231}{}
\begin{proposition}
\color{red}
Seja $X$ um conjunto finito, nĂŁo vazio e munido de uma ordem total. Logo, toda transposiĂ§ĂŁo de $\mathfrak{S}_X$ ĂŠ uma permutaĂ§ĂŁo Ă­mpar.
\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Seja $\tau = (x\;y) \in \mathfrak{S}_X$ uma transposiĂ§ĂŁo tal que, pela ordem total de $X$, $x < y$ e $\tau(x) > \tau(y)$.

Note que, como $x<y$ e $\tau(x) > \tau(y)$, entĂŁo temos diretamente que $\forall z_x\in X_{<x},\; \tau(x) > z$.

Simetricamente, temos que $\forall z_y\in X_{>y},\; \tau(y) < z$.

Portanto, resta contar o nĂşmero de inversĂľes para $z\in\llbracket x, y\rrbracket$.

Ora, para todo $z\in\llbracket x,y\rrbracket$ temos diretamente que $\tau(y) > z, \; \forall z\in \llbracket x,y\rrbracket$.

Seja $M$ o nĂşmero de inversĂľes $(\tau(y), z)$ calculado acima. 

Note que, de forma anĂĄloga, temos que $\tau(x) < z, \; \forall z\in \llbracket x, y \rrbracket$. 

Portanto, o nĂşmero de inversĂľes dos pares $(\tau(x), z)$ ĂŠ tambĂŠm $M$. 

Entretanto, como contabilizamos duas vezes a inversĂŁo $(\tau(x),\tau(y))$, temos que $\mathscr{I}(\tau) = 2M-1$. 

Portanto, $\varepsilon(\tau) = -1$ e $\tau$ ĂŠ Ă­mpar. 

\qed}

\hypertarget{teo231}{}
\begin{theorem}
\color{red}
Sejam $X$ um conjunto finito munido de uma ordem total tal que $\text{Card}(X) = n\in\mathbb{Z}_{\geq 2}$ e uma permutaĂ§ĂŁo $\sigma\in\mathfrak{S}_X$.

Portanto, ao decompor $\sigma = \tau_1\ldots\tau_m$ para $m\in\mathbb{Z}_{+}^{*}$ onde $\tau_i$ ĂŠ uma transposiĂ§ĂŁo para todo $i\in\llbracket 1,m\rrbracket$, temos que:

\[
\varepsilon(\sigma) = (-1)^m
\]

\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Note, primeiramente, que podemos supor sem perda de generalidade que $\sigma$ ĂŠ decomposto em um nĂşmero finito $p\in\mathbb{Z}_{\geq m}$ de transposiĂ§Ăľes elementares. 

Isso se dĂĄ pois, pela \hyperlink{prop224}{\textbf{ProposiĂ§ĂŁo 2.2.4}}, temos diretamente via a construĂ§ĂŁo da demonstraĂ§ĂŁo que uma dada transposiĂ§ĂŁo se decompĂľe em um nĂşmero Ă­mpar de transposiĂ§Ăľes elementares. Portanto, $(-1)^m = (-1)^p$.

Sejam $\sigma' \in \mathfrak{S}_X$ uma permutaĂ§ĂŁo arbitrĂĄria e $\tau' = (x_d\; x_{d+1})\in \mathfrak{S}_X$ uma transposiĂ§ĂŁo elementar tal que $x_d < x_{d+1}$ via a ordem total de $X$.

Note que, ao tomar sem perda de generalidade o produto $\sigma'\tau'$, a transposiĂ§ĂŁo $\tau'$ atua trocando os elementos apenas nas posiĂ§Ăľes adjacentes $d$ e $d+1$. Consequentemente, a quantidade total de inversĂľes envolvendo qualquer outro par de posiĂ§Ăľes permanece idĂŞntica. Temos, portanto, apenas o par $(x_d, x_{d+1})$ a analisar:

Suponha que $\sigma'\tau'(x_d) > \sigma'\tau'(x_{d+1})$. Logo, a aplicaĂ§ĂŁo de $\tau'$ gerou uma nova inversĂŁo, de modo que $\mathscr{I}(\sigma'\tau') = \mathscr{I}(\sigma') + 1$.

Suponha, agora, que $\sigma'\tau'(x_d) < \sigma'\tau'(x_{d+1})$. Logo, simetricamente, $\mathscr{I}(\sigma'\tau') = \mathscr{I}(\sigma') - 1$.

Logo, concluĂ­mos que $\varepsilon(\sigma'\tau') = -\varepsilon(\sigma')$.

Como $\sigma = \tau_{1}'\ldots \tau_{p}'$ onde $\tau_{i}'$ ĂŠ transposiĂ§ĂŁo elementar para todo $i\in\llbracket 1,p\rrbracket$, temos diretamente que:

\[
\varepsilon(\sigma) = (-1)^p = (-1)^m
\]

\qed}

A partir do \hyperlink{teo231}{\textbf{Teorema 2.3.1}} podemos enunciar a seguinte definiĂ§ĂŁo equivalente para o conceito de \textbf{assinatura de uma permutaĂ§ĂŁo}:

\begin{definition}
\color{red}
Sejam $X$ um conjunto finito tal que $\text{Card}(X)\geq 2$ e $\sigma\in\mathfrak{S}_X$ uma permutaĂ§ĂŁo.
 
EntĂŁo, se $\sigma = \tau_1\ldots\tau_m$ ĂŠ a decomposiĂ§ĂŁo de $\sigma$ em transposiĂ§Ăľes de $\mathfrak{S}_X$, onde $m\in\mathbb{Z}_{+}^{*}$, definimos a \textbf{assinatura de $\sigma$} o valor denotado $\varepsilon(\sigma)$ dado por:

\[

\]

\end{definition}

{\color{red}
\textit{ObservaĂ§ĂŁo: note que a definiĂ§ĂŁo equivale a afirmar que uma permutaĂ§ĂŁo ĂŠ par se ĂŠ composta por um nĂşmero par de transposiĂ§Ăľes e que uma permutaĂ§ĂŁo ĂŠ Ă­mpar se ĂŠ composta por um nĂşmero Ă­mpar de transposiĂ§Ăľes.}

}

\begin{theorem}
\color{red}
Sejam $X$ um conjunto finito com $\text{Card}(X)\geq 2$ e a \textbf{funĂ§ĂŁo assinatura $\varepsilon$} tal que:

\[
\varepsilon: 
\begin{array}{r c l}
	\mathfrak{S}_X &\to &\{-1,1\} \\
	\sigma &\mapsto &\varepsilon (\sigma)
\end{array}
\]

Logo, $\varepsilon$ ĂŠ um epimorfismo de grupos.
\end{theorem}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Sejam $\sigma,\sigma' \in\mathfrak{S}_X$ tais que suas decomposiĂ§Ăľes em transposiĂ§Ăľes de $\mathfrak{S}_X$ sĂŁo dadas por:

\[
\sigma = \tau_1 \ldots \tau_p,\; p\in\mathbb{Z}_{+}^{*}
\]

e, 

\[
\sigma' = \tau_{1}' \ldots \tau_{q}',\; q\in\mathbb{Z}_{+}^{*}
\]

Logo, temos que:

\[
\varepsilon(\sigma\sigma') = (-1)^{p+q} = (-1)^p(-1)^q = \varepsilon(\sigma)\varepsilon(\sigma').
\]

Como $\varepsilon(\text{id}_X) = 1$ e $\varepsilon(\tau) = -1$ para qualquer $\tau\in\mathfrak{S}_X$ (cuja existĂŞncia ĂŠ garantida, uma vez que $\text{Card}(X)\geq 2$), entĂŁo concluĂ­mos que $\varepsilon$ ĂŠ um epimorfismo de grupos.

\qed

\textit{ObservaĂ§ĂŁo: note que $\forall\sigma,\tau\in\mathfrak{S}_X$ tal que $\tau$ ĂŠ uma transposiĂ§ĂŁo, temos que $\varepsilon(\sigma\tau)=\varepsilon(\tau\sigma)=-\varepsilon(\sigma)$}.

}

\section{Grupo Alternado $\mathfrak{A}_X$}

\begin{definition}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio. Definimos o conjunto $\mathfrak{A}_X$ como o \textbf{conjunto alternado de $\mathfrak{S}_X$}, tal que:

\[
\mathfrak{A}_X \overset{def}{=} \{\sigma\in\mathfrak{S}_X\; ; \; \varepsilon(\sigma) = 1\}.
\]

Onde $\varepsilon$ denota a funĂ§ĂŁo assinatura.
\end{definition}

\begin{proposition}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio, entĂŁo: 

\[
\mathfrak{A}_X \trianglelefteq \mathfrak{S}_X
\]
\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Sabemos que a funĂ§ĂŁo assinatura $\varepsilon$ ĂŠ um epimorfismo de grupos. Por definiĂ§ĂŁo, sabemos que o nĂşcleo de $\varepsilon$ ĂŠ definido por:

\[
\ker{\varepsilon} \overset{def}{=} \{\sigma\in\mathfrak{S}_X \; ; \; \varepsilon(\sigma)=1 \} = \mathfrak{A}_X
\]

Como $\ker{\varepsilon}\trianglelefteq \mathfrak{S}_X$, entĂŁo temos diretamente que:

\[
\mathfrak{A}_X \trianglelefteq \mathfrak{S}_X
\]

\qed}

\begin{example}
\color{red}
Enumeremos todos os elementos de $\mathfrak{S}_3$:

\[
(1\;2\;3),\;(1\;3\;2),\;(1\;2),\;(1\;3),\;(2\;3),\;\text{id}_{\{1,2,3\}}
\]

Logo, fica claro que os elementos de $\mathfrak{A}_3$ sĂŁo:

\[
(1\;2\;3),\;(1\;3\;2),\;\text{id}_{\{1,2,3\}}.
\]

\end{example}

\begin{proposition}
\color{red}
Seja $X$ um conjunto finito tal que $\text{Card}(X) = n\in\mathbb{Z}_{\geq 2}$. Logo:

\[
\vert \mathfrak{A}_X \vert = \frac{n!}{2}.
\]
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Seja $\sigma,\tau \in \mathfrak{S}_X$, tal que $\tau$ ĂŠ uma transposiĂ§ĂŁo fixada.

Definimos a funĂ§ĂŁo $\varphi$ de forma que:

\[
\varphi: 
\begin{array}{r c l}
	\mathfrak{S}_X &\to &\mathfrak{S}_X \\
	\sigma &\mapsto &\sigma\tau
\end{array}
\]

Note que $\varphi$ ĂŠ de fato uma funĂ§ĂŁo e ĂŠ bijetiva. Sejam $\sigma_1,\sigma_2\in\mathfrak{S}_X$ tal que $\sigma_1=\sigma_2$, temos que:

\[
\sigma_1 = \sigma_2 \iff \sigma_1 \tau = \sigma_2 \tau \iff \varphi(\sigma_1) = \varphi(\sigma_2)
\]

Como a funĂ§ĂŁo $\varphi$ ĂŠ injetiva e tem como domĂ­nio e contradomĂ­nio o mesmo conjunto $\mathfrak{S}_X$, deduzimos que $\varphi$ ĂŠ de fato uma bijeĂ§ĂŁo.

Sabemos, tambĂŠm, que $\varepsilon(\sigma\tau) = -\varepsilon(\sigma)$ para qualquer $\sigma\in\mathfrak{S}_X$. Portanto, deduzimos diretamente que $\varphi(\mathfrak{A}_X) = \mathfrak{S}_X \setminus \mathfrak{A}_X$.

Como a funĂ§ĂŁo $\varphi$ ĂŠ bijetiva, entĂŁo tambĂŠm temos que $\vert \mathfrak{A}_X\vert = \text{Card}(\mathfrak{S}_X\setminus \mathfrak{A}_X)$.

Ă evidente ainda que $\mathfrak{S}_X = \mathfrak{A}_X \sqcup \mathfrak{S}_X\setminus\mathfrak{A}_X$, logo:

\[
\vert \mathfrak{S}_X \vert = \vert \mathfrak{A}_X\vert + \text{Card}(\mathfrak{S}_X\setminus\mathfrak{A}_X) = 2\mathfrak{A}_X \iff \vert \mathfrak{A}_X \vert = \frac{n!}{2}
\]

\qed

\textit{ObservaĂ§ĂŁo: note que para o caso em que $\text{Card}(X) = 1$ temos evidentemente que $\mathfrak{A}_X = \{\text{id}_X\}\implies \vert \mathfrak{A}_X\vert = 1$}.

}

\begin{proposition}
\color{red}

Sejam $X$ um conjunto finito e nĂŁo vazio e $x,y,z\in X$, entĂŁo:

\[
\mathfrak{A}_X = \langle(x\;y\;z)\rangle
\]

Isto ĂŠ, o grupo alternado $\mathfrak{A}_X$ ĂŠ gerado por $3$-ciclos de $\mathfrak{S}_X$.
\end{proposition}

{\color{red}

\textbf{DemonstraĂ§ĂŁo:}

Mostremos, primeiramente, que $\langle (x\;y\;z)\rangle\subseteq \mathfrak{A}_X$.

Note que $(x\;y\;z)^3 = \text{id}_X \in \mathfrak{A}_X$.

Pela \hyperlink{prop232}{\textbf{ProposiĂ§ĂŁo 2.3.2}} temos ainda que $\varepsilon\left((x\;y\;z)^k\right) = 1,\; \forall k\in\mathbb{Z}$. 

Portanto, temos diretamente que $\langle (x\;y\;z)\rangle \subseteq \mathfrak{A}_X$.

Mostremos que $\mathfrak{A}_X \subseteq \langle(x\;y\;z)\rangle$.

Note, primeiramente, que $\text{id}_X = (x\;y\;z)^3 \in \langle(x\;y\;z)\rangle$.

Perceba tambĂŠm que qualquer par de transposiĂ§Ăľes de $\mathfrak{S}_X$ pode ser escrito como um $3$-ciclo. Sejam $a,b,c\in X$, temos que:

\[
(a\;b)(b\;c) = (a\;b\;c)
\]

Ora, temos ainda que $\forall \sigma \in \mathfrak{A}_X$, $\sigma$ pode ser decomposto em transposiĂ§Ăľes. Para $\sigma \in \mathfrak{A}_X$, tal decomposiĂ§ĂŁo ĂŠ obrigatoriamente realizada em pares de transposiĂ§Ăľes, uma vez que, caso tenhamos um nĂşmero Ă­mpar de transposiĂ§Ăľes, $\varepsilon(\sigma) = -1$.

Como qualquer par de transposiĂ§Ăľes pode ser escrito como um $3$-ciclo, entĂŁo $\mathfrak{A}_X \subseteq \langle (x\;y\;z)\rangle$.

ConcluĂ­mos que:

\[
\mathfrak{A}_X = \langle(x\;y\;z)\rangle.
\]

\qed}


\begin{proposition}
\color{red}
Seja $X$ um conjunto finito e nĂŁo vazio. Logo, se $H \leq \mathfrak{S}_X$, entĂŁo ou $H\leq \mathfrak{A}_X$ ou $[H : H\cap \mathfrak{A}_X] = 2$.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Note, primeiramente, que nĂŁo existe subgrupo $H\leq \mathfrak{S}_X$ tal que $H$ ĂŠ composto exclusivamente de permutaĂ§Ăľes Ă­mpares. Ora, para quaisquer permutaĂ§Ăľes $\sigma_1,\sigma_2\in H$ terĂ­amos $\varepsilon(\sigma_1\sigma_2) = 1 \implies \sigma_1\sigma_2\notin H$.

Logo, podemos somente tomar os subgrupos $H\leq\mathfrak{A}_X\trianglelefteq \mathfrak{S}_X$ ou $H\cap\mathfrak{A}_X \leq \mathfrak{S}_X$.

A respeito do subgrupo $H\cap\mathfrak{A}_X$, consideremos a funĂ§ĂŁo assinatura $\varepsilon: H\to\{-1,1\}$. Temos diretamente que $\ker{\varepsilon} = H\cap\mathfrak{A}_X$.

Ora, pelo \hyperlink{primteoiso}{\textbf{Primeiro Teorema dos Isomorfismos}} temos portanto:

\[
\frac{H}{H\cap\mathfrak{A}_X} \cong \{-1,1\}
\]

Como $\text{Card}(\{-1,1\}) = 2$, entĂŁo deduzimos que $[H:H\cap\mathfrak{A}_X] = 2$.

\qed

\textit{ObservaĂ§ĂŁo: isto significa que para quaisquer subgrupos $H$ de $\mathfrak{S}_X$, entĂŁo ou $H$ ĂŠ composto de permutaĂ§Ăľes pares ou $H$ possui o mesmo nĂşmero de permutaĂ§Ăľes pares e permutaĂ§Ăľes Ă­mpares.}

}


\hypertarget{prop245}{}
\begin{proposition}
\color{red}
Seja $X$ um conjunto finito tal que $\text{Card}(X) \geq 5$. Logo, todos $3$-ciclos sĂŁo dois a dois conjugados em $\mathfrak{A}_X$. 
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Sejam $x,y,z\in X$ e $\sigma\in\mathfrak{S}_X$ uma permutaĂ§ĂŁo arbitrĂĄria.

Sabemos, da \hyperlink{prop225}{\textbf{ProposiĂ§ĂŁo 2.2.5}}, que quaisquer ciclos sĂŁo conjugados entre si em $\mathfrak{S}_X$ se possuem mesmo comprimento. Mostremos que para os $3$-ciclos essa conjugaĂ§ĂŁo permanece em $\mathfrak{A}_X$. Isto ĂŠ:

\[
\varepsilon(\sigma(x\;y\;z)\sigma^{-1}) = 1
\]

Note que, se $\sigma$ for par, entĂŁo estĂĄ demonstrado. Suponha que $\sigma$ ĂŠ Ă­mpar, logo, podemos tomar uma transposiĂ§ĂŁo $(a\;b),\;a,b\in X\setminus \{x,y,z\}$ da decomposiĂ§ĂŁo em transposiĂ§Ăľes de $\sigma$. A existĂŞncia de $a,b$ ĂŠ garantida pois $\text{Card}(X)\geq 5$.

Tomemos $\sigma \overset{def}{=} \sigma'(a\;b)$. Temos que:

\[
	\sigma(x\;y\;z)\sigma^{-1} = \sigma'(a\;b)(x\;y\;z)(\sigma'(a\;b))^{-1} = \sigma'(a\;b)(x\;y\;z)(a\;b)\sigma'^{-1} = \sigma'(x\;y\;z)\sigma'^{-1}
\]

Como $\sigma'$ ĂŠ claramente par, entĂŁo concluĂ­mos que $\sigma(x\;y\;z)\sigma^{-1}$ ĂŠ par para $\sigma$ Ă­mpar quando $\text{Card}(X) \geq 5$.

\qed}

\begin{theorem}

\begin{definition}

\end{definition}

\textit{ObservaĂ§ĂŁo: note que de fato $\mathfrak{K}\subseteq\mathfrak{A}_X$, uma vez que $\varepsilon(\sigma) = 1, \;\forall \sigma\in\mathfrak{K}$.}

\begin{proposition}
\color{red}
Seja $\mathfrak{K}$ o conjunto simĂŠtrico de Klein e $\mathcal{K}$ o grupo de Klein. 

EntĂŁo, se $\mathfrak{K}$ for munido da composiĂ§ĂŁo usual de funĂ§Ăľes, temos:

\[
\mathfrak{K} \cong \mathcal{K}
\]

Isto ĂŠ, $\mathfrak{K}$ ĂŠ isomorfo ao grupo de Klein.
\end{proposition}

{\color{red}
\textbf{DemonstraĂ§ĂŁo:}

Sejam $X = \{x_1,x_2,x_3,x_4\}$ e $\varphi : \mathfrak{K} \to \mathcal{K}$ uma funĂ§ĂŁo tal que:

\begin{align*}
\varphi(\text{id}_X) &= (\overline{0},\overline{0}) \\
\varphi(\kappa_1) &= (\overline{1},\overline{0}) \\
\varphi(\kappa_2) &= (\overline{0},\overline{1}) \\
\varphi(\kappa_3) &= (\overline{1},\overline{1})
\end{align*}

Note que $\varphi$ ĂŠ bijetiva por construĂ§ĂŁo. Mostremos que se trata de um homomorfismo de grupos.

Perceba, primeiramente, que: 

\[
\varphi(\kappa_i\text{id}_X) = \varphi(\text{id}_X\kappa_i) = \varphi(\kappa_i) + \underbrace{(\overline{0},\overline{0})}_{\varphi(\text{id}_X)} = \underbrace{(\overline{0},\overline{0})}_{\varphi(\text{id}_X)} + \varphi(\kappa_i),\;\forall i\in\llbracket 1,3\rrbracket
\]

Da mesma forma, temos diretamente que:

\[
\varphi(\kappa_i\kappa_i) = \varphi(\text{id}_X) = \varphi(\kappa_i) + \varphi(\kappa_i),\;\forall i\in\llbracket 1,3\rrbracket 
\]

Tomando agora o produto de cada elemento distinto e diferente do neutro de $\mathfrak{K}$, temos que:

\[
\varphi(\kappa_1\kappa_2) = \varphi(\kappa_3) = (\overline{1},\overline{1}) = (\overline{1},\overline{0}) + (\overline{0},\overline{1}) = \varphi(\kappa_1) + \varphi(\kappa_2)
\]

\[
\varphi(\kappa_1\kappa_3) = \varphi(\kappa_2) = (\overline{0},\overline{1}) = (\overline{1}+\overline{0})+(\overline{1}+\overline{1}) = \varphi(\kappa_1) + \varphi(\kappa_3)
\]

\[
\varphi(\kappa_2\kappa_3) = \varphi(\kappa_1) = (\overline{1},\overline{0}) = (\overline{0},\overline{1})+(\overline{1},\overline{1}) = \varphi(\kappa_2) + \varphi(\kappa_3)
\]

Ora, de forma anĂĄloga, temos que $\varphi(\kappa_1\kappa_2) = \varphi(\kappa_2\kappa_1),\;\varphi(\kappa_1\kappa_3)=\varphi(\kappa_3\kappa_1),\;\varphi(\kappa_2\kappa_3)=\varphi(\kappa_3\kappa_2)$. Como a soma usual de inteiro ĂŠ abeliana em $\mathcal{K}$, concluĂ­mos que todas as imagens preservam, portanto, a estrutura de grupo.

Finalmente, concluĂ­mos que $\varphi$ ĂŠ um isomorfismo de grupos e temos:

\[
\mathfrak{K} \cong \mathcal{K}
\]

\qed

	\item \textit{Temos diretamente via o isomorfismo que $\mathfrak{K}$ ĂŠ abeliano e $\mathfrak{K}\trianglelefteq \mathfrak{A}_X$}.
